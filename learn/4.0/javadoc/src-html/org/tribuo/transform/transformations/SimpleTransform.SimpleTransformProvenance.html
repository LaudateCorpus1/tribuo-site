036/**
037 * This is used for stateless functions such as exp, log, addition or multiplication by a constant.
038 * <p>
039 * It's a Transformation, Transformer and TransformStatistics as it has
040 * no data dependent state. This means a single Transformer can be
041 * used for every feature in a dataset.
042 * <p>
043 * Wraps a {@link DoubleUnaryOperator} which actually performs the
044 * transformation.
045 */
046public final class SimpleTransform implements Transformer, Transformation, TransformStatistics {
047    private static final long serialVersionUID = 1L;
048
049    private static final String OP = "op";
050    private static final String OPERAND = "operand";
051    private static final String SECOND_OPERAND = "secondOperand";
052
053    public static final double EPSILON = 1e-12;
054
055    /**
056     * Operations understood by this Transformation.
057     */
058    public enum Operation {
059        /**
060         * Exponentiates the inputs
061         */
062        exp,
063        /**
064         * Logs the inputs (base_e)
065         */
066        log,
067        /**
068         * Adds the specified constant.
069         */
070        add,
071        /**
072         * Subtracts the specified constant.
073         */
074        sub,
075        /**
076         * Multiplies by the specified constant.
077         */
078        mul,
079        /**
080         * Divides by the specified constant.
081         */
082        div,
083        /**
084         * Binarises the output around 1.0.
085         */
086        binarise,
087        /**
088         * Min and max thresholds applied to the input.
089         */
090        threshold
091    }
092
093    @Config(mandatory = true,description="Type of the simple transformation.")
094    private Operation op;
095
096    @Config(description="Operand (if required).")
097    private double operand = Double.NaN;
098
099    @Config(description="Second operand (if required).")
100    private double secondOperand = Double.NaN;
101
102    private SerializableDoubleUnaryOperator operation;
103
104    private transient TransformationProvenance provenance;
105
106    /**
107     * For OLCUT.
108     */
109    private SimpleTransform() {}
110
111    private SimpleTransform(Operation op, double operand, double secondOperand) {
112        this.op = op;
113        this.operand = operand;
114        this.secondOperand = secondOperand;
115        postConfig();
116    }
117
118    private SimpleTransform(Operation op, double operand) {
119        this.op = op;
120        this.operand = operand;
121        postConfig();
122    }
123
124    private SimpleTransform(Operation op) {
125        this.op = op;
126        postConfig();
127    }
128
129    @Override
130    public void postConfig() {
131        switch (op) {
132            case exp:
133                operation = Math::exp;
134                break;
135            case log:
136                operation = Math::log;
137                break;
138            case add:
139                if (Double.isNaN(operand)) {
140                    throw new IllegalArgumentException("operand must not be NaN");
141                }
142                operation = (double input) -> input + operand;
143                break;
144            case sub:
145                if (Double.isNaN(operand)) {
146                    throw new IllegalArgumentException("operand must not be NaN");
147                }
148                operation = (double input) -> input - operand;
149                break;
150            case mul:
151                if (Double.isNaN(operand)) {
152                    throw new IllegalArgumentException("operand must not be NaN");
153                }
154                operation = (double input) -> input * operand;
155                break;
156            case div:
157                if (Double.isNaN(operand)) {
158                    throw new IllegalArgumentException("operand must not be NaN");
159                }
160                operation = (double input) -> input / operand;
161                break;
162            case binarise:
163                operation = (double input) -> input < EPSILON ? 0.0 : 1.0;
164                break;
165            case threshold:
166                if (operand > secondOperand) {
167                    throw new IllegalArgumentException("Min must be greater than max, min = " + operand + ", max = " + secondOperand);
168                } else if (Double.isNaN(operand) || Double.isNaN(secondOperand)) {
169                    throw new IllegalArgumentException("min and/or max must not be NaN");
170                }
171                operation = (double input) -> { if (input < operand) { return operand; } else if (input > secondOperand) { return secondOperand; } else { return input; } };
172                break;
173            default:
174                throw new IllegalArgumentException("Operation " + op + " is unknown");
175        }
176    }
177
178    @Override
179    public TransformationProvenance getProvenance() {
180        if (provenance == null) {
181            provenance = new SimpleTransformProvenance(this);
182        }
183        return provenance;
184    }
185
186    public final static class SimpleTransformProvenance implements TransformationProvenance {
187        private static final long serialVersionUID = 1L;
188
189        private final EnumProvenance<Operation> op;
190        private final DoubleProvenance operand;
191        private final DoubleProvenance secondOperand;
192
193        SimpleTransformProvenance(SimpleTransform host) {
194            this.op = new EnumProvenance<>(OP,host.op);
195            this.operand = new DoubleProvenance(OPERAND,host.operand);
196            this.secondOperand = new DoubleProvenance(SECOND_OPERAND,host.secondOperand);
197        }
198
199        @SuppressWarnings("unchecked") // Enum cast
200        public SimpleTransformProvenance(Map<String,Provenance> map) {
201            op = ObjectProvenance.checkAndExtractProvenance(map,OP,EnumProvenance.class, SimpleTransformProvenance.class.getSimpleName());
202            operand = ObjectProvenance.checkAndExtractProvenance(map,OPERAND,DoubleProvenance.class, SimpleTransformProvenance.class.getSimpleName());
203            secondOperand = ObjectProvenance.checkAndExtractProvenance(map,SECOND_OPERAND,DoubleProvenance.class,SimpleTransformProvenance.class.getSimpleName());
204        }
205
206        @Override
207        public String getClassName() {
208            return SimpleTransform.class.getName();
209        }
210
211        @Override
212        public boolean equals(Object o) {
213            if (this == o) return true;
214            if (!(o instanceof SimpleTransformProvenance)) return false;
215            SimpleTransformProvenance pairs = (SimpleTransformProvenance) o;
216            return op.equals(pairs.op) &&
217                    operand.equals(pairs.operand) &&
218                    secondOperand.equals(pairs.secondOperand);
219        }
220
221        @Override
222        public int hashCode() {
223            return Objects.hash(op, operand, secondOperand);
224        }
225
226        @Override
227        public Map<String, Provenance> getConfiguredParameters() {
228            Map<String,Provenance> map = new HashMap<>();
229            map.put(OP,op);
230            map.put(OPERAND,operand);
231            map.put(SECOND_OPERAND,secondOperand);
232            return Collections.unmodifiableMap(map);
233        }
234    }

273    /**
274     * Apply the operation to the input.
275     * @param input The input value to transform.
276     * @return The transformed value.
277     */
278    @Override
279    public double transform(double input) {
280        return operation.applyAsDouble(input);
281    }
282
283    @Override
284    public String toString() {
285        switch (op) {
286            case exp:
287                return "exp()";
288            case log:
289                return "log()";
290            case add:
291                return "add("+operand+")";
292            case sub:
293                return "sub("+operand+")";
294            case mul:
295                return "mul("+operand+")";
296            case div:
297                return "div("+operand+")";
298            case binarise:
299                return "binarise()";
300            case threshold:
301                return "threshold(min="+operand+",max="+secondOperand+")";
302            default:
303                return op.toString();
304        }
305    }

309     * {@link Math#exp}.

313        return new SimpleTransform(Operation.exp);
314    }
315
316    /**
317     * Generate a SimpleTransform that applies
318     * {@link Math#log}.
319     * @return The logarithm function.
320     */
321    public static SimpleTransform log() {
322        return new SimpleTransform(Operation.log);
323    }

325    /**
326     * Generate a SimpleTransform that
327     * adds the operand to each value.
328     * @param operand The operand to add.
329     * @return An addition function.
330     */
331    public static SimpleTransform add(double operand) {
332        return new SimpleTransform(Operation.add,operand);
333    }
334
335    /**
336     * Generate a SimpleTransform that
337     * subtracts the operand from each value.
338     * @param operand The operand to subtract.
339     * @return A subtraction function.
340     */
341    public static SimpleTransform sub(double operand) {
342        return new SimpleTransform(Operation.sub,operand);
343    }
344
345    /**
346     * Generate a SimpleTransform that
347     * multiplies each value by the operand.
348     * @param operand The operand to multiply.
349     * @return A multiplication function.
350     */
351    public static SimpleTransform mul(double operand) {
352        return new SimpleTransform(Operation.mul,operand);
353    }
354
355    /**
356     * Generate a SimpleTransform that
357     * divides each value by the operand.
358     * @param operand The divisor.
359     * @return A division function.
360     */
361    public static SimpleTransform div(double operand) {
362        return new SimpleTransform(Operation.div,operand);
363    }
364
365    /**
366     * Generate a SimpleTransform that sets negative and
367     * zero values to zero and positive values to one.
368     * @return A binarising function.
369     */
370    public static SimpleTransform binarise() {
371        return new SimpleTransform(Operation.binarise);
372    }
373
374    /**
375     * Generate a SimpleTransform that sets values below min to
376     * min, and values above max to max.
377     * @param min The minimum value. To not threshold below, set to {@link Double#NEGATIVE_INFINITY}.
378     * @param max The maximum value. To not threshold above, set to {@link Double#POSITIVE_INFINITY}.
379     * @return A thresholding function.
380     */
381    public static SimpleTransform threshold(double min, double max) {
382        return new SimpleTransform(Operation.threshold,min,max);
383    }
