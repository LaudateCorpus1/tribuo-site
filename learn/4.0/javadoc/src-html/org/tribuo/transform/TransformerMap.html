041/**
042 * A collection of {@link Transformer}s which can be applied to a {@link Dataset}
043 * or {@link Example}. A TransformerMap is created by applying a {@link TransformationMap}
044 * to a Dataset. It contains Transformers which are specific to the Dataset which created
045 * it, for example the range of a feature used in binning is fixed to the value from
046 * that Dataset.
047 * <p>
048 * Transformations only operate on observed values. To operate on implicit zeros then
049 * first call {@link MutableDataset#densify} on the datasets.
050 */
051public final class TransformerMap implements Provenancable<TransformerMapProvenance>, Serializable {
052    private static final long serialVersionUID = 2L;
053
054    private final Map<String, List<Transformer>> map;
055    private final DatasetProvenance datasetProvenance;
056    private final ConfiguredObjectProvenance transformationMapProvenance;
057
058    public TransformerMap(Map<String,List<Transformer>> map, DatasetProvenance datasetProvenance, ConfiguredObjectProvenance transformationMapProvenance) {
059        this.map = map;
060        this.datasetProvenance = datasetProvenance;
061        this.transformationMapProvenance = transformationMapProvenance;
062    }
063
064    /**
065     * Applies a {@link List} of {@link Transformer}s to the supplied double value,
066     * returning the transformed value.
067     * @param value The value to transform.
068     * @param transformerList The transformers to apply.
069     * @return The transformed value.
070     */
071    public static double applyTransformerList(double value, List<Transformer> transformerList) {
072        if (transformerList != null) {
073            for (Transformer t : transformerList) {
074                value = t.transform(value);
075            }
076        }
077        return value;
078    }
079
080    /**
081     * Copies the supplied example and applies the transformers to it.
082     * @param example The example to transform.
083     * @param <T> The type of Output.
084     * @return A copy of the example with the transformers applied to it's features.
085     */
086    public <T extends Output<T>> Example<T> transformExample(Example<T> example) {
087        ArrayExample<T> newExample = new ArrayExample<>(example);
088        newExample.transform(this);
089        return newExample;
090    }
091
092    /**
093     * Copies the supplied example and applies the transformers to it.
094     * @param example The example to transform.
095     * @param featureNames The feature names to densify.
096     * @param <T> The type of Output.
097     * @return A copy of the example with the transformers applied to it's features.
098     */
099    public <T extends Output<T>> Example<T> transformExample(Example<T> example, List<String> featureNames) {
100        ArrayExample<T> newExample = new ArrayExample<>(example);
101        newExample.densify(featureNames);
102        newExample.transform(this);
103        return newExample;
104    }
105
106    /**
107     * Copies the supplied dataset and applies the transformers to each example in it.
108     * <p>
109     * Does not densify the dataset first.
110     * @param dataset The dataset to transform.
111     * @param <T> The type of Output.
112     * @return A deep copy of the dataset (and it's examples) with the transformers applied to it's features.
113     */
114    public <T extends Output<T>> MutableDataset<T> transformDataset(Dataset<T> dataset) {
115        return transformDataset(dataset,false);
116    }
117
118    /**
119     * Copies the supplied dataset and applies the transformers to each example in it.
120     * @param dataset The dataset to transform.
121     * @param densify Densify the dataset before transforming it.
122     * @param <T> The type of Output.
123     * @return A deep copy of the dataset (and it's examples) with the transformers applied to it's features.
124     */
125    public <T extends Output<T>> MutableDataset<T> transformDataset(Dataset<T> dataset, boolean densify) {
126        MutableDataset<T> newDataset = MutableDataset.createDeepCopy(dataset);
127
128        if (densify) {
129            newDataset.densify();
130        }
131
132        newDataset.transform(this);
133
134        return newDataset;
135    }
136
137    @Override
138    public String toString() {
139        return "TransformerMap(map="+map.toString()+")";
140    }
141
142    /**
143     * Get the feature names and associated list of transformers.
144     * @return The entry set of the transformer map.
145     */
146    public Set<Map.Entry<String,List<Transformer>>> entrySet() {
147        return map.entrySet();
148    }
149
150    @Override
151    public TransformerMapProvenance getProvenance() {
152        return new TransformerMapProvenance(this);
153    }
154
155    public final static class TransformerMapProvenance implements ObjectProvenance {
156        private static final long serialVersionUID = 1L;
157
158        private static final String TRANSFORMATION_MAP = "transformation-map";
159        private static final String DATASET = "dataset";
160
161        private final String className;
162        private final ConfiguredObjectProvenance transformationMapProvenance;
163        private final DatasetProvenance datasetProvenance;
164
165        TransformerMapProvenance(TransformerMap host) {
166            this.className = host.getClass().getName();
167            this.transformationMapProvenance = host.transformationMapProvenance;
168            this.datasetProvenance = host.datasetProvenance;
169        }
170
171        public TransformerMapProvenance(Map<String,Provenance> map) {
172            this.className = ObjectProvenance.checkAndExtractProvenance(map,CLASS_NAME,StringProvenance.class,TransformerMapProvenance.class.getSimpleName()).getValue();
173            this.transformationMapProvenance = ObjectProvenance.checkAndExtractProvenance(map,TRANSFORMATION_MAP,ConfiguredObjectProvenance.class,TransformerMapProvenance.class.getSimpleName());
174            this.datasetProvenance = ObjectProvenance.checkAndExtractProvenance(map,DATASET,DatasetProvenance.class,TransformerMapProvenance.class.getSimpleName());
175        }
176
177        @Override
178        public String getClassName() {
179            return className;
180        }
181
182        @Override
183        public Iterator<Pair<String, Provenance>> iterator() {
184            ArrayList<Pair<String,Provenance>> list = new ArrayList<>();
185
186            list.add(new Pair<>(CLASS_NAME,new StringProvenance(CLASS_NAME,className)));
187            list.add(new Pair<>(TRANSFORMATION_MAP,transformationMapProvenance));
188            list.add(new Pair<>(DATASET,datasetProvenance));
189
190            return list.iterator();
191        }
192
193        @Override
194        public boolean equals(Object o) {
195            if (this == o) return true;
196            if (!(o instanceof TransformerMapProvenance)) return false;
197            TransformerMapProvenance pairs = (TransformerMapProvenance) o;
198            return className.equals(pairs.className) &&
199                    transformationMapProvenance.equals(pairs.transformationMapProvenance) &&
200                    datasetProvenance.equals(pairs.datasetProvenance);
201        }
202
203        @Override
204        public int hashCode() {
205            return Objects.hash(className, transformationMapProvenance, datasetProvenance);
206        }
207
208        @Override
209        public String toString() {
210            return generateString("TransformerMap");
211        }
212    }
