031/**
032 * Generates two example train and test datasets, used for unit testing.
033 * They don't necessarily have linear regressed values,
034 * it's for testing the machinery rather than accuracy.
035 * <p>
036 * Also can generate a variety of single dimensional gaussian datasets.
037 */
038public abstract class RegressionDataGenerator {
039
040    public static final String firstDimensionName = "dim1";
041    public static final String secondDimensionName= "dim2";
042
043    public static final String SINGLE_DIM_NAME = "REGRESSOR";
044
045    private static final RegressionFactory REGRESSION_FACTORY = new RegressionFactory();
046    private static final String[] dimensionNames = new String[]{firstDimensionName,secondDimensionName};
047
048    public static Pair<Dataset<Regressor>,Dataset<Regressor>> multiDimDenseTrainTest() {
049        return multiDimDenseTrainTest(-1.0);
050    }

052    /**
053     * Generates a train/test dataset pair which is dense in the features,
054     * each example has 4 features,{A,B,C,D}.
055     * @param negate Supply -1.0 to negate some features.
056     * @return A pair of datasets.
057     */
058    public static Pair<Dataset<Regressor>,Dataset<Regressor>> multiDimDenseTrainTest(double negate) {
059        MutableDataset<Regressor> train = new MutableDataset<>(new SimpleDataSourceProvenance("TrainingData", OffsetDateTime.now(), REGRESSION_FACTORY), REGRESSION_FACTORY);
060
061        String[] names = new String[]{"A","B","C","D"};
062        double[] values = new double[]{1.0,0.5,1.0,negate*1.0};
063        train.add(new ArrayExample<>(new Regressor(dimensionNames,new double[]{5.0,-5.0}),names,values));
064        values = new double[]{1.5,0.35,1.3,negate*1.2};
065        train.add(new ArrayExample<>(new Regressor(dimensionNames,new double[]{5.8,-5.8}),names.clone(),values));
066        values = new double[]{1.2,0.45,1.5,negate*1.0};
067        train.add(new ArrayExample<>(new Regressor(dimensionNames,new double[]{8.0,-8.0}),names.clone(),values));
068
069        values = new double[]{negate*1.1,0.55,negate*1.5,0.5};
070        train.add(new ArrayExample<>(new Regressor(dimensionNames,new double[]{10.0,-10.0}),names.clone(),values));
071        values = new double[]{negate*1.5,0.25,negate*1,0.125};
072        train.add(new ArrayExample<>(new Regressor(dimensionNames,new double[]{10.0,-10.0}),names.clone(),values));
073        values = new double[]{negate*1,0.5,negate*1.123,0.123};
074        train.add(new ArrayExample<>(new Regressor(dimensionNames,new double[]{10.0,-10.0}),names.clone(),values));
075
076        values = new double[]{1.5,5.0,0.5,4.5};
077        train.add(new ArrayExample<>(new Regressor(dimensionNames,new double[]{20,-20}),names.clone(),values));
078        values = new double[]{1.234,5.1235,0.1235,6.0};
079        train.add(new ArrayExample<>(new Regressor(dimensionNames,new double[]{20,-20}),names.clone(),values));
080        values = new double[]{1.734,4.5,0.5123,5.5};
081        train.add(new ArrayExample<>(new Regressor(dimensionNames,new double[]{20,-20}),names.clone(),values));
082
083        values = new double[]{negate*1,0.25,5,10.0};
084        train.add(new ArrayExample<>(new Regressor(dimensionNames,new double[]{50,-50}),names.clone(),values));
085        values = new double[]{negate*1.4,0.55,5.65,12.0};
086        train.add(new ArrayExample<>(new Regressor(dimensionNames,new double[]{50,-50}),names.clone(),values));
087        values = new double[]{negate*1.9,0.25,5.9,15};
088        train.add(new ArrayExample<>(new Regressor(dimensionNames,new double[]{50,-50}),names.clone(),values));
089
090        MutableDataset<Regressor> test = new MutableDataset<>(new SimpleDataSourceProvenance("TestingData", OffsetDateTime.now(), REGRESSION_FACTORY), REGRESSION_FACTORY);
091
092        values = new double[]{2.0,0.45,3.5,negate*2.0};
093        test.add(new ArrayExample<>(new Regressor(dimensionNames,new double[]{5.1,-5.1}),names.clone(),values));
094        values = new double[]{negate*2.0,0.55,negate*2.5,2.5};
095        test.add(new ArrayExample<>(new Regressor(dimensionNames,new double[]{10.0,-10.0}),names.clone(),values));
096        values = new double[]{1.75,5.0,1.0,6.5};
097        test.add(new ArrayExample<>(new Regressor(dimensionNames,new double[]{20,-20}),names.clone(),values));
098        values = new double[]{negate*1.5,0.25,5.0,20.0};
099        test.add(new ArrayExample<>(new Regressor(dimensionNames,new double[]{50,-50}),names.clone(),values));
100
101        return new Pair<>(train,test);
102    }
