040/**
041 * A trainer for regression models that uses LibSVM. Trains an independent model for each output dimension.
042 * <p>
043 * See:
044 * <pre>
045 * Chang CC, Lin CJ.
046 * "LIBSVM: a library for Support Vector Machines"
047 * ACM transactions on intelligent systems and technology (TIST), 2011.
048 * </pre>
049 * for the nu-svr algorithm:
050 * <pre>
051 * Sch√∂lkopf B, Smola A, Williamson R, Bartlett P L.
052 * "New support vector algorithms"
053 * Neural Computation, 2000, 1207-1245.
054 * </pre>
055 * and for the original algorithm:
056 * <pre>
057 * Cortes C, Vapnik V.
058 * "Support-Vector Networks"
059 * Machine Learning, 1995.
060 * </pre>
061 */
062public class LibSVMRegressionTrainer extends LibSVMTrainer<Regressor> {
063    private static final Logger logger = Logger.getLogger(LibSVMRegressionTrainer.class.getName());
064
065    /**
066     * For olcut.
067     */
068    protected LibSVMRegressionTrainer() {}
069
070    public LibSVMRegressionTrainer(SVMParameters<Regressor> parameters) {
071        super(parameters);
072    }
073
074    @Override
075    public void postConfig() {
076        super.postConfig();
077        if (!svmType.isRegression()) {
078            throw new IllegalArgumentException("Supplied classification or anomaly detection parameters to a regression SVM.");
079        }
080    }
081
082    @Override
083    protected LibSVMModel<Regressor> createModel(ModelProvenance provenance, ImmutableFeatureMap featureIDMap, ImmutableOutputInfo<Regressor> outputIDInfo, List<svm_model> models) {
084        return new LibSVMRegressionModel("svm-regression-model", provenance, featureIDMap, outputIDInfo, models);
085    }
086
087    @Override
088    protected List<svm_model> trainModels(svm_parameter curParams, int numFeatures, svm_node[][] features, double[][] outputs) {
089        ArrayList<svm_model> models = new ArrayList<>();
090
091        for (int i = 0; i < outputs.length; i++) {
092            svm_problem problem = new svm_problem();
093            problem.l = outputs[i].length;
094            problem.x = features;
095            problem.y = outputs[i];
096            if (parameters.gamma == 0) {
097                parameters.gamma = 1.0 / numFeatures;
098            }
099            String checkString = svm.svm_check_parameter(problem, curParams);
100            if(checkString != null) {
101                throw new IllegalArgumentException("Error checking SVM parameters: " + checkString);
102            }
103            models.add(svm.svm_train(problem, curParams));
104        }
105
106        return Collections.unmodifiableList(models);
107    }
108
109    @Override
110    protected Pair<svm_node[][], double[][]> extractData(Dataset<Regressor> data, ImmutableOutputInfo<Regressor> outputInfo, ImmutableFeatureMap featureMap) {
111        int numOutputs = outputInfo.size();
112        ArrayList<svm_node> buffer = new ArrayList<>();
113        svm_node[][] features = new svm_node[data.size()][];
114        double[][] outputs = new double[numOutputs][data.size()];
115        int i = 0;
116        for (Example<Regressor> e : data) {
117            double[] curOutputs = e.getOutput().getValues();
118            for (int j = 0; j < curOutputs.length; j++) {
119                outputs[j][i] = curOutputs[j];
120            }
121            features[i] = exampleToNodes(e,featureMap,buffer);
122            i++;
123        }
124        return new Pair<>(features,outputs);
125    }
126}
