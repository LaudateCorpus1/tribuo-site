027import org.tribuo.Example;

029import org.tribuo.Trainer;
030import org.tribuo.provenance.ModelProvenance;
031import org.tribuo.provenance.TrainerProvenance;

041/**
042 * A trainer for simple baseline regressors. Use this only for comparison purposes, if you can't beat these
043 * baselines, your ML system doesn't work.
044 */
045public final class DummyRegressionTrainer implements Trainer<Regressor> {
046
047    public enum DummyType { MEAN, MEDIAN, QUARTILE, CONSTANT, GAUSSIAN }
048
049    @Config(mandatory = true, description="Type of dummy regressor.")
050    private DummyType dummyType;
051
052    @Config(description="Constant value to use for the constant regressor.")
053    private double constantValue = Double.NaN;
054
055    @Config(description="Quartile to use.")
056    private double quartile = Double.NaN;
057
058    @Config(description="The seed for the RNG.")
059    private long seed = 1L;
060
061    private int invocationCount = 0;
062
063    private DummyRegressionTrainer() { }
064
065    @Override
066    public void postConfig() {
067        if ((dummyType == DummyType.CONSTANT) && (Double.isNaN(constantValue))) {
068            throw new PropertyException("","constantValue","Please supply a constant value when using the type CONSTANT.");
069        }
070    }
071
072    @Override
073    public DummyRegressionModel train(Dataset<Regressor> examples, Map<String, Provenance> instanceProvenance) {
074        ModelProvenance provenance = new ModelProvenance(DummyRegressionModel.class.getName(), OffsetDateTime.now(), examples.getProvenance(), getProvenance(), instanceProvenance);
075        invocationCount++;
076        ImmutableOutputInfo<Regressor> outputInfo = examples.getOutputIDInfo();
077        Set<Regressor> domain = outputInfo.getDomain();
078        double[][] outputs = new double[outputInfo.size()][examples.size()];
079        int i = 0;
080        for (Example<Regressor> e : examples) {
081            for (Regressor.DimensionTuple r : e.getOutput()) {
082                int id = outputInfo.getID(r);
083                outputs[id][i] = r.getValue();
084            }
085            i++;
086        }
087        Regressor regressor;
088        switch (dummyType) {
089            case CONSTANT: {
090                Regressor.DimensionTuple[] output = new Regressor.DimensionTuple[outputs.length];
091                for (Regressor r : domain) {
092                    int id = outputInfo.getID(r);
093                    output[id] = new Regressor.DimensionTuple(r.getNames()[0],constantValue);
094                }
095                regressor = new Regressor(output);
096                return new DummyRegressionModel(provenance,examples.getFeatureIDMap(),outputInfo,dummyType,regressor);
097            }
098            case MEAN: {
099                Regressor.DimensionTuple[] output = new Regressor.DimensionTuple[outputs.length];
100                for (Regressor r : domain) {
101                    int id = outputInfo.getID(r);
102                    output[id] = new Regressor.DimensionTuple(r.getNames()[0],Util.mean(outputs[id]));
103                }
104                regressor = new Regressor(output);
105                return new DummyRegressionModel(provenance,examples.getFeatureIDMap(),outputInfo,dummyType,regressor);
106            }
107            case MEDIAN: {
108                Regressor.DimensionTuple[] output = new Regressor.DimensionTuple[outputs.length];
109                for (Regressor r : domain) {
110                    int id = outputInfo.getID(r);
111                    Arrays.sort(outputs[id]);
112                    output[id] = new Regressor.DimensionTuple(r.getNames()[0],outputs[id][outputs[id].length/2]);
113                }
114                regressor = new Regressor(output);
115                return new DummyRegressionModel(provenance,examples.getFeatureIDMap(),outputInfo,dummyType,regressor);
116            }
117            case QUARTILE: {
118                Regressor.DimensionTuple[] output = new Regressor.DimensionTuple[outputs.length];
119                for (Regressor r : domain) {
120                    int id = outputInfo.getID(r);
121                    Arrays.sort(outputs[id]);
122                    output[id] = new Regressor.DimensionTuple(r.getNames()[0],outputs[id][(int) (quartile*outputs[id].length)]);
123                }
124                regressor = new Regressor(output);
125                return new DummyRegressionModel(provenance,examples.getFeatureIDMap(),outputInfo,dummyType,regressor);
126            }
127            case GAUSSIAN: {
128                double[] means = new double[outputs.length];
129                double[] variances = new double[outputs.length];
130                String[] names = new String[outputs.length];
131                for (Regressor r : domain) {
132                    int id = outputInfo.getID(r);
133                    names[id] = r.getNames()[0];
134                    Pair<Double,Double> meanVariance = Util.meanAndVariance(outputs[id]);
135                    means[id] = meanVariance.getA();
136                    variances[id] = meanVariance.getB();
137                }
138                return new DummyRegressionModel(provenance,examples.getFeatureIDMap(),outputInfo,seed,means,variances,names);
139            }
140            default:
141                throw new IllegalStateException("Unknown dummyType " + dummyType);
142        }
143    }
144
145    @Override
146    public String toString() {
147        switch (dummyType) {
148            case CONSTANT:
149                return "DummyRegressionTrainer(dummyType=CONSTANT,constantValue="+constantValue+")";
150            case MEAN:
151                return "DummyRegressionTrainer(dummyType=MEAN)";
152            case MEDIAN:
153                return "DummyRegressionTrainer(dummyType=MEDIAN)";
154            case QUARTILE:
155                return "DummyRegressionTrainer(dummyType=QUARTILE,quartile="+quartile+")";
156            case GAUSSIAN:
157                return "DummyRegressionTrainer(dummyType=GAUSSIAN,seed="+seed+")";
158            default:
159                return "DummyRegressionTrainer(dummyType="+dummyType+")";
160        }
161    }
162
163    @Override
164    public int getInvocationCount() {
165        return invocationCount;
166    }
167
168    @Override
169    public TrainerProvenance getProvenance() {
170        return new DummyRegressionTrainerProvenance(this);
171    }
172
173    /**
174     * Creates a trainer which create models which return a fixed value.
175     * @param value The value to return
176     * @return A regression trainer.
177     */
178    public static DummyRegressionTrainer createConstantTrainer(double value) {
179        DummyRegressionTrainer trainer = new DummyRegressionTrainer();
180        trainer.dummyType = DummyType.CONSTANT;
181        trainer.constantValue = value;
182        return trainer;
183    }
184
185    /**
186     * Creates a trainer which create models which sample the output from a gaussian distribution fit to the training data.
187     * @param seed The RNG seed.
188     * @return A regression trainer.
189     */
190    public static DummyRegressionTrainer createGaussianTrainer(long seed) {
191        DummyRegressionTrainer trainer = new DummyRegressionTrainer();
192        trainer.dummyType = DummyType.GAUSSIAN;
193        trainer.seed = seed;
194        return trainer;
195    }
196
197    /**
198     * Creates a trainer which create models which return the mean of the training data.
199     * @return A regression trainer.
200     */
201    public static DummyRegressionTrainer createMeanTrainer() {
202        DummyRegressionTrainer trainer = new DummyRegressionTrainer();
203        trainer.dummyType = DummyType.MEAN;
204        return trainer;
205    }
206
207    /**
208     * Creates a trainer which create models which return the median of the training data.
209     * @return A regression trainer.
210     */
211    public static DummyRegressionTrainer createMedianTrainer() {
212        DummyRegressionTrainer trainer = new DummyRegressionTrainer();
213        trainer.dummyType = DummyType.MEDIAN;
214        return trainer;
215    }
216
217    /**
218     * Creates a trainer which create models which return the value at the specified fraction of the sorted training data.
219     * @param value The quartile value.
220     * @return A regression trainer.
221     */
222    public static DummyRegressionTrainer createQuartileTrainer(double value) {
223        if (Double.isNaN(value) || value < 0.0 || value > 1.0) {
224            throw new IllegalArgumentException("Please provide an appropriate value between 0.0 and 1.0, found " + value);
225        }
226        DummyRegressionTrainer trainer = new DummyRegressionTrainer();
227        trainer.dummyType = DummyType.QUARTILE;
228        trainer.quartile = value;
229        return trainer;
230    }
231
232    public static class DummyRegressionTrainerProvenance implements TrainerProvenance {
233        private static final long serialVersionUID = 1L;
234
235        private final String className;
236        private final DummyType dummyType;
237        private final long seed;
238        private final double constantValue;
239        private final double quartile;
240
241        public DummyRegressionTrainerProvenance(DummyRegressionTrainer host) {
242            this.className = host.getClass().getName();
243            this.dummyType = host.dummyType;
244            this.seed = host.seed;
245            this.constantValue = host.constantValue;
246            this.quartile = host.quartile;
247        }
248
249        @Override
250        public Map<String, Provenance> getConfiguredParameters() {
251            Map<String, Provenance> map = new HashMap<>();
252
253            map.put("dummyType",new EnumProvenance<>("dummyType",dummyType));
254            map.put("constantValue",new DoubleProvenance("constantValue",constantValue));
255            map.put("quartile",new DoubleProvenance("quartile",quartile));
256            map.put("seed",new LongProvenance("seed",seed));
257
258            return map;
259        }
260
261        @Override
262        public String getClassName() {
263            return className;
264        }
265
266        @Override
267        public String toString() {
268            return generateString("Trainer");
269        }
270    }
