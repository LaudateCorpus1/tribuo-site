017package org.tribuo;
018
019import com.oracle.labs.mlrg.olcut.command.Command;
020import com.oracle.labs.mlrg.olcut.command.CommandGroup;
021import com.oracle.labs.mlrg.olcut.command.CommandInterpreter;
022import com.oracle.labs.mlrg.olcut.config.ConfigurationManager;
023import com.oracle.labs.mlrg.olcut.config.Option;
024import com.oracle.labs.mlrg.olcut.config.Options;
025import org.jline.builtins.Completers;
026import org.jline.reader.Completer;
027import org.jline.reader.impl.completer.NullCompleter;
028
029import java.io.BufferedInputStream;
030import java.io.File;
031import java.io.FileInputStream;
032import java.io.FileNotFoundException;
033import java.io.IOException;
034import java.io.ObjectInputStream;
035import java.util.logging.Level;
036import java.util.logging.Logger;
037
038/**
039 * A command line interface for loading in models and inspecting their feature and output spaces.
040 */
041public class ModelExplorer implements CommandGroup {
042    private static final Logger logger = Logger.getLogger(ModelExplorer.class.getName());
043
044    protected CommandInterpreter shell;
045
046    private Model<?> model;
047
048    public ModelExplorer() {
049        shell = new CommandInterpreter();
050        shell.setPrompt("model sh% ");
051    }
052
053    @Override
054    public String getName() {
055        return "Model Explorer";
056    }
057
058    @Override
059    public String getDescription() {
060        return "Commands for inspecting a Model.";
061    }
062
063    public Completer[] fileCompleter() {
064        return new Completer[]{
065                new Completers.FileNameCompleter(),
066                new NullCompleter()
067        };
068    }
069
070    /**
071     * Start the command shell
072     */
073    public void startShell() {
074        shell.add(this);
075        shell.start();
076    }
077
078    @Command(usage = "<filename> - Load a model from disk.", completers="fileCompleter")
079    public String loadModel(CommandInterpreter ci, File path) {
080        try (ObjectInputStream ois = new ObjectInputStream(new BufferedInputStream(new FileInputStream(path)))) {
081            model = (Model<?>) ois.readObject();
082        } catch (ClassNotFoundException e) {
083            logger.log(Level.SEVERE,"Failed to load class from stream " + path.getAbsolutePath(),e);
084            return "Failed to load model";
085        } catch (FileNotFoundException e) {
086            logger.log(Level.SEVERE,"Failed to open file " + path.getAbsolutePath(),e);
087            return "Failed to load model";
088        } catch (IOException e) {
089            logger.log(Level.SEVERE,"IOException when reading from file " + path.getAbsolutePath(),e);
090            return "Failed to load model";
091        }
092
093        return "Loaded model from path " + path.toString();
094    }
095
096    @Command(usage="Does the model generate probabilities")
097    public String generatesProbabilities(CommandInterpreter ci) {
098        return ""+model.generatesProbabilities();
099    }
100
101    @Command(usage="Shows the model provenance")
102    public String modelProvenance(CommandInterpreter ci) {
103        return model.getProvenance().toString();
104    }
105
106    @Command(usage="Shows the information on a particular feature")
107    public String featureInfo(CommandInterpreter ci, String featureName) {
108        VariableInfo f = model.getFeatureIDMap().get(featureName);
109        if (f != null) {
110            return "" + f.toString();
111        } else {
112            return "Feature " + featureName + " not found.";
113        }
114    }
115
116    @Command(usage="Shows the output information.")
117    public String outputInfo(CommandInterpreter ci) {
118        return model.getOutputIDInfo().toReadableString();
119    }
120
121    @Command(usage="<int> - Shows the top N features in the model")
122    public String topFeatures(CommandInterpreter ci, int numFeatures) {
123        return ""+ model.getTopFeatures(numFeatures);
124    }
125
126    @Command(usage="Shows the number of features in the model")
127    public String numFeatures(CommandInterpreter ci) {
128        return ""+ model.getFeatureIDMap().size();
129    }
130
131    @Command(usage="<min count> - Shows the number of features that occurred more than min count times.")
132    public String minCount(CommandInterpreter ci, int minCount) {
133        int counter = 0;
134        for (VariableInfo f : model.getFeatureIDMap()) {
135            if (f.getCount() > minCount) {
136                counter++;
137            }
138        }
139        return counter + " features occurred more than " + minCount + " times.";
140    }
141
142    public static class ModelExplorerOptions implements Options {
143        @Option(charName='f',longName="filename",usage="Model file to load. Optional.")
144        public String modelFilename;
145    }
146
147    public static void main(String[] args) {
148        ModelExplorerOptions options = new ModelExplorerOptions();
149        ConfigurationManager cm = new ConfigurationManager(args,options,false);
150        ModelExplorer driver = new ModelExplorer();
151        if (options.modelFilename != null) {
152            logger.log(Level.INFO,driver.loadModel(driver.shell, new File(options.modelFilename)));
153        }
154        driver.startShell();
155    }
156}
