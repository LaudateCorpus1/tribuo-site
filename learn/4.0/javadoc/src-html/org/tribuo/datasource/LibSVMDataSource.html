058/**
059 * A DataSource which can read LibSVM formatted data.
060 * <p>
061 * It also provides a static save method which writes LibSVM format data.
062 */
063public final class LibSVMDataSource<T extends Output<T>> implements ConfigurableDataSource<T> {
064    private static final Logger logger = Logger.getLogger(LibSVMDataSource.class.getName());

066    private static final Pattern splitPattern = Pattern.compile("\\s+");

112    private LibSVMDataSource(Path path, URL url, OutputFactory<T> outputFactory, boolean rangeSet, boolean zeroIndexed, int maxFeatureID) throws IOException {
113        this.outputFactory = outputFactory;
114        this.path = path;
115        this.url = url;
116        this.rangeSet = rangeSet;
117        if (rangeSet) {
118            this.zeroIndexed = zeroIndexed;
119            this.minFeatureID = zeroIndexed ? 0 : 1;
120            if (maxFeatureID < minFeatureID + 1) {
121                throw new IllegalArgumentException("maxFeatureID must be positive, found " + maxFeatureID);
122            }
123            this.maxFeatureID = maxFeatureID;
124        }
125        read();
126    }

128    @Override
129    public void postConfig() throws IOException {
130        if (maxFeatureID != Integer.MIN_VALUE) {
131            rangeSet = true;
132            minFeatureID = zeroIndexed ? 0 : 1;
133            if (maxFeatureID < minFeatureID + 1) {
134                throw new IllegalArgumentException("maxFeatureID must be positive, found " + maxFeatureID);
135            }
136        }
137        if ((url == null) && (path == null)) {
138            throw new PropertyException("","path","At most one of url and path must be set.");
139        } else if ((url != null) && (path != null) && !path.toUri().toURL().equals(url)) {
140            throw new PropertyException("","path","At most one of url and path must be set");
141        } else if (path != null) {
142            // url is the store of record.
143            try {
144                url = path.toUri().toURL();
145            } catch (MalformedURLException e) {
146                throw new PropertyException(e,"","path","Path was not a valid URL");
147            }
148        }
149        read();
150    }

195    private void read() throws IOException {
196        int pos = 0;
197        ArrayList<HashMap<Integer,Double>> processedData = new ArrayList<>();
198        ArrayList<String> labels = new ArrayList<>();
199
200        // Idiom copied from Files.readAllLines,
201        // but this doesn't require keeping the whole file in RAM.
202        String line;
203        // Parse the libsvm file, ignoring malformed lines.
204        try (BufferedReader r = new BufferedReader(new InputStreamReader(url.openStream(),StandardCharsets.UTF_8))) {
205            for (;;) {
206                line = r.readLine();
207                if (line == null) {
208                    break;
209                }
210                pos++;
211                String[] fields = splitPattern.split(line);
212                try {
213                    boolean valid = true;
214                    HashMap<Integer, Double> features = new HashMap<>();
215                    for (int i = 1; i < fields.length && valid; i++) {
216                        int ind = fields[i].indexOf(':');
217                        if (ind < 0) {
218                            logger.warning(String.format("Weird line at %d", pos));
219                            valid = false;
220                        }
221                        String ids = fields[i].substring(0, ind);
222                        int id = Integer.parseInt(ids);
223                        if ((!rangeSet) && (maxFeatureID < id)) {
224                            maxFeatureID = id;
225                        }
226                        if ((!rangeSet) && (minFeatureID > id)) {
227                            minFeatureID = id;
228                        }
229                        double val = Double.parseDouble(fields[i].substring(ind + 1));
230                        Double value = features.put(id, val);
231                        if (value != null) {
232                            logger.warning(String.format("Repeated features at line %d", pos));
233                            valid = false;
234                        }
235                    }
236                    if (valid) {
237                        // Store the label
238                        labels.add(fields[0]);
239                        // Store the features
240                        processedData.add(features);
241                    } else {
242                        throw new IOException("Invalid LibSVM format file");
243                    }
244                } catch (NumberFormatException ex) {
245                    logger.warning(String.format("Weird line at %d", pos));
246                    throw new IOException("Invalid LibSVM format file", ex);
247                }
248            }
249        }
250
251        // Calculate the string width
252        int width = (""+maxFeatureID).length();
253        String formatString = "%0"+width+"d";
254
255        // Check to see if it's zero indexed or one indexed, if we didn't observe the zero feature
256        // we assume it's one indexed.
257        int maxID = maxFeatureID;
258        if (minFeatureID != 0) {
259            minFeatureID = 1;
260            zeroIndexed = false;
261        } else {
262            maxID++;
263            zeroIndexed = true;
264        }
265
266        String[] featureNames = new String[maxID];
267        for (int i = 0; i < maxID; i++) {
268            featureNames[i] = String.format(formatString,i);
269        }
270
271        // Generate examples from the processed data
272        ArrayList<Feature> buffer = new ArrayList<>();
273        for (int i = 0; i < processedData.size(); i++) {
274            String labelStr = labels.get(i);
275            HashMap<Integer,Double> features = processedData.get(i);
276            try {
277                T curLabel = outputFactory.generateOutput(labelStr);
278                ArrayExample<T> example = new ArrayExample<>(curLabel);
279                buffer.clear();
280                for (Map.Entry<Integer, Double> e : features.entrySet()) {
281                    // Null check to remove out of range feature indices from test data, if rangeSet was true
282                    int id = e.getKey() - minFeatureID;
283                    if (id < maxID)  {
284                        double value = e.getValue();
285                        Feature f = new Feature(featureNames[id], value);
286                        buffer.add(f);
287                    }
288                }
289                example.addAll(buffer);
290                data.add(example);
291            } catch (NumberFormatException e) {
292                // If the output isn't a valid number for regression tasks.
293                // Features are checked in the input loop above.
294                logger.warning(String.format("Failed to parse example %d",i));
295                throw new IOException("Invalid LibSVM format file");
296            }
297        }
298    }

309    /**
310     * Writes out a dataset in LibSVM format.
311     * <p>
312     * Can write either zero indexed or one indexed.
313     *
314     * @param dataset The dataset to write out.
315     * @param out A stream to write it to.
316     * @param zeroIndexed If true start the feature numbers from zero, otherwise start from one.
317     * @param transformationFunc A function which transforms an {@link Output} into a number.
318     * @param <T> The type of the Output.
319     */
320    public static <T extends Output<T>> void writeLibSVMFormat(Dataset<T> dataset, PrintStream out, boolean zeroIndexed, Function<T,Number> transformationFunc) {
321        int modifier = zeroIndexed ? 0 : 1;
322        ImmutableFeatureMap featureMap = dataset.getFeatureIDMap();
323        for (Example<T> example : dataset) {
324            out.print(transformationFunc.apply(example.getOutput()));
325            out.print(' ');
326            for (Feature feature : example) {
327                out.print(featureMap.get(feature.getName()).getID() + modifier);
328                out.print(':');
329                out.print(feature.getValue());
330                out.print(' ');
331            }
332            out.print('\n');
333        }
334    }

336    public static final class LibSVMDataSourceProvenance extends SkeletalConfiguredObjectProvenance implements DataSourceProvenance {
337        private static final long serialVersionUID = 1L;
338
339        private final DateTimeProvenance fileModifiedTime;
340        private final DateTimeProvenance dataSourceCreationTime;
341        private final HashProvenance sha256Hash;
342
343        <T extends Output<T>> LibSVMDataSourceProvenance(LibSVMDataSource<T> host) {
344            super(host,"DataSource");
345            Optional<OffsetDateTime> time = ProvenanceUtil.getModifiedTime(host.url);
346            this.fileModifiedTime = time.map(offsetDateTime -> new DateTimeProvenance(FILE_MODIFIED_TIME, offsetDateTime)).orElseGet(() -> new DateTimeProvenance(FILE_MODIFIED_TIME, OffsetDateTime.MIN));
347            this.dataSourceCreationTime = new DateTimeProvenance(DATASOURCE_CREATION_TIME,OffsetDateTime.now());
348            this.sha256Hash = new HashProvenance(DEFAULT_HASH_TYPE,RESOURCE_HASH,ProvenanceUtil.hashResource(DEFAULT_HASH_TYPE,host.url));
349        }
350
351        public LibSVMDataSourceProvenance(Map<String,Provenance> map) {
352            this(extractProvenanceInfo(map));
353        }
354
355        private LibSVMDataSourceProvenance(ExtractedInfo info) {
356            super(info);
357            this.fileModifiedTime = (DateTimeProvenance) info.instanceValues.get(FILE_MODIFIED_TIME);
358            this.dataSourceCreationTime = (DateTimeProvenance) info.instanceValues.get(DATASOURCE_CREATION_TIME);
359            this.sha256Hash = (HashProvenance) info.instanceValues.get(RESOURCE_HASH);
360        }
361
362        protected static ExtractedInfo extractProvenanceInfo(Map<String,Provenance> map) {
363            Map<String,Provenance> configuredParameters = new HashMap<>(map);
364            String className = ObjectProvenance.checkAndExtractProvenance(configuredParameters,CLASS_NAME, StringProvenance.class, LibSVMDataSourceProvenance.class.getSimpleName()).getValue();
365            String hostTypeStringName = ObjectProvenance.checkAndExtractProvenance(configuredParameters,HOST_SHORT_NAME, StringProvenance.class, LibSVMDataSourceProvenance.class.getSimpleName()).getValue();
366
367            Map<String,PrimitiveProvenance<?>> instanceParameters = new HashMap<>();
368            instanceParameters.put(FILE_MODIFIED_TIME,ObjectProvenance.checkAndExtractProvenance(configuredParameters,FILE_MODIFIED_TIME,DateTimeProvenance.class, LibSVMDataSourceProvenance.class.getSimpleName()));
369            instanceParameters.put(DATASOURCE_CREATION_TIME,ObjectProvenance.checkAndExtractProvenance(configuredParameters,DATASOURCE_CREATION_TIME,DateTimeProvenance.class, LibSVMDataSourceProvenance.class.getSimpleName()));
370            instanceParameters.put(RESOURCE_HASH,ObjectProvenance.checkAndExtractProvenance(configuredParameters,RESOURCE_HASH,HashProvenance.class, LibSVMDataSourceProvenance.class.getSimpleName()));
371
372            return new ExtractedInfo(className,hostTypeStringName,configuredParameters,instanceParameters);
373        }
374
375        @Override
376        public boolean equals(Object o) {
377            if (this == o) return true;
378            if (!(o instanceof LibSVMDataSourceProvenance)) return false;
379            if (!super.equals(o)) return false;
380            LibSVMDataSourceProvenance pairs = (LibSVMDataSourceProvenance) o;
381            return fileModifiedTime.equals(pairs.fileModifiedTime) &&
382                    dataSourceCreationTime.equals(pairs.dataSourceCreationTime) &&
383                    sha256Hash.equals(pairs.sha256Hash);
384        }
385
386        @Override
387        public int hashCode() {
388            return Objects.hash(super.hashCode(), fileModifiedTime, dataSourceCreationTime, sha256Hash);
389        }
390
391        @Override
392        public Map<String, PrimitiveProvenance<?>> getInstanceValues() {
393            Map<String,PrimitiveProvenance<?>> map = super.getInstanceValues();
394
395            map.put(FILE_MODIFIED_TIME,fileModifiedTime);
396            map.put(DATASOURCE_CREATION_TIME,dataSourceCreationTime);
397            map.put(RESOURCE_HASH,sha256Hash);
398
399            return map;
400        }
401    }
