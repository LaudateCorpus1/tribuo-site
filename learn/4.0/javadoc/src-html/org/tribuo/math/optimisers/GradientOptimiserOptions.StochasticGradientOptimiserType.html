017package org.tribuo.math.optimisers;
018
019import com.oracle.labs.mlrg.olcut.config.Option;
020import com.oracle.labs.mlrg.olcut.config.Options;
021import org.tribuo.math.StochasticGradientOptimiser;
022
023import java.util.logging.Logger;
024
025/**
026 * CLI options for configuring a gradient optimiser.
027 */
028public class GradientOptimiserOptions implements Options {
029    private static final Logger logger = Logger.getLogger(GradientOptimiserOptions.class.getName());
030
031    public enum StochasticGradientOptimiserType {
032        ADADELTA,
033        ADAGRAD,
034        ADAGRADRDA,
035        ADAM,
036        PEGASOS,
037        RMSPROP,
038        CONSTANTSGD,
039        LINEARSGD,
040        SQRTSGD
041    }
042
043    @Option(longName = "sgo-type", usage = "Selects the gradient optimiser. Defaults to ADAGRAD.")
044    private StochasticGradientOptimiserType optimiserType = StochasticGradientOptimiserType.ADAGRAD;
045
046    @Option(longName = "sgo-learning-rate", usage = "Learning rate for AdaGrad, AdaGradRDA, Adam, Pegasos.")
047    public double learningRate = 0.18;
048
049    @Option(longName = "sgo-epsilon", usage = "Epsilon for AdaDelta, AdaGrad, AdaGradRDA, Adam.")
050    public double epsilon = 0.066;
051
052    @Option(longName = "sgo-rho", usage = "Rho for RMSProp, AdaDelta, SGD with Momentum.")
053    public double rho = 0.95;
054
055    @Option(longName = "sgo-lambda", usage = "Lambda for Pegasos.")
056    public double lambda = 1e-2;
057
058    @Option(longName="sgo-parameter-averaging",usage="Use parameter averaging.")
059    public boolean paramAve = false;
060
061    @Option(longName="sgo-momentum",usage="Use momentum in SGD.")
062    public SGD.Momentum momentum = SGD.Momentum.NONE;
063
064    /**
065     * Gets the configured gradient optimiser.
066     * @return The gradient optimiser.
067     */
068    public StochasticGradientOptimiser getOptimiser() {
069        StochasticGradientOptimiser sgo;
070        switch(optimiserType) {
071            case ADADELTA:
072                sgo = new AdaDelta(rho,epsilon);
073                break;
074            case ADAGRAD:
075                sgo = new AdaGrad(learningRate, epsilon);
076                break;
077            case ADAGRADRDA:
078                sgo = new AdaGradRDA(learningRate, epsilon);
079                break;
080            case ADAM:
081                sgo = new Adam(learningRate,epsilon);
082                break;
083            case PEGASOS:
084                sgo = new Pegasos(learningRate,lambda);
085                break;
086            case RMSPROP:
087                sgo = new RMSProp(learningRate,rho);
088                break;
089            case CONSTANTSGD:
090                sgo = SGD.getSimpleSGD(learningRate,rho,momentum);
091                break;
092            case LINEARSGD:
093                sgo = SGD.getLinearDecaySGD(learningRate,rho,momentum);
094                break;
095            case SQRTSGD:
096                sgo = SGD.getSqrtDecaySGD(learningRate,rho,momentum);
097                break;
098            default:
099                throw new IllegalArgumentException("Unhandled StochasticGradientOptimiser type: "+optimiserType);
100        }
101        if (paramAve) {
102            logger.info("Using parameter averaging");
103            return new ParameterAveraging(sgo);
104        } else {
105            return sgo;
106        }
107    }
108
109}
