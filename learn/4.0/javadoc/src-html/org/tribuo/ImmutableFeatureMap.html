017package org.tribuo;
018
019import java.io.Serializable;
020import java.util.HashMap;
021import java.util.List;
022import java.util.Map;
023import java.util.TreeMap;
024
025/**
026 * ImmutableFeatureMap is used when unknown features should not be added to the FeatureMap.
027 * <p>
028 * It's also got feature ids as those are only generated for immutable maps.
029 * <p>
030 * The feature ids are generated by sorting the feature names by the String comparator.
031 * This ensures that any Example with sorted names has sorted int ids, even if some of
032 * those features are unobserved. This is an extremely important property of {@link Feature}s,
033 * {@link Example}s and {@link ImmutableFeatureMap}.
034 */
035public class ImmutableFeatureMap extends FeatureMap implements Serializable {
036    private static final long serialVersionUID = 1L;
037
038    protected final Map<Integer,VariableIDInfo> idMap;
039
040    protected int size;
041
042    public ImmutableFeatureMap(FeatureMap map) {
043        this(generateIDs(map));
044    }
045
046    public ImmutableFeatureMap(List<VariableInfo> infoList) {
047        this(generateIDs(infoList));
048    }
049
050    private ImmutableFeatureMap(Map<String,VariableIDInfo> map) {
051        super(map);
052        idMap = new HashMap<>();
053        for (Map.Entry<String, VariableInfo> e : m.entrySet()) {
054            VariableIDInfo idInfo = (VariableIDInfo) e.getValue();
055            idMap.put(idInfo.getID(),idInfo);
056        }
057        size = m.size();
058    }
059
060    protected ImmutableFeatureMap() {
061        super();
062        idMap = new HashMap<>();
063    }
064
065    /**
066     * Gets the {@link VariableIDInfo}
067     * for this id number. Returns null if it's unknown.
068     * @param id The id number to lookup.
069     * @return The VariableInfo, or null.
070     */
071    public VariableIDInfo get(int id) {
072        return idMap.get(id);
073    }
074
075    /**
076     * Gets the {@link VariableIDInfo}
077     * for this name. Returns null if it's unknown.
078     * @param name The name to lookup.
079     * @return The VariableInfo, or null.
080     */
081    @Override
082    public VariableIDInfo get(String name) {
083        return (VariableIDInfo) super.get(name);
084    }
085
086    /**
087     * Gets the id number for this feature, returns -1 if it's unknown.
088     * @param name The name of the feature.
089     * @return A non-negative integer if the feature is known, -1 otherwise.
090     */
091    public int getID(String name) {
092        VariableIDInfo info = get(name);
093        if (info != null) {
094            return info.getID();
095        } else {
096            return -1;
097        }
098    }
099
100    @Override
101    public int size() {
102        return size;
103    }
104
105    /**
106     * Generates the feature ids by sorting the features with the String comparator,
107     * then sequentially numbering them.
108     * @param map A feature map to convert.
109     * @return A map from feature names to VariableIDInfo objects.
110     */
111    public static Map<String,VariableIDInfo> generateIDs(FeatureMap map) {
112        TreeMap<String,VariableInfo> sortedMap = new TreeMap<>(map.m);
113        return generateIDs(sortedMap);
114    }
115
116    /**
117     * Generates the feature ids by sorting the features with the String comparator,
118     * then sequentially numbering them.
119     * @param list A list of {@link VariableInfo}s to generate a map from.
120     * @return A map from feature names to VariableIDInfo objects.
121     */
122    public static Map<String,VariableIDInfo> generateIDs(List<? extends VariableInfo> list) {
123        TreeMap<String,VariableInfo> sortedMap = new TreeMap<>();
124        for (VariableInfo m : list) {
125            sortedMap.put(m.getName(),m);
126        }
127        return generateIDs(sortedMap);
128    }
129
130    /**
131     * Generates the feature ids from a sorted Map.
132     * @param sortedMap A sorted map of the VariableInfos.
133     * @return A map from feature names to VariableIDInfo objects.
134     */
135    private static Map<String, VariableIDInfo> generateIDs(TreeMap<String, VariableInfo> sortedMap) {
136        Map<String,VariableIDInfo> outputMap = new HashMap<>();
137        int counter = 0;
138        for (Map.Entry<String, VariableInfo> e : sortedMap.entrySet()) {
139            VariableIDInfo newInfo = e.getValue().makeIDInfo(counter);
140            outputMap.put(e.getKey(),newInfo);
141            counter++;
142        }
143        return outputMap;
144    }
145
146}
