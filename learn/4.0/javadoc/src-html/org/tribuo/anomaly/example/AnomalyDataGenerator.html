038/**
039 * Generates three example train and test datasets, used for unit testing.
040 * They don't necessarily have sensible boundaries,
041 * it's for testing the machinery rather than accuracy.
042 * <p>
043 * Also has a dataset generator which returns a training dataset
044 * with no anomalies sampled from a single gaussian, and a test dataset
045 * sampled from two gaussians where the second is labelled anomalous.
046 */
047public abstract class AnomalyDataGenerator {
048
049    private static AnomalyFactory anomalyFactory = new AnomalyFactory();
050
051    /**
052     * Generates two datasets, one without anomalies drawn from a single gaussian
053     * and the second drawn from a mixture of two gaussians, with the second tagged
054     * anomalous.
055     *
056     * Generates 200 training examples and 200 test examples, with 20% anomalies.
057     * @return A pair of datasets.
058     */
059    public static Pair<Dataset<Event>,Dataset<Event>> gaussianAnomaly() {
060        return gaussianAnomaly(200,0.2);
061    }
062
063    /**
064     * Generates two datasets, one without anomalies drawn from a single gaussian
065     * and the second drawn from a mixture of two gaussians, with the second tagged
066     * anomalous.
067     *
068     * @param size The number of points to sample for each dataset.
069     * @param fractionAnomalous The fraction of anomalous data to generate.
070     * @return A pair of datasets.
071     */
072    public static Pair<Dataset<Event>,Dataset<Event>> gaussianAnomaly(long size, double fractionAnomalous) {
073        if (size < 1) {
074            throw new IllegalArgumentException("Size must be a positive number, received " + size);
075        } else if ((fractionAnomalous > 1) || (fractionAnomalous < 0)) {
076            throw new IllegalArgumentException("FractionAnomalous must be between zero and one, received " + fractionAnomalous);
077        }
078        Random rng = new Random(1L);
079
080        // Dataset parameters
081        String[] featureNames = new String[]{"A","B","C","D","E"};
082        double[] expectedMeans = new double[]{1.0,2.0,1.0,2.0,5.0};
083        double[] anomalousMeans = new double[]{-2.0,2.0,-2.0,2.0,-10.0};
084        double[] variances = new double[]{1.0,0.5,0.25,1.0,0.1};
085
086        List<Example<Event>> trainingData = new ArrayList<>();
087        for (int i = 0; i < size; i++) {
088            List<Feature> featureList = generateFeatures(rng,featureNames,expectedMeans,variances);
089            trainingData.add(new ArrayExample<>(EXPECTED_EVENT,featureList));
090        }
091
092        List<Example<Event>> testingData = new ArrayList<>();
093        for (int i = 0; i < size; i++) {
094            double draw = rng.nextDouble();
095            if (draw < fractionAnomalous) {
096                List<Feature> featureList = generateFeatures(rng, featureNames, anomalousMeans, variances);
097                testingData.add(new ArrayExample<>(ANOMALOUS_EVENT, featureList));
098            } else {
099                List<Feature> featureList = generateFeatures(rng, featureNames, expectedMeans, variances);
100                testingData.add(new ArrayExample<>(EXPECTED_EVENT, featureList));
101            }
102        }
103
104        SimpleDataSourceProvenance trainingProvenance = new SimpleDataSourceProvenance("Anomaly training data",anomalyFactory);
105        MutableDataset<Event> train = new MutableDataset<>(new ListDataSource<>(trainingData,anomalyFactory,trainingProvenance));
106        SimpleDataSourceProvenance testingProvenance = new SimpleDataSourceProvenance("Anomaly testing data",anomalyFactory);
107        MutableDataset<Event> test = new MutableDataset<>(new ListDataSource<>(testingData,anomalyFactory,testingProvenance));
108
109        return new Pair<>(train,test);
110    }
111
112    private static List<Feature> generateFeatures(Random rng, String[] names, double[] means, double[] variances) {
113        if ((names.length != means.length) || (names.length != variances.length)) {
114            throw new IllegalArgumentException("Names, means and variances must be the same length");
115        }
116
117        List<Feature> features = new ArrayList<>();
118
119        for (int i = 0; i < names.length; i++) {
120            double value = (rng.nextGaussian() * Math.sqrt(variances[i])) + means[i];
121            features.add(new Feature(names[i],value));
122        }
123
124        return features;
125    }

149        train.add(new ArrayExample<>(EXPECTED_EVENT,names,values));

151        train.add(new ArrayExample<>(EXPECTED_EVENT,names,values));
152        values = new double[]{1.2,0.45,1.5,negate*1.0};
153        train.add(new ArrayExample<>(EXPECTED_EVENT,names,values));
154
155        values = new double[]{negate*1.1,0.55,negate*1.5,0.5};
156        train.add(new ArrayExample<>(EXPECTED_EVENT,names,values));
157        values = new double[]{negate*1.5,0.25,negate*1,0.125};
158        train.add(new ArrayExample<>(EXPECTED_EVENT,names,values));
159        values = new double[]{negate*1,0.5,negate*1.123,0.123};
160        train.add(new ArrayExample<>(EXPECTED_EVENT,names,values));
161
162        values = new double[]{1.5,5.0,0.5,4.5};
163        train.add(new ArrayExample<>(EXPECTED_EVENT,names,values));
164        values = new double[]{1.234,5.1235,0.1235,6.0};
165        train.add(new ArrayExample<>(EXPECTED_EVENT,names,values));
166        values = new double[]{1.734,4.5,0.5123,5.5};
167        train.add(new ArrayExample<>(EXPECTED_EVENT,names,values));
168
169        values = new double[]{negate*1,0.25,5,10.0};

174        train.add(new ArrayExample<>(EXPECTED_EVENT,names,values));
175
176        MutableDataset<Event> test = new MutableDataset<>(new SimpleDataSourceProvenance("TestingData", OffsetDateTime.now(), anomalyFactory), anomalyFactory);
177
178        values = new double[]{2.0,0.45,3.5,negate*2.0};

185        test.add(new ArrayExample<>(EXPECTED_EVENT,names,values));
186
187        return new Pair<>(train,test);
188    }
