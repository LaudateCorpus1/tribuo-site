017package org.tribuo.anomaly;
018
019import com.oracle.labs.mlrg.olcut.util.Pair;
020import org.tribuo.ImmutableOutputInfo;
021import org.tribuo.MutableOutputInfo;
022import org.tribuo.OutputInfo;
023import org.tribuo.anomaly.Event.EventType;
024
025import java.util.ArrayList;
026import java.util.Collections;
027import java.util.HashSet;
028import java.util.List;
029import java.util.Set;
030
031/**
032 * The base class for tracking anomalous events.
033 */
034public abstract class AnomalyInfo implements OutputInfo<Event>  {
035    private static final long serialVersionUID = 1L;
036
037    private static final Set<Event> DOMAIN = makeDomain();
038
039    protected long expectedCount = 0;
040    protected long anomalyCount = 0;
041    protected int unknownCount = 0;
042
043    protected AnomalyInfo() { }
044
045    protected AnomalyInfo(AnomalyInfo other) {
046        this.expectedCount = other.expectedCount;
047        this.anomalyCount = other.anomalyCount;
048        this.unknownCount = other.unknownCount;
049    }
050
051    @Override
052    public int getUnknownCount() {
053        return unknownCount;
054    }
055
056    public long getAnomalyCount() {
057        return anomalyCount;
058    }
059
060    public long getExpectedCount() {
061        return expectedCount;
062    }
063
064    /**
065     * Returns the set of possible {@link Event}s.
066     *
067     * Each event has the default score of Double.NaN.
068     * @return The set of possible events.
069     */
070    @Override
071    public Set<Event> getDomain() {
072        return DOMAIN;
073    }
074
075    /**
076     * Gets the count of the supplied EventType.
077     * @param type An EventType.
078     * @return A non-negative long.
079     */
080    public long getEventCount(EventType type) {
081        switch (type) {
082            case ANOMALOUS:
083                return anomalyCount;
084            case EXPECTED:
085                return expectedCount;
086            case UNKNOWN:
087                return unknownCount;
088            default:
089                return 0;
090        }
091    }
092
093    @Override
094    public Iterable<Pair<String,Long>> outputCountsIterable() {
095        List<Pair<String,Long>> list = new ArrayList<>();
096
097        list.add(new Pair<>(EventType.ANOMALOUS.toString(),anomalyCount));
098        list.add(new Pair<>(EventType.EXPECTED.toString(),expectedCount));
099
100        return list;
101    }
102
103    /**
104     * The number of possible event types (i.e. 2).
105     * @return The number of possible event types.
106     */
107    @Override
108    public int size() {
109        return DOMAIN.size();
110    }
111
112    @Override
113    public ImmutableOutputInfo<Event> generateImmutableOutputInfo() {
114        return new ImmutableAnomalyInfo(this);
115    }
116
117    @Override
118    public MutableOutputInfo<Event> generateMutableOutputInfo() {
119        return new MutableAnomalyInfo(this);
120    }
121
122    @Override
123    public String toReadableString() {
124        return "{Anomalies:"+anomalyCount+",expected:"+expectedCount+"}";
125    }
126
127    @Override
128    public abstract AnomalyInfo copy();
129
130    private static Set<Event> makeDomain() {
131        HashSet<Event> set = new HashSet<>();
132
133        set.add(AnomalyFactory.EXPECTED_EVENT);
134        set.add(AnomalyFactory.ANOMALOUS_EVENT);
135
136        return Collections.unmodifiableSet(set);
137    }
138}
