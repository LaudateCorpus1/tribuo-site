043/**
044 * A trainer for anomaly models that uses LibSVM.
045 * <p>
046 * See:
047 * <pre>
048 * Chang CC, Lin CJ.
049 * "LIBSVM: a library for Support Vector Machines"
050 * ACM transactions on intelligent systems and technology (TIST), 2011.
051 * </pre>
052 * <p>
053 * and for the anomaly detection algorithm:
054 * <pre>
055 * Sch√∂lkopf B, Platt J, Shawe-Taylor J, Smola A J, Williamson R C.
056 * "Estimating the support of a high-dimensional distribution"
057 * Neural Computation, 2001, 1443-1471.
058 * </pre>
059 */
060public class LibSVMAnomalyTrainer extends LibSVMTrainer<Event> {
061    private static final Logger logger = Logger.getLogger(LibSVMAnomalyTrainer.class.getName());
062
063    /**
064     * For OLCUT.
065     */
066    protected LibSVMAnomalyTrainer() {}
067
068    /**
069     * Creates a one-class LibSVM trainer using the supplied parameters.
070     * @param parameters The training parameters.
071     */
072    public LibSVMAnomalyTrainer(SVMParameters<Event> parameters) {
073        super(parameters);
074    }
075
076    @Override
077    public void postConfig() {
078        super.postConfig();
079        if (!svmType.isAnomaly()) {
080            throw new IllegalArgumentException("Supplied classification or regression parameters to an anomaly detection SVM.");
081        }
082    }
083
084    @Override
085    public LibSVMModel<Event> train(Dataset<Event> dataset, Map<String, Provenance> instanceProvenance) {
086        for (Pair<String,Long> p : dataset.getOutputInfo().outputCountsIterable()) {
087            if (p.getA().equals(EventType.ANOMALOUS.toString()) && (p.getB() > 0)) {
088                throw new IllegalArgumentException("LibSVMAnomalyTrainer only supports EXPECTED events at training time.");
089            }
090        }
091        return super.train(dataset,instanceProvenance);
092    }
093
094    @Override
095    protected LibSVMModel<Event> createModel(ModelProvenance provenance, ImmutableFeatureMap featureIDMap, ImmutableOutputInfo<Event> outputIDInfo, List<svm_model> models) {
096        return new LibSVMAnomalyModel("svm-anomaly-detection-model", provenance, featureIDMap, outputIDInfo, models);
097    }
098
099    @Override
100    protected List<svm_model> trainModels(svm_parameter curParams, int numFeatures, svm_node[][] features, double[][] outputs) {
101        svm_problem problem = new svm_problem();
102        problem.l = outputs[0].length;
103        problem.x = features;
104        problem.y = outputs[0];
105        if (parameters.gamma == 0) {
106            parameters.gamma = 1.0 / numFeatures;
107        }
108        String checkString = svm.svm_check_parameter(problem, curParams);
109        if(checkString != null) {
110            throw new IllegalArgumentException("Error checking SVM parameters: " + checkString);
111        }
112        return Collections.singletonList(svm.svm_train(problem, curParams));
113    }
114
115    @Override
116    protected Pair<svm_node[][], double[][]> extractData(Dataset<Event> data, ImmutableOutputInfo<Event> outputInfo, ImmutableFeatureMap featureMap) {
117        double[][] ys = new double[1][data.size()];
118        svm_node[][] xs = new svm_node[data.size()][];
119        List<svm_node> buffer = new ArrayList<>();
120        int i = 0;
121        for (Example<Event> example : data) {
122            ys[0][i] = extractOutput(example.getOutput());
123            xs[i] = exampleToNodes(example, featureMap, buffer);
124            i++;
125        }
126        return new Pair<>(xs,ys);
127    }
128
129    /**
130     * Converts an output into a double for use in training.
131     * <p>
132     * By convention {@link EventType#EXPECTED} is 1.0, other events are -1.0.
133     * @param output The output to convert.
134     * @return The double value.
135     */
136    protected double extractOutput(Event output) {
137        if (output.getType() == Event.EventType.EXPECTED) {
138            return 1.0;
139        } else {
140            return -1.0;
141        }
142    }
143}
