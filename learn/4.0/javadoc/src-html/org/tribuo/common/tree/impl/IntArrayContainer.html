017package org.tribuo.common.tree.impl;
018
019import java.util.Arrays;
020import java.util.List;
021import java.util.logging.Logger;
022
023/**
024 * An array container which maintains the array and the size.
025 * This class is barely more than a Tuple, it's up to you to maintain the size invariant.
026 */
027public class IntArrayContainer {
028    private static final Logger logger = Logger.getLogger(IntArrayContainer.class.getName());
029
030    public int[] array;
031    public int size;
032
033    public IntArrayContainer(int initialCapacity) {
034        array = new int[initialCapacity];
035        size = 0;
036    }
037
038    /**
039     * Grows the backing array, copying the elements.
040     * @param requestedSize The size to grow the array to.
041     */
042    public void grow(int requestedSize) {
043        if (requestedSize > array.length) {
044            // overflow-conscious code
045            int oldCapacity = array.length;
046            int newCapacity = oldCapacity + (oldCapacity >> 1);
047            if (newCapacity - requestedSize < 0) {
048                newCapacity = requestedSize;
049            }
050            // minCapacity is usually close to size, so this is a win:
051            array = Arrays.copyOf(array, newCapacity);
052        }
053    }
054
055    /**
056     * Returns a copy of the elements in use.
057     * @return A copy of the elements.
058     */
059    public int[] copy() {
060        return Arrays.copyOf(array,size);
061    }
062
063    /**
064     * Overwrites values from the supplied array into this array.
065     * @param otherArray The array to copy from.
066     */
067    public void fill(int[] otherArray) {
068        if (otherArray.length > array.length) {
069            array = Arrays.copyOf(otherArray,otherArray.length);
070        } else {
071            System.arraycopy(otherArray,0,array,0,otherArray.length);
072        }
073        size = otherArray.length;
074    }
075
076    /**
077     * Overwrites values in this array with the supplied array.
078     * @param other The array to copy from.
079     */
080    public void fill(IntArrayContainer other) {
081        if (other.array.length > array.length) {
082            array = Arrays.copyOf(other.array,other.size);
083        } else {
084            System.arraycopy(other.array,0,array,0,other.size);
085        }
086        size = other.size;
087    }
088
089    /**
090     * Copies from input to output excluding the values in otherArray.
091     * <p>
092     * This assumes both input and otherArray are sorted. Behaviour is undefined if they aren't.
093     * @param input The input container.
094     * @param otherArray Another (sorted) int array.
095     * @param output The container to write the output to.
096     */
097    public static void removeOther(IntArrayContainer input, int[] otherArray, IntArrayContainer output) {
098        //logger.info("input.size = " + input.size + ", otherArray.length = " + otherArray.length + ", output.length = " + output.array.length);
099        int newSize = input.size - otherArray.length;
100        if (newSize > output.array.length) {
101            output.grow(newSize);
102        }
103
104        int[] inputArray = input.array;
105        int inputSize = input.size;
106        int[] outputArray = output.array;
107        //logger.info("input = " + Arrays.toString(inputArray));
108        //logger.info("otherArray = " + Arrays.toString(otherArray));
109
110        int i = 0; //index into input
111        int j = 0; //index into otherArray
112        int k = 0; //index into output
113        while (i < inputSize) {
114            if (j == otherArray.length) {
115                // Reached end of other, copy from input
116                outputArray[k] = inputArray[i];
117                i++;
118                k++;
119            } else if (inputArray[i] < otherArray[j]) {
120                // Input less than other, copy input
121                outputArray[k] = inputArray[i];
122                i++;
123                k++;
124            } else if (inputArray[i] == otherArray[j]) {
125                // skip both
126                i++;
127                j++;
128            } else {
129                // other less than input, skip
130                j++;
131            }
132        }
133        output.size = k;
134        assert(k == newSize);
135        //logger.info("output = " + Arrays.toString(outputArray));
136    }
137
138    /**
139     * Merges the list of int arrays into a single int array, using the two supplied buffers.
140     * Requires that all arrays in the list are sorted, and that they contain unique values.
141     * @param input A list of int arrays.
142     * @param firstBuffer A buffer.
143     * @param secondBuffer Another buffer.
144     * @return A sorted array containing all the elements from the input.
145     */
146    public static int[] merge(List<int[]> input, IntArrayContainer firstBuffer, IntArrayContainer secondBuffer) {
147        if (input.size() > 0) {
148            firstBuffer.fill(input.get(0));
149            for (int i = 0; i < input.size(); i++) {
150                merge(firstBuffer,input.get(i),secondBuffer);
151                IntArrayContainer tmp = secondBuffer;
152                secondBuffer = firstBuffer;
153                firstBuffer = tmp;
154            }
155            return firstBuffer.copy();
156        } else {
157            return new int[0];
158        }
159    }
160
161    /**
162     * Merges input and otherArray writing to output.
163     * This assumes both input and otherArray are sorted. Behaviour is undefined if they aren't.
164     * @param input The input container.
165     * @param otherArray Another (sorted) int array.
166     * @param output The container to write the output to.
167     */
168    public static void merge(IntArrayContainer input, int[] otherArray, IntArrayContainer output) {
169        int newSize = input.size + otherArray.length;
170        if (newSize > output.array.length) {
171            output.grow(newSize);
172        }
173
174        int[] inputArray = input.array;
175        int inputSize = input.size;
176        int[] outputArray = output.array;
177
178        int i = 0; //index into input
179        int j = 0; //index into otherArray
180        int k = 0; //index into output
181        while ((i < inputSize) || (j < otherArray.length)) {
182            if (i == inputSize) {
183                // Reached end of input, copy from other
184                outputArray[k] = otherArray[j];
185                j++;
186                k++;
187            } else if (j == otherArray.length) {
188                // Reached end of other, copy from input
189                outputArray[k] = inputArray[i];
190                i++;
191                k++;
192            } else if (inputArray[i] < otherArray[j]) {
193                // Input less than other, copy input
194                outputArray[k] = inputArray[i];
195                i++;
196                k++;
197            } else {
198                // other less than input, copy other
199                outputArray[k] = otherArray[j];
200                j++;
201                k++;
202            }
203        }
204        output.size = k;
205        assert(k == newSize);
206        //logger.info("input = " + Arrays.toString(inputArray));
207        //logger.info("otherArray = " + Arrays.toString(otherArray));
208        //logger.info("output = " + Arrays.toString(outputArray));
209    }
210}
