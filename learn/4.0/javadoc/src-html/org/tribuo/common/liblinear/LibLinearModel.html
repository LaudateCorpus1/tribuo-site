017package org.tribuo.common.liblinear;
018
019import org.tribuo.Example;
020import org.tribuo.Excuse;
021import org.tribuo.ImmutableFeatureMap;
022import org.tribuo.ImmutableOutputInfo;
023import org.tribuo.Model;
024import org.tribuo.Output;
025import org.tribuo.provenance.ModelProvenance;
026import de.bwaldvogel.liblinear.Linear;
027
028import java.io.IOException;
029import java.io.StringReader;
030import java.io.StringWriter;
031import java.util.ArrayList;
032import java.util.List;
033import java.util.Optional;
034import java.util.logging.Logger;
035
036/**
037 * A {@link Model} which wraps a LibLinear-java model.
038 * <p>
039 * It disables the LibLinear debug output as it's very chatty.
040 * <p>
041 * See:
042 * <pre>
043 * Fan RE, Chang KW, Hsieh CJ, Wang XR, Lin CJ.
044 * "LIBLINEAR: A library for Large Linear Classification"
045 * Journal of Machine Learning Research, 2008.
046 * </pre>
047 * and for the original algorithm:
048 * <pre>
049 * Cortes C, Vapnik V.
050 * "Support-Vector Networks"
051 * Machine Learning, 1995.
052 * </pre>
053 */
054public abstract class LibLinearModel<T extends Output<T>> extends Model<T> {
055    private static final long serialVersionUID = 3L;
056
057    private static final Logger logger = Logger.getLogger(LibLinearModel.class.getName());
058
059    protected final List<de.bwaldvogel.liblinear.Model> models;
060
061    protected LibLinearModel(String name, ModelProvenance description, ImmutableFeatureMap featureIDMap, ImmutableOutputInfo<T> labelIDMap, boolean generatesProbabilities, List<de.bwaldvogel.liblinear.Model> models) {
062        super(name, description, featureIDMap, labelIDMap, generatesProbabilities);
063        this.models = models;
064        Linear.disableDebugOutput();
065    }
066
067    /**
068     * This call is expensive as it copies out the weight matrix from the
069     * LibLinear model.
070     * <p>
071     * Prefer {@link LibLinearModel#getExcuses} to get multiple excuses.
072     * <p>
073     * @param e The example to excuse.
074     * @return An {@link Excuse} for this example.
075     */
076    @Override
077    public Optional<Excuse<T>> getExcuse(Example<T> e) {
078        double[][] featureWeights = getFeatureWeights();
079        return Optional.of(innerGetExcuse(e, featureWeights));
080    }
081
082    @Override
083    public Optional<List<Excuse<T>>> getExcuses(Iterable<Example<T>> examples) {
084        //This call copies out the weights, so it's better to do it once
085        double[][] featureWeights = getFeatureWeights();
086        List<Excuse<T>> excuses = new ArrayList<>();
087
088        for (Example<T> e : examples) {
089            excuses.add(innerGetExcuse(e, featureWeights));
090        }
091
092        return Optional.of(excuses);
093    }
094
095    /**
096     * Copies the model by writing it out to a String and loading it back in.
097     * <p>
098     * Unfortunately liblinear-java doesn't have a copy constructor on it's model.
099     * @param model THe model to copy.
100     * @return A deep copy of the model.
101     */
102    protected static de.bwaldvogel.liblinear.Model copyModel(de.bwaldvogel.liblinear.Model model) {
103        try {
104            StringWriter writer = new StringWriter();
105            Linear.saveModel(writer,model);
106            String modelString = writer.toString();
107            StringReader reader = new StringReader(modelString);
108            return Linear.loadModel(reader);
109        } catch (IOException e) {
110            throw new IllegalStateException("IOException found when copying the model in memory via a String.",e);
111        }
112    }
113
114    /**
115     * Extracts the feature weights from the models.
116     * The first dimension corresponds to the model index.
117     * @return The feature weights.
118     */
119    protected abstract double[][] getFeatureWeights();
120
121    /**
122     * The call to getFeatureWeights in the public methods copies the
123     * weights array so this inner method exists to save the copy in getExcuses.
124     * <p>
125     * If it becomes a problem then we could cache the feature weights in the
126     * model.
127     * <p>
128     * @param e The example.
129     * @param featureWeights The per dimension feature weights.
130     * @return An excuse for this example.
131     */
132    protected abstract Excuse<T> innerGetExcuse(Example<T> e, double[][] featureWeights);
133}
