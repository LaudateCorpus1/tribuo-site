045/**
046 * A {@link Trainer} which wraps the XGBoost training procedure.
047 * <p>
048 * This only exposes a few of XGBoost's training parameters.
049 * <p>
050 * It uses pthreads outside of the JVM to parallelise the computation.
051 * <p>
052 * See:
053 * <pre>
054 * Chen T, Guestrin C.
055 * "XGBoost: A Scalable Tree Boosting System"
056 * Proceedings of the 22nd ACM SIGKDD International Conference on Knowledge Discovery and Data Mining, 2016.
057 * </pre>
058 * and for the original algorithm:
059 * <pre>
060 * Friedman JH.
061 * "Greedy Function Approximation: a Gradient Boosting Machine"
062 * Annals of statistics, 2001.
063 * </pre>
064 * N.B.: This uses a native C implementation of xgboost that links to various C libraries, including libgomp
065 * and glibc. If you're running on Alpine, which does not natively use glibc, you'll need to install glibc
066 * into the container.
067 */
068public abstract class XGBoostTrainer<T extends Output<T>> implements Trainer<T>, WeightedExamples {

155    /**
156     * Create an XGBoost trainer.
157     *
158     * @param numTrees Number of trees to boost.
159     * @param eta Step size shrinkage parameter (default 0.3, range [0,1]).
160     * @param gamma Minimum loss reduction to make a split (default 0, range
161     * [0,inf]).
162     * @param maxDepth Maximum tree depth (default 6, range [1,inf]).
163     * @param minChildWeight Minimum sum of instance weights needed in a leaf
164     * (default 1, range [0, inf]).
165     * @param subsample Subsample size for each tree (default 1, range (0,1]).
166     * @param featureSubsample Subsample features for each tree (default 1,
167     * range (0,1]).
168     * @param lambda L2 regularization term on weights (default 1).
169     * @param alpha L1 regularization term on weights (default 0).
170     * @param nThread Number of threads to use (default 4).
171     * @param silent Silence the training output text.
172     * @param seed RNG seed.
173     */
174    protected XGBoostTrainer(int numTrees, double eta, double gamma, int maxDepth, double minChildWeight, double subsample, double featureSubsample, double lambda, double alpha, int nThread, boolean silent, long seed) {
175        if (numTrees < 1) {
176            throw new IllegalArgumentException("Must supply a positive number of trees. Received " + numTrees);
177        }
178        this.numTrees = numTrees;
179        this.eta = eta;
180        this.gamma = gamma;
181        this.maxDepth = maxDepth;
182        this.minChildWeight = minChildWeight;
183        this.subsample = subsample;
184        this.featureSubsample = featureSubsample;
185        this.lambda = lambda;
186        this.alpha = alpha;
187        this.nThread = nThread;
188        this.silent = silent ? 1 : 0;
189        this.seed = seed;
190    }

193     * This gives direct access to the XGBoost parameter map.
194     * <p>
195     * It lets you pick things that we haven't exposed like dropout trees, binary classification etc.
196     * <p>
197     * This sidesteps the validation that Tribuo provides for the hyperparameters, and so can produce unexpected results.
198     * @param numTrees Number of trees to boost.
199     * @param parameters A map from string to object, where object can be Number or String.
200     */
201    protected XGBoostTrainer(int numTrees, Map<String,Object> parameters) {
202        if (numTrees < 1) {
203            throw new IllegalArgumentException("Must supply a positive number of trees. Received " + numTrees);
204        }
205        this.numTrees = numTrees;
206        this.parameters.putAll(parameters);
207    }

264    /**
265     * Converts an iterable of examples into a DMatrix.
266     * @param examples The examples to convert.
267     * @param featureMap The feature id map which supplies the indices.
268     * @param responseExtractor The extraction function for the output.
269     * @param <T> The type of the output.
270     * @return A DMatrixTuple.
271     * @throws XGBoostError If the native library failed to construct the DMatrix.
272     */
273    protected static <T extends Output<T>> DMatrixTuple<T> convertExamples(Iterable<Example<T>> examples, ImmutableFeatureMap featureMap, Function<T,Float> responseExtractor) throws XGBoostError {
274        // headers = array of start points for a row
275        // indices = array of feature indices for all data
276        // data = array of feature values for all data
277        // SparseType = DMatrix.SparseType.CSR
278        //public DMatrix(long[] headers, int[] indices, float[] data, SparseType st) throws XGBoostError
279        //
280        // then call
281        //public void setLabel(float[] labels) throws XGBoostError
282
283        boolean labelled = responseExtractor != null;
284        ArrayList<Float> labelsList = new ArrayList<>();
285        ArrayList<Float> dataList = new ArrayList<>();
286        ArrayList<Long> headersList = new ArrayList<>();
287        ArrayList<Integer> indicesList = new ArrayList<>();
288        ArrayList<Float> weightsList = new ArrayList<>();
289        ArrayList<Integer> numValidFeatures = new ArrayList<>();
290        ArrayList<Example<T>> examplesList = new ArrayList<>();
291
292        long rowHeader = 0;
293        headersList.add(rowHeader);
294        for (Example<T> e : examples) {
295            if (labelled) {
296                labelsList.add(responseExtractor.apply(e.getOutput()));
297                weightsList.add(e.getWeight());
298            }
299            examplesList.add(e);
300            long newRowHeader = convertSingleExample(e,featureMap,dataList,indicesList,headersList,rowHeader);
301            numValidFeatures.add((int) (newRowHeader-rowHeader));
302            rowHeader = newRowHeader;
303        }
304
305        float[] data = Util.toPrimitiveFloat(dataList);
306        int[] indices = Util.toPrimitiveInt(indicesList);
307        long[] headers = Util.toPrimitiveLong(headersList);
308
309        DMatrix dataMatrix = new DMatrix(headers, indices, data, DMatrix.SparseType.CSR,featureMap.size());
310        if (labelled) {
311            float[] labels = Util.toPrimitiveFloat(labelsList);
312            dataMatrix.setLabel(labels);
313            float[] weights = Util.toPrimitiveFloat(weightsList);
314            dataMatrix.setWeight(weights);
315        }
316        @SuppressWarnings("unchecked") // Generic array creation
317        Example<T>[] exampleArray = (Example<T>[])examplesList.toArray(new Example[0]);
318        return new DMatrixTuple<>(dataMatrix,Util.toPrimitiveInt(numValidFeatures),exampleArray);
319    }

325    /**
326     * Converts an examples into a DMatrix.
327     * @param example The example to convert.
328     * @param featureMap The feature id map which supplies the indices.
329     * @param responseExtractor The extraction function for the output.
330     * @param <T> The type of the output.
331     * @return A DMatrixTuple.
332     * @throws XGBoostError If the native library failed to construct the DMatrix.
333     */
334    protected static <T extends Output<T>> DMatrixTuple<T> convertExample(Example<T> example, ImmutableFeatureMap featureMap, Function<T,Float> responseExtractor) throws XGBoostError {
335        // headers = array of start points for a row
336        // indices = array of feature indices for all data
337        // data = array of feature values for all data
338        // SparseType = DMatrix.SparseType.CSR
339        //public DMatrix(long[] headers, int[] indices, float[] data, SparseType st) throws XGBoostError
340        //
341        // then call
342        //public void setLabel(float[] labels) throws XGBoostError
343
344        boolean labelled = responseExtractor != null;
345        ArrayList<Float> dataList = new ArrayList<>();
346        ArrayList<Integer> indicesList = new ArrayList<>();
347        ArrayList<Long> headersList = new ArrayList<>();
348        headersList.add(0L);
349
350        long header = convertSingleExample(example,featureMap,dataList,indicesList,headersList,0);
351
352        float[] data = Util.toPrimitiveFloat(dataList);
353        int[] indices = Util.toPrimitiveInt(indicesList);
354        long[] headers = Util.toPrimitiveLong(headersList);
355
356        DMatrix dataMatrix = new DMatrix(headers, indices, data, DMatrix.SparseType.CSR,featureMap.size());
357        if (labelled) {
358            float[] labels = new float[1];
359            labels[0] = responseExtractor.apply(example.getOutput());
360            dataMatrix.setLabel(labels);
361            float[] weights = new float[1];
362            weights[0] = example.getWeight();
363            dataMatrix.setWeight(weights);
364        }
365        @SuppressWarnings("unchecked") // Generic array creation
366        Example<T>[] exampleArray = (Example<T>[])new Example[]{example};
367        return new DMatrixTuple<>(dataMatrix,new int[]{(int)header},exampleArray);
368    }

370    /**
371     * Writes out the features from an example into the three supplied {@link ArrayList}s.
372     * <p>
373     * This is used to transform examples into the right format for an XGBoost call.
374     * It's used in both the Classification and Regression XGBoost backends.
375     * The ArrayLists must be non-null, and can contain existing values (as this
376     * method is called multiple times to build up an arraylist containing all the
377     * feature values for a dataset).
378     * <p>
379     * Features with colliding feature ids are summed together.
380     * <p>
381     * Can throw IllegalArgumentException if the {@link Example} contains no features.
382     * @param example The example to inspect.
383     * @param featureMap The feature map of the model/dataset (used to preserve hash information).
384     * @param dataList The output feature values.
385     * @param indicesList The output indices.
386     * @param headersList The output header position (an integer saying how long each sparse example is).
387     * @param header The current header position.
388     * @param <T> The type of the example.
389     * @return The updated header position.
390     */
391    protected static <T extends Output<T>> long convertSingleExample(Example<T> example, ImmutableFeatureMap featureMap, ArrayList<Float> dataList, ArrayList<Integer> indicesList, ArrayList<Long> headersList, long header) {
392        int numActiveFeatures = 0;
393        int prevIdx = -1;
394        int indicesSize = indicesList.size();
395        for (Feature f : example) {
396            int id = featureMap.getID(f.getName());
397            if (id > prevIdx){
398                prevIdx = id;
399                dataList.add((float) f.getValue());
400                indicesList.add(id);
401                numActiveFeatures++;
402            } else if (id > -1) {
403                //
404                // Collision, deal with it.
405                int collisionIdx = Util.binarySearch(indicesList,id,indicesSize,numActiveFeatures+indicesSize);
406                if (collisionIdx < 0) {
407                    //
408                    // Collision but not present in tmpIndices
409                    // move data and bump i
410                    collisionIdx = - (collisionIdx + 1);
411                    indicesList.add(collisionIdx,id);
412                    dataList.add(collisionIdx,(float) f.getValue());
413                    numActiveFeatures++;
414                } else {
415                    //
416                    // Collision present in tmpIndices
417                    // add the values.
418                    dataList.set(collisionIdx, dataList.get(collisionIdx) + (float) f.getValue());
419                }
420            }
421        }
422        if (numActiveFeatures == 0) {
423            throw new IllegalArgumentException("No features found in Example " + example.toString());
424        }
425        header += numActiveFeatures;
426        headersList.add(header);
427        return header;
428    }

430    /**
431     * Writes out the features from a SparseVector into the three supplied {@link ArrayList}s.
432     * <p>
433     * This is used to transform examples into the right format for an XGBoost call.
434     * It's used when predicting with an externally trained XGBoost model, as the
435     * external training may not respect Tribuo's feature ordering constraints.
436     * The ArrayLists must be non-null, and can contain existing values (as this
437     * method is called multiple times to build up an arraylist containing all the
438     * feature values for a dataset).
439     * </p>
440     * <p>
441     * This is much simpler than {@link XGBoostTrainer#convertSingleExample} as the validation
442     * of feature indices is done in the {@link org.tribuo.interop.ExternalModel} class.
443     * </p>
444     * @param vector The features to convert.
445     * @param dataList The output feature values.
446     * @param indicesList The output indices.
447     * @param headersList The output header position (an integer saying how long each sparse example is).
448     * @param header The current header position.
449     * @return The updated header position.
450     */
451    static long convertSingleExample(SparseVector vector, ArrayList<Float> dataList, ArrayList<Integer> indicesList, ArrayList<Long> headersList, long header) {
452        int numActiveFeatures = 0;
453        for (VectorTuple v : vector) {
454            dataList.add((float) v.value);
455            indicesList.add(v.index);
456            numActiveFeatures++;
457        }
458        header += numActiveFeatures;
459        headersList.add(header);
460        return header;
461    }

463    /**
464     * Used when predicting with an externally trained XGBoost model.
465     * @param vector The features to convert.
466     * @return A DMatrix representing the features.
467     * @throws XGBoostError If the native library returns an error state.
468     */
469    protected static DMatrix convertSparseVector(SparseVector vector) throws XGBoostError {
470        // headers = array of start points for a row
471        // indices = array of feature indices for all data
472        // data = array of feature values for all data
473        // SparseType = DMatrix.SparseType.CSR
474        //public DMatrix(long[] headers, int[] indices, float[] data, SparseType st) throws XGBoostError
475        ArrayList<Float> dataList = new ArrayList<>();
476        ArrayList<Long> headersList = new ArrayList<>();
477        ArrayList<Integer> indicesList = new ArrayList<>();
478
479        long rowHeader = 0;
480        headersList.add(rowHeader);
481        convertSingleExample(vector,dataList,indicesList,headersList,rowHeader);
482
483        float[] data = Util.toPrimitiveFloat(dataList);
484        int[] indices = Util.toPrimitiveInt(indicesList);
485        long[] headers = Util.toPrimitiveLong(headersList);
486
487        return new DMatrix(headers, indices, data, DMatrix.SparseType.CSR,vector.size());
488    }

490    /**
491     * Used when predicting with an externally trained XGBoost model.
492     * <p>
493     * It is assumed all vectors are the same size when passed into this function.
494     * @param vectors The batch of features to convert.
495     * @return A DMatrix representing the batch of features.
496     * @throws XGBoostError If the native library returns an error state.
497     */
498    protected static DMatrix convertSparseVectors(List<SparseVector> vectors) throws XGBoostError {
499        // headers = array of start points for a row
500        // indices = array of feature indices for all data
501        // data = array of feature values for all data
502        // SparseType = DMatrix.SparseType.CSR
503        //public DMatrix(long[] headers, int[] indices, float[] data, SparseType st) throws XGBoostError
504        ArrayList<Float> dataList = new ArrayList<>();
505        ArrayList<Long> headersList = new ArrayList<>();
506        ArrayList<Integer> indicesList = new ArrayList<>();
507
508        int numFeatures = 0;
509        long rowHeader = 0;
510        headersList.add(rowHeader);
511        for (SparseVector e : vectors) {
512            rowHeader = convertSingleExample(e,dataList,indicesList,headersList,rowHeader);
513            numFeatures = e.size(); // All vectors are assumed to be the same size.
514        }
515
516        float[] data = Util.toPrimitiveFloat(dataList);
517        int[] indices = Util.toPrimitiveInt(indicesList);
518        long[] headers = Util.toPrimitiveLong(headersList);
519
520        return new DMatrix(headers, indices, data, DMatrix.SparseType.CSR, numFeatures);
521    }
