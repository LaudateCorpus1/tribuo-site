050/**
051 * A {@link Model} which wraps around a XGBoost.Booster.
052 * <p>
053 * XGBoost is a fast implementation of gradient boosted decision trees.
054 * <p>
055 * Throws IllegalStateException if the XGBoost C++ library fails to load or throws an exception.
056 * <p>
057 * See:
058 * <pre>
059 * Chen T, Guestrin C.
060 * "XGBoost: A Scalable Tree Boosting System"
061 * Proceedings of the 22nd ACM SIGKDD International Conference on Knowledge Discovery and Data Mining, 2016.
062 * </pre>
063 * and for the original algorithm:
064 * <pre>
065 * Friedman JH.
066 * "Greedy Function Approximation: a Gradient Boosting Machine"
067 * Annals of statistics, 2001.
068 * </pre>
069 */
070public final class XGBoostModel<T extends Output<T>> extends Model<T> {
071    private static final long serialVersionUID = 4L;
072
073    private static final Logger logger = Logger.getLogger(XGBoostModel.class.getName());
074
075    private final XGBoostOutputConverter<T> converter;
076
077    protected transient List<Booster> models;
078
079    XGBoostModel(String name, ModelProvenance description,
080                 ImmutableFeatureMap featureIDMap, ImmutableOutputInfo<T> labelIDMap,
081                 List<Booster> models, XGBoostOutputConverter<T> converter) {
082        super(name,description,featureIDMap,labelIDMap,converter.generatesProbabilities());
083        this.converter = converter;
084        this.models = models;
085    }
086
087    /**
088     * Sets the number of threads to use at prediction time.
089     * <p>
090     * If set to 0 sets nthreads = num hardware threads.
091     * @param threads The new number of threads.
092     */
093    public void setNumThreads(int threads) {
094        if (threads > -1) {
095            try {
096                for (Booster model : models) {
097                    model.setParam("nthread", threads);
098                }
099            } catch (XGBoostError e) {
100                logger.log(Level.SEVERE, "XGBoost threw an error", e);
101                throw new IllegalStateException(e);
102            }
103        }
104    }
105
106    /**
107     * Uses the model to predict the labels for multiple examples contained in
108     * a data set.
109     * @param examples the data set containing the examples to predict.
110     * @return the results of the predictions, in the same order as the
111     * data set generates the example.
112     */
113    @Override
114    public List<Prediction<T>> predict(Dataset<T> examples) {
115        return predict(examples.getData());
116    }
117
118    /**
119     * Uses the model to predict the label for multiple examples.
120     * @param examples the examples to predict.
121     * @return the results of the prediction, in the same order as the
122     * examples.
123     */
124    @Override
125    public List<Prediction<T>> predict(Iterable<Example<T>> examples) {
126        try {
127            DMatrixTuple<T> testMatrix = XGBoostTrainer.convertExamples(examples,featureIDMap);
128            List<float[][]> outputs = new ArrayList<>();
129            for (Booster model : models) {
130                outputs.add(model.predict(testMatrix.data));
131            }
132
133            int[] numValidFeatures = testMatrix.numValidFeatures;
134            Example<T>[] exampleArray = testMatrix.examples;
135            return converter.convertBatchOutput(outputIDInfo,outputs,numValidFeatures,exampleArray);
136        } catch (XGBoostError e) {
137            logger.log(Level.SEVERE, "XGBoost threw an error", e);
138            throw new IllegalStateException(e);
139        }
140
141    }
142
143    @Override
144    public Prediction<T> predict(Example<T> example) {
145        try {
146            DMatrixTuple<T> testData = XGBoostTrainer.convertExample(example,featureIDMap);
147            List<float[]> outputs = new ArrayList<>();
148            for (Booster model : models) {
149                outputs.add(model.predict(testData.data)[0]);
150            }
151            Prediction<T> pred = converter.convertOutput(outputIDInfo,outputs,testData.numValidFeatures[0],example);
152            return pred;
153        } catch (XGBoostError e) {
154            logger.log(Level.SEVERE, "XGBoost threw an error", e);
155            throw new IllegalStateException(e);
156        }
157    }
158
159    @Override
160    public Map<String, List<Pair<String,Double>>> getTopFeatures(int n) {
161        try {
162            int maxFeatures = n < 0 ? featureIDMap.size() : n;
163            // Aggregate feature scores across all the models.
164            // This throws away model specific information which is useful in the case of regression,
165            // but it's very tricky to get the dimension name associated with the model.
166            Map<String, MutableDouble> outputMap = new HashMap<>();
167            for (Booster model : models) {
168                Map<String, Integer> xgboostMap = model.getFeatureScore("");
169                for (Map.Entry<String,Integer> f : xgboostMap.entrySet()) {
170                    int id = Integer.parseInt(f.getKey().substring(1));
171                    String name = featureIDMap.get(id).getName();
172                    MutableDouble curVal = outputMap.computeIfAbsent(name,(k)->new MutableDouble());
173                    curVal.increment(f.getValue());
174                }
175            }
176            Comparator<Pair<String, Double>> comparator = Comparator.comparingDouble(p -> Math.abs(p.getB()));
177            PriorityQueue<Pair<String,Double>> q = new PriorityQueue<>(maxFeatures,comparator);
178            for (Map.Entry<String,MutableDouble> e : outputMap.entrySet()) {
179                Pair<String,Double> cur = new Pair<>(e.getKey(), e.getValue().doubleValue());
180
181                if (q.size() < maxFeatures) {
182                    q.offer(cur);
183                } else if (comparator.compare(cur,q.peek()) > 0) {
184                    q.poll();
185                    q.offer(cur);
186                }
187            }
188            List<Pair<String,Double>> list = new ArrayList<>();
189            while(q.size() > 0) {
190                list.add(q.poll());
191            }
192            Collections.reverse(list);
193
194            Map<String, List<Pair<String,Double>>> map = new HashMap<>();
195            map.put(Model.ALL_OUTPUTS,list);
196
197            return map;
198        } catch (XGBoostError e) {
199            logger.log(Level.SEVERE, "XGBoost threw an error", e);
200            return Collections.emptyMap();
201        }
202    }
203
204    /**
205     * Returns the string model dumps from each Booster.
206     * @return The model dumps.
207     */
208    public List<String[]> getModelDump() {
209        try {
210            List<String[]> list = new ArrayList<>();
211            for (Booster m : models) {
212                list.add(m.getModelDump("", true));
213            }
214            return list;
215        } catch (XGBoostError e) {
216            throw new IllegalStateException(e);
217        }
218    }
219
220    @Override
221    public Optional<Excuse<T>> getExcuse(Example<T> example) {
222        return Optional.empty();
223    }
224
225    @Override
226    protected Model<T> copy(String newName, ModelProvenance newProvenance) {
227        try {
228            List<Booster> newModels = new ArrayList<>();
229            for (Booster model : models) {
230                byte[] serialisedBooster = model.toByteArray();
231                newModels.add(XGBoost.loadModel(new ByteArrayInputStream(serialisedBooster)));
232            }
233            return new XGBoostModel<>(newName,newProvenance,featureIDMap,outputIDInfo,newModels,converter);
234        } catch (XGBoostError | IOException e) {
235            throw new IllegalStateException("Unable to copy XGBoost model.",e);
236        }
237    }
238
239    private void writeObject(ObjectOutputStream out) throws IOException {
240        out.defaultWriteObject();
241        try {
242            out.writeInt(models.size());
243            for (Booster model : models) {
244                byte[] serialisedBooster = model.toByteArray();
245                out.writeObject(serialisedBooster);
246            }
247        } catch (XGBoostError e) {
248            throw new IOException("Failed to serialize the XGBoost model",e);
249        }
250    }
251
252    private void readObject(ObjectInputStream in) throws IOException, ClassNotFoundException {
253        in.defaultReadObject();
254        try {
255            models = new ArrayList<>();
256            int numModels = in.readInt();
257            for (int i = 0; i < numModels; i++) {
258                // Now read in the byte array and rebuild each Booster
259                byte[] serialisedBooster = (byte[]) in.readObject();
260                models.add(XGBoost.loadModel(new ByteArrayInputStream(serialisedBooster)));
261            }
262        } catch (XGBoostError e) {
263            throw new IOException("Failed to deserialize the XGBoost model",e);
264        }
265    }
