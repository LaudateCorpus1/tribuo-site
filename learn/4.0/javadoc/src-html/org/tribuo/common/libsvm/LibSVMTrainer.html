045/**
046 * A trainer that will train using libsvm's Java implementation.
047 * <p>
048 * See:
049 * <pre>
050 * Chang CC, Lin CJ.
051 * "LIBSVM: a library for Support Vector Machines"
052 * ACM transactions on intelligent systems and technology (TIST), 2011.
053 * </pre>
054 * for the nu-svm algorithm:
055 * <pre>
056 * Sch√∂lkopf B, Smola A, Williamson R, Bartlett P L.
057 * "New support vector algorithms"
058 * Neural Computation, 2000, 1207-1245.
059 * </pre>
060 * and for the original algorithm:
061 * <pre>
062 * Cortes C, Vapnik V.
063 * "Support-Vector Networks"
064 * Machine Learning, 1995.
065 * </pre>
066 */
067public abstract class LibSVMTrainer<T extends Output<T>> implements Trainer<T> {
068    
069    private static final Logger logger = Logger.getLogger(LibSVMTrainer.class.getName());
070
071    protected svm_parameter parameters;
072
073    @Config(mandatory=true,description="Type of SVM algorithm.")
074    protected SVMType<T> svmType;
075
076    @Config(description="Type of Kernel.")
077    private KernelType kernelType = KernelType.LINEAR;
078
079    @Config(description="Polynomial degree.")
080    private int degree = 3;
081
082    @Config(description="Width of the RBF kernel, or scalar on sigmoid kernel.")
083    private double gamma = 0.0;
084
085    @Config(description="Polynomial coefficient or shift in sigmoid kernel.")
086    private double coef0 = 0.0;
087
088    @Config(description="nu value in NU SVM.")
089    private double nu = 0.5;
090
091    @Config(description="Internal cache size, most of the time should be left at default.")
092    private double cache_size = 500;
093
094    @Config(description="Cost parameter for incorrect predictions.")
095    private double cost = 1.0; // aka svm_parameters.C
096
097    @Config(description="Tolerance of the termination criterion.")
098    private double eps = 1e-3;
099
100    @Config(description="Epsilon in EPSILON_SVR.")
101    private double p = 1e-3;
102
103    @Config(description="Regularise the weight parameters.")
104    private boolean shrinking = true;
105
106    @Config(description="Generate probability estimates.")
107    private boolean probability = false;
108
109    private int trainInvocationCounter = 0;
110
111    /**
112     * For olcut.
113     */
114    protected LibSVMTrainer() {}
115
116    protected LibSVMTrainer(SVMParameters<T> parameters) {
117        this.parameters = parameters.getParameters();
118        // Unpack the parameters for the provenance system.
119        this.svmType = parameters.getSvmType();
120        this.kernelType = parameters.getKernelType();
121        this.degree = this.parameters.degree;
122        this.gamma = parameters.getGamma();
123        this.coef0 = this.parameters.coef0;
124        this.nu = this.parameters.nu;
125        this.cache_size = this.parameters.cache_size;
126        this.cost = this.parameters.C;
127        this.eps = this.parameters.eps;
128        this.p = this.parameters.p;
129        this.shrinking = this.parameters.shrinking == 1;
130        this.probability = this.parameters.probability == 1;
131    }
132
133    @Override
134    public void postConfig() {
135        parameters = new svm_parameter();
136        parameters.svm_type = svmType.getNativeType();
137        parameters.kernel_type = kernelType.getNativeType();
138        parameters.degree = degree;
139        parameters.gamma = gamma;
140        parameters.coef0 = coef0;
141        parameters.nu = nu;
142        parameters.cache_size = cache_size;
143        parameters.C = cost;
144        parameters.eps = eps;
145        parameters.p = p;
146        parameters.shrinking = shrinking ? 1 : 0;
147        parameters.probability = probability ? 1 : 0;
148    }
149
150    @Override
151    public String toString() {
152        StringBuilder buffer = new StringBuilder();
153
154        buffer.append("LibSVMTrainer(");
155        buffer.append("svm_params=");
156        buffer.append(SVMParameters.svmParamsToString(parameters));
157        buffer.append(")");
158
159        return buffer.toString();
160    }
161
162    @Override
163    public LibSVMModel<T> train(Dataset<T> examples) {
164        return train(examples, Collections.emptyMap());
165    }
166
167    @Override
168    public LibSVMModel<T> train(Dataset<T> examples, Map<String, Provenance> runProvenance) {
169        if (examples.getOutputInfo().getUnknownCount() > 0) {
170            throw new IllegalArgumentException("The supplied Dataset contained unknown Outputs, and this Trainer is supervised.");
171        }
172        ImmutableFeatureMap featureIDMap = examples.getFeatureIDMap();
173        ImmutableOutputInfo<T> outputIDInfo = examples.getOutputIDInfo();
174
175        TrainerProvenance trainerProvenance = getProvenance();
176        ModelProvenance provenance = new ModelProvenance(LibSVMModel.class.getName(), OffsetDateTime.now(), examples.getProvenance(), trainerProvenance, runProvenance);
177        trainInvocationCounter++;
178
179        svm_parameter curParams = setupParameters(outputIDInfo);
180
181        Pair<svm_node[][],double[][]> data = extractData(examples,outputIDInfo,featureIDMap);
182
183        List<svm_model> models = trainModels(curParams,featureIDMap.size()+1,data.getA(),data.getB());
184
185        return createModel(provenance,featureIDMap,outputIDInfo,models);
186    }
187
188    /**
189     * Construct the appropriate subtype of LibSVMModel for the prediction task.
190     * @param provenance The model provenance.
191     * @param featureIDMap The feature id map.
192     * @param outputIDInfo The output id info.
193     * @param models The svm models.
194     * @return An implementation of LibSVMModel.
195     */
196    protected abstract LibSVMModel<T> createModel(ModelProvenance provenance, ImmutableFeatureMap featureIDMap, ImmutableOutputInfo<T> outputIDInfo, List<svm_model> models);
197
198    /**
199     * Train all the liblinear instances necessary for this dataset.
200     * @param curParams The LibLinear parameters.
201     * @param numFeatures The number of features in this dataset.
202     * @param features The features themselves.
203     * @param outputs The outputs.
204     * @return A list of liblinear models.
205     */
206    protected abstract List<svm_model> trainModels(svm_parameter curParams, int numFeatures, svm_node[][] features, double[][] outputs);
207
208    /**
209     * Extracts the features and {@link Output}s in LibLinear's format.
210     * @param data The input data.
211     * @param outputInfo The output info.
212     * @param featureMap The feature info.
213     * @return The features and outputs.
214     */
215    protected abstract Pair<svm_node[][],double[][]> extractData(Dataset<T> data, ImmutableOutputInfo<T> outputInfo, ImmutableFeatureMap featureMap);
216
217    /**
218     * Constructs the svm_parameter. Most of the time this is a no-op, but
219     * classification overrides it to incorporate label weights if they exist.
220     * @param info The output info.
221     * @return The svm_parameters to use for training.
222     */
223    protected svm_parameter setupParameters(ImmutableOutputInfo<T> info) {
224        return parameters;
225    }
226
227    @Override
228    public int getInvocationCount() {
229        return trainInvocationCounter;
230    }
231
232    /**
233     * Convert the example into an array of svm_node which represents a sparse feature vector.
234     * <p>
235     * If there are collisions in the feature ids then the values are summed.
236     * @param example The example to convert.
237     * @param featureIDMap The feature id map which holds the indices.
238     * @param features A buffer to use.
239     * @param <T> The type of the ouput.
240     * @return A sparse feature vector.
241     */
242    public static <T extends Output<T>> svm_node[] exampleToNodes(Example<T> example, ImmutableFeatureMap featureIDMap, List<svm_node> features) {
243        if (features == null) {
244            features = new ArrayList<>();
245        }
246        features.clear();
247        int prevIdx = -1;
248        for (Feature f : example) {
249            int id = featureIDMap.getID(f.getName());
250            double value = f.getValue();
251            if (id > prevIdx){
252                prevIdx = id;
253                svm_node n = new svm_node();
254                n.index = id;
255                n.value = value;
256                features.add(n);
257            } else if (id > -1) {
258                //
259                // Collision, deal with it.
260                int collisionIdx = Util.binarySearch(features,id,(svm_node n) -> n.index);
261                if (collisionIdx < 0) {
262                    //
263                    // Collision but not present in features
264                    // move data and bump i
265                    collisionIdx = - (collisionIdx + 1);
266                    svm_node n = new svm_node();
267                    n.index = id;
268                    n.value = value;
269                    features.add(collisionIdx,n);
270                } else {
271                    //
272                    // Collision present in features
273                    // add the values.
274                    svm_node n = features.get(collisionIdx);
275                    n.value += value;
276                }
277            }
278        }
279        return features.toArray(new svm_node[0]);
280    }
281
282    @Override
283    public TrainerProvenance getProvenance() {
284        return new TrainerProvenanceImpl(this);
285    }
