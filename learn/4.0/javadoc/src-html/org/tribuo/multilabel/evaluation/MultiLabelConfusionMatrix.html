017package org.tribuo.multilabel.evaluation;
018
019import org.tribuo.ImmutableOutputInfo;
020import org.tribuo.Model;
021import org.tribuo.Prediction;
022import org.tribuo.classification.Label;
023import org.tribuo.classification.evaluation.ConfusionMatrix;
024import org.tribuo.math.la.DenseMatrix;
025import org.tribuo.multilabel.MultiLabel;
026
027import java.util.List;
028import java.util.Set;
029import java.util.function.Function;
030
031/**
032 * A {@link ConfusionMatrix} which accepts {@link MultiLabel}s.
033 *
034 * <p>
035 * In a multi-label confusion matrix M,
036 * <pre>
037 * tn = M[:, 0, 0]
038 * fn = M[:, 1, 0]
039 * tp = M[:, 1, 1]
040 * fp = M[:, 0, 1]
041 * </pre>
042 * <p>
043 * For class-wise values,
044 * <pre>
045 * tn(class i) = M[i, 0, 0]
046 * fn(class i) = M[i, 1, 0]
047 * tp(class i) = M[i, 1, 1]
048 * fp(class i) = M[i, 0, 1]
049 * </pre>
050 */
051public final class MultiLabelConfusionMatrix implements ConfusionMatrix<MultiLabel> {
052
053    private final ImmutableOutputInfo<MultiLabel> domain;
054    private final DenseMatrix[] mcm;
055    private final DenseMatrix confusion;
056
057    public MultiLabelConfusionMatrix(Model<MultiLabel> model, List<Prediction<MultiLabel>> predictions) {
058        this(model.getOutputIDInfo(), predictions);
059    }
060
061    MultiLabelConfusionMatrix(ImmutableOutputInfo<MultiLabel> domain, List<Prediction<MultiLabel>> predictions) {
062        this.domain = domain;
063        ConfusionMatrixTuple tab = tabulate(domain, predictions);
064        this.mcm = tab.mcm;
065        this.confusion = tab.confusion;
066    }
067
068    @Override
069    public double support(MultiLabel cls) {
070        double total = 0d;
071        for (Label label : cls.getLabelSet()) {
072            int ix = getDomain().getID(new MultiLabel(label));
073            /*
074            mcm[i] =
075            [tn, fn]
076            [fp, tp]
077
078            support = false negatives + true positives
079
080            false neg => ground truth was [label] but we predicted something else
081            true pos  => ground truth was [label] and we predicted [label]
082
083            (whereas: false pos => ground truth was NOT [label] but we predicted [label])
084
085            so
086
087            support = false neg + true pos = mcm[i, 0, 1] + mcm[i, 1, 1] = mcm[i, :, 1].sum()
088             */
089            total += mcm[ix].getColumn(1).sum();
090        }
091        return total;
092    }
093
094    @Override
095    public ImmutableOutputInfo<MultiLabel> getDomain() {
096        return domain;
097    }
098
099    @Override
100    public double support() {
101        double total = 0d;
102        for (int i = 0; i < domain.size(); i++) {
103            total += mcm[i].getColumn(1).sum();
104        }
105        return total;
106    }
107
108    @Override
109    public double tp(MultiLabel cls) {
110        return compute(cls, (cm) -> cm.get(1, 1));
111    }
112
113    @Override
114    public double fp(MultiLabel cls) {
115        return compute(cls, (cm) -> cm.get(0, 1));
116    }
117
118    @Override
119    public double fn(MultiLabel cls) {
120        return compute(cls, (cm) -> cm.get(1, 0));
121    }
122
123    @Override
124    public double tn(MultiLabel cls) {
125        return compute(cls, (cm) -> cm.get(0, 0));
126    }
127
128    private double compute(MultiLabel cls, Function<DenseMatrix, Double> getter) {
129        double total = 0d;
130        for (Label label : cls.getLabelSet()) {
131            int i = domain.getID(new MultiLabel(label.getLabel()));
132            //
133            // When input class is not in the domain, ID will be -1.
134            if (i < 0) {
135                continue;
136            }
137            DenseMatrix cm = mcm[i];
138            total += getter.apply(cm);
139        }
140        return total;
141    }
142
143    @Override
144    public double confusion(MultiLabel predicted, MultiLabel truth) {
145        double total = 0d;
146        Set<Label> trueSet = truth.getLabelSet();
147        Set<Label> predSet = predicted.getLabelSet();
148        for (Label predLabel : predSet) {
149            int idx = domain.getID(new MultiLabel(predLabel.getLabel()));
150            for (Label trueLabel : trueSet) {
151                int jdx = domain.getID(new MultiLabel(trueLabel.getLabel()));
152                total += this.confusion.get(idx, jdx);
153            }
154        }
155        return total;
156    }
157
158    @Override
159    public String toString() {
160        StringBuilder sb = new StringBuilder();
161        sb.append("[");
162        for (int i = 0; i < mcm.length; i++) {
163            DenseMatrix cm = mcm[i];
164            sb.append(cm.toString());
165            sb.append("\n");
166        }
167        sb.append("]");
168        return sb.toString();
169    }
170
171    static ConfusionMatrixTuple tabulate(ImmutableOutputInfo<MultiLabel> domain, List<Prediction<MultiLabel>> predictions) {
172        // this just keeps track of how many times [class x] was predicted to be [class y]
173        DenseMatrix confusion = new DenseMatrix(domain.size(), domain.size());
174
175        DenseMatrix[] mcm = new DenseMatrix[domain.size()];
176        for (int i = 0; i < domain.size(); i++) {
177            mcm[i] = new DenseMatrix(2, 2);
178        }
179
180        for (Prediction<MultiLabel> prediction : predictions) {
181            MultiLabel predictedOutput = prediction.getOutput();
182            MultiLabel trueOutput = prediction.getExample().getOutput();
183
184            Set<Label> trueSet = trueOutput.getLabelSet();
185            Set<Label> predSet = predictedOutput.getLabelSet();
186
187            //
188            // Count true positives and false positives
189            for (Label pred : predSet) {
190                int idx = domain.getID(new MultiLabel(pred.getLabel()));
191                if (trueSet.contains(pred)) {
192                    //
193                    // true positive: mcm[i, 1, 1]++
194                    mcm[idx].add(1, 1, 1d);
195                } else {
196                    //
197                    // false positive: mcm[i, 1, 0]++
198                    mcm[idx].add(1, 0, 1d);
199                }
200            }
201
202            //
203            // Count false negatives and populate the confusion table
204            for (Label trueLabel : trueSet) {
205                int idx = domain.getID(new MultiLabel(trueLabel.getLabel()));
206
207                //
208                // Doing two things in this loop:
209                // 1) Checking if predSet contains trueLabel
210                // 2) Counting the # of times [trueLabel] was predicted to be [predLabel] to populate the confusion table
211                boolean found = false;
212                for (Label predLabel : predSet) {
213                    int jdx = domain.getID(new MultiLabel(predLabel.getLabel()));
214                    confusion.add(jdx, idx, 1d);
215
216                    if (predLabel.equals(trueLabel)) {
217                        found = true;
218                    }
219                }
220
221                if (!found) {
222                    //
223                    // false negative: mcm[i, 0, 1]++
224                    mcm[idx].add(0, 1, 1d);
225                }
226                // else { true positive: already counted }
227            }
228
229            //
230            // True negatives everywhere else
231            for (MultiLabel multilabel : domain.getDomain()) {
232                Set<Label> labels = multilabel.getLabelSet();
233                for (Label label : labels) {
234                    if (!trueSet.contains(label) && !predSet.contains(label)) {
235                        int ix = domain.getID(new MultiLabel(label));
236                        mcm[ix].add(0, 0, 1d);
237                    }
238                }
239            }
240        }
241
242        return new ConfusionMatrixTuple(mcm, confusion);
243    }
244
245    /**
246     * It's a record, ooops not yet, we don't require Java 14.
247     */
248    static final class ConfusionMatrixTuple {
249        final DenseMatrix[] mcm;
250        final DenseMatrix confusion;
251        ConfusionMatrixTuple(DenseMatrix[] mcm, DenseMatrix confusion) {
252            this.mcm = mcm;
253            this.confusion = confusion;
254        }
255
256        DenseMatrix[] getMCM() {
257            return mcm;
258        }
259    }
