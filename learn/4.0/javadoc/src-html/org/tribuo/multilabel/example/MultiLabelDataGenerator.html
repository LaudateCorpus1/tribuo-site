017package org.tribuo.multilabel.example;
018
019import com.oracle.labs.mlrg.olcut.util.Pair;
020import org.tribuo.Dataset;
021import org.tribuo.Example;
022import org.tribuo.Feature;
023import org.tribuo.MutableDataset;
024import org.tribuo.impl.ArrayExample;
025import org.tribuo.multilabel.MultiLabel;
026import org.tribuo.multilabel.MultiLabelFactory;
027import org.tribuo.provenance.DataSourceProvenance;
028import org.tribuo.provenance.SimpleDataSourceProvenance;
029
030import java.time.OffsetDateTime;
031
032/**
033 * Generates three example train and test datasets, used for unit testing.
034 * They don't necessarily have sensible classification boundaries,
035 * it's for testing the machinery rather than accuracy.
036 */
037public class MultiLabelDataGenerator {
038
039    private static final MultiLabelFactory factory = new MultiLabelFactory();
040
041    private MultiLabelDataGenerator() {
042    }
043
044    public static Dataset<MultiLabel> generateTrainData() {
045        DataSourceProvenance provenance = new SimpleDataSourceProvenance("TrainingData", OffsetDateTime.now(), factory);
046        MutableDataset<MultiLabel> dataset = new MutableDataset<>(provenance, factory);
047
048        ArrayExample<MultiLabel> e = new ArrayExample<>(factory.generateOutput("MONKEY"));
049        e.add(new Feature("A-MONKEY",1.0));
050        e.add(new Feature("B-PUZZLE",0.0));
051        e.add(new Feature("C-TREE",0.0));
052        dataset.add(e);
053
054        e = new ArrayExample<>(factory.generateOutput("PUZZLE"));
055        e.add(new Feature("A-MONKEY",0.0));
056        e.add(new Feature("B-PUZZLE",1.0));
057        e.add(new Feature("C-TREE",0.0));
058        dataset.add(e);
059
060        e = new ArrayExample<>(factory.generateOutput("TREE"));
061        e.add(new Feature("A-MONKEY",0.0));
062        e.add(new Feature("B-PUZZLE",0.0));
063        e.add(new Feature("C-TREE",1.0));
064        dataset.add(e);
065
066        e = new ArrayExample<>(factory.generateOutput("MONKEY,TREE"));
067        e.add(new Feature("A-MONKEY",1.0));
068        e.add(new Feature("B-PUZZLE",0.0));
069        e.add(new Feature("C-TREE",1.0));
070        dataset.add(e);
071
072        e = new ArrayExample<>(factory.generateOutput("PUZZLE,MONKEY"));
073        e.add(new Feature("A-MONKEY",1.0));
074        e.add(new Feature("B-PUZZLE",1.0));
075        e.add(new Feature("C-TREE",0.0));
076        dataset.add(e);
077
078        e = new ArrayExample<>(factory.generateOutput("TREE,PUZZLE"));
079        e.add(new Feature("A-MONKEY",0.0));
080        e.add(new Feature("B-PUZZLE",1.0));
081        e.add(new Feature("C-TREE",1.0));
082        dataset.add(e);
083
084        return dataset;
085    }
086
087    public static Dataset<MultiLabel> generateTestData() {
088        DataSourceProvenance provenance = new SimpleDataSourceProvenance("TestingData", OffsetDateTime.now(), factory);
089        MutableDataset<MultiLabel> dataset = new MutableDataset<>(provenance,factory);
090
091        ArrayExample<MultiLabel> e = new ArrayExample<>(factory.generateOutput("MONKEY,PUZZLE,TREE"));
092        e.add(new Feature("A-MONKEY",1.0));
093        e.add(new Feature("B-PUZZLE",1.0));
094        e.add(new Feature("C-TREE",1.0));
095        dataset.add(e);
096
097        e = new ArrayExample<>(factory.generateOutput("MONKEY"));
098        e.add(new Feature("A-MONKEY",1.0));
099        e.add(new Feature("B-PUZZLE",0.0));
100        e.add(new Feature("C-TREE",0.0));
101        dataset.add(e);
102
103        e = new ArrayExample<>(factory.generateOutput("PUZZLE"));
104        e.add(new Feature("A-MONKEY",0.0));
105        e.add(new Feature("B-PUZZLE",1.0));
106        e.add(new Feature("C-TREE",0.0));
107        dataset.add(e);
108
109        e = new ArrayExample<>(factory.generateOutput("TREE"));
110        e.add(new Feature("A-MONKEY",0.0));
111        e.add(new Feature("B-PUZZLE",0.0));
112        e.add(new Feature("C-TREE",1.0));
113        dataset.add(e);
114
115        return dataset;
116    }
117
118    public static Pair<Dataset<MultiLabel>,Dataset<MultiLabel>> generateDataset() {
119        return new Pair<>(generateTrainData(),generateTestData());
120    }
121
122    /**
123     * Generates an example with the feature ids 1,5,8, which does not intersect with the
124     * ids used elsewhere in this class. This should make the example empty at prediction time.
125     * @return An example with features {1:1.0,5:5.0,8:8.0}.
126     */
127    public static Example<MultiLabel> invalidSparseExample() {
128        return new ArrayExample<>(new MultiLabel("MONKEY"),new String[]{"1","5","8"},new double[]{1.0,5.0,8.0});
129    }
130
131    /**
132     * Generates an example with no features.
133     * @return An example with no features.
134     */
135    public static Example<MultiLabel> emptyExample() {
136        return new ArrayExample<>(new MultiLabel("MONKEY"),new String[]{},new double[]{});
137    }
138
139}
