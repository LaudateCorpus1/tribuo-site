017package org.tribuo.multilabel;
018
019import com.oracle.labs.mlrg.olcut.util.MutableLong;
020import com.oracle.labs.mlrg.olcut.util.MutableNumber;
021import com.oracle.labs.mlrg.olcut.util.Pair;
022import org.tribuo.ImmutableOutputInfo;
023import org.tribuo.MutableOutputInfo;
024import org.tribuo.OutputInfo;
025import org.tribuo.classification.Label;
026
027import java.io.IOException;
028import java.util.HashMap;
029import java.util.HashSet;
030import java.util.Iterator;
031import java.util.Map;
032import java.util.Set;
033
034/**
035 * The base class for information about {@link MultiLabel} outputs.
036 */
037public abstract class MultiLabelInfo implements OutputInfo<MultiLabel> {
038    private static final long serialVersionUID = 1L;
039
040    protected final Map<String,MutableLong> labelCounts;
041    protected int unknownCount = 0;
042    protected transient Map<String,MultiLabel> labels;
043
044    protected int totalCount = 0;
045
046    MultiLabelInfo() {
047        labelCounts = new HashMap<>();
048        labels = new HashMap<>();
049    }
050
051    MultiLabelInfo(MultiLabelInfo other) {
052        labelCounts = MutableNumber.copyMap(other.labelCounts);
053        labels = new HashMap<>(other.labels);
054        totalCount = other.totalCount;
055    }
056
057    @Override
058    public int getUnknownCount() {
059        return unknownCount;
060    }
061
062    /**
063     * Returns a set of MultiLabel, where each has a single Label inside it.
064     * The set covers the space of Labels that this MultiLabelInfo has seen.
065     * @return The set of possible labels.
066     */
067    @Override
068    public Set<MultiLabel> getDomain() {
069        return new HashSet<>(labels.values());
070    }
071
072    /**
073     * Get the number of times this Label was observed, or 0 if unknown.
074     * @param label The Label to look for.
075     * @return A non-negative long.
076     */
077    public long getLabelCount(Label label) {
078        MutableLong l = labelCounts.get(label.getLabel());
079        if (l != null) {
080            return l.longValue();
081        } else {
082            return 0;
083        }
084    }
085
086    /**
087     * Get the number of times this String was observed, or 0 if unknown.
088     * @param label The String to look for.
089     * @return A non-negative long.
090     */
091    public long getLabelCount(String label) {
092        MutableLong l = labelCounts.get(label);
093        if (l != null) {
094            return l.longValue();
095        } else {
096            return 0;
097        }
098    }
099
100    @Override
101    public Iterable<Pair<String,Long>> outputCountsIterable() {
102        return () -> new Iterator<Pair<String, Long>>() {
103            Iterator<Map.Entry<String, MutableLong>> itr = labelCounts.entrySet().iterator();
104
105            @Override
106            public boolean hasNext() {
107                return itr.hasNext();
108            }
109
110            @Override
111            public Pair<String, Long> next() {
112                Map.Entry<String, MutableLong> e = itr.next();
113                return new Pair<>(e.getKey(), e.getValue().longValue());
114            }
115        };
116    }
117
118    @Override
119    public int size() {
120        return labelCounts.size();
121    }
122
123    @Override
124    public ImmutableOutputInfo<MultiLabel> generateImmutableOutputInfo() {
125        return new ImmutableMultiLabelInfo(this);
126    }
127
128    @Override
129    public MutableOutputInfo<MultiLabel> generateMutableOutputInfo() {
130        return new MutableMultiLabelInfo(this);
131    }
132
133    @Override
134    public abstract MultiLabelInfo copy();
135
136    private void readObject(java.io.ObjectInputStream in) throws IOException, ClassNotFoundException {
137        in.defaultReadObject();
138        labels = new HashMap<>();
139        for (Map.Entry<String,MutableLong> e : labelCounts.entrySet()) {
140            labels.put(e.getKey(),new MultiLabel(e.getKey()));
141        }
142    }
143}
