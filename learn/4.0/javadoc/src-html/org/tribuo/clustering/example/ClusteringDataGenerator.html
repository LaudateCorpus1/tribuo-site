038/**
039 * Generates three example train and test datasets, used for unit testing.
040 * They don't necessarily have sensible cluster boundaries,
041 * it's for testing the machinery rather than accuracy.
042 * <p>
043 * Also has a dataset generator which returns a dataset
044 * sampled from a mixture of 2 dimensional gaussians.
045 */
046public abstract class ClusteringDataGenerator {
047
048    private static ClusteringFactory clusteringFactory = new ClusteringFactory();
049
050    /**
051     * Generates a dataset drawn from a mixture of 5 2d gaussians.
052     *
053     * @param size The number of points to sample for the dataset.
054     * @param seed The RNG seed.
055     * @return A pair of datasets.
056     */
057    public static Dataset<ClusterID> gaussianClusters(long size, long seed) {
058        if (size < 1) {
059            throw new IllegalArgumentException("Size must be a positive number, received " + size);
060        }
061        Random rng = new Random(seed);
062
063        // Dataset parameters
064        String[] featureNames = new String[]{"A","B"};
065        double[] mixingPMF = new double[]{0.1,0.35,0.05,0.25,0.25};
066        double[] mixingCDF = Util.generateCDF(mixingPMF);
067        MultivariateNormalDistribution first = new MultivariateNormalDistribution(new JDKRandomGenerator(rng.nextInt()),
068                new double[]{0.0,0.0}, new double[][]{{1.0,0.0},{0.0,1.0}}
069        );
070        MultivariateNormalDistribution second = new MultivariateNormalDistribution(new JDKRandomGenerator(rng.nextInt()),
071                new double[]{5.0,5.0}, new double[][]{{1.0,0.0},{0.0,1.0}}
072        );
073        MultivariateNormalDistribution third = new MultivariateNormalDistribution(new JDKRandomGenerator(rng.nextInt()),
074                new double[]{2.5,2.5}, new double[][]{{1.0,0.5},{0.5,1.0}}
075        );
076        MultivariateNormalDistribution fourth = new MultivariateNormalDistribution(new JDKRandomGenerator(rng.nextInt()),
077                new double[]{10.0,0.0}, new double[][]{{0.1,0.0},{0.0,0.1}}
078        );
079        MultivariateNormalDistribution fifth = new MultivariateNormalDistribution(new JDKRandomGenerator(rng.nextInt()),
080                new double[]{-1.0,0.0}, new double[][]{{1.0,0.0},{0.0,0.1}}
081        );
082        MultivariateNormalDistribution[] gaussians = new MultivariateNormalDistribution[]{first,second,third,fourth,fifth};
083
084        List<Example<ClusterID>> trainingData = new ArrayList<>();
085        for (int i = 0; i < size; i++) {
086            int centroid = Util.sampleFromCDF(mixingCDF,rng);
087            double[] sample = gaussians[centroid].sample();
088            trainingData.add(new ArrayExample<>(new ClusterID(centroid),featureNames,sample));
089        }
090
091        SimpleDataSourceProvenance trainingProvenance = new SimpleDataSourceProvenance("Generated clustering data",clusteringFactory);
092        return new MutableDataset<>(new ListDataSource<>(trainingData,clusteringFactory,trainingProvenance));
093    }

099    /**
100     * Generates a train/test dataset pair which is dense in the features,
101     * each example has 4 features,{A,B,C,D}, and there are 4 clusters,
102     * {0,1,2,3}.
103     * @param negate Supply -1.0 to negate some feature values.
104     * @return A pair of datasets.
105     */
106    public static Pair<Dataset<ClusterID>,Dataset<ClusterID>> denseTrainTest(double negate) {
107        MutableDataset<ClusterID> train = new MutableDataset<>(new SimpleDataSourceProvenance("TrainingData", OffsetDateTime.now(), clusteringFactory), clusteringFactory);
108
109        String[] names = new String[]{"A","B","C","D"};
110        double[] values = new double[]{1.0,0.5,1.0,negate*1.0};
111        train.add(new ArrayExample<>(new ClusterID(1),names,values));
112        values = new double[]{1.5,0.35,1.3,negate*1.2};
113        train.add(new ArrayExample<>(new ClusterID(1),names.clone(),values));
114        values = new double[]{1.2,0.45,1.5,negate*1.0};
115        train.add(new ArrayExample<>(new ClusterID(1),names.clone(),values));
116
117        values = new double[]{negate*1.1,0.55,negate*1.5,0.5};
118        train.add(new ArrayExample<>(new ClusterID(2),names.clone(),values));
119        values = new double[]{negate*1.5,0.25,negate*1,0.125};
120        train.add(new ArrayExample<>(new ClusterID(2),names.clone(),values));
121        values = new double[]{negate*1,0.5,negate*1.123,0.123};
122        train.add(new ArrayExample<>(new ClusterID(2),names.clone(),values));
123
124        values = new double[]{1.5,5.0,0.5,4.5};
125        train.add(new ArrayExample<>(new ClusterID(3),names.clone(),values));
126        values = new double[]{1.234,5.1235,0.1235,6.0};
127        train.add(new ArrayExample<>(new ClusterID(3),names.clone(),values));
128        values = new double[]{1.734,4.5,0.5123,5.5};
129        train.add(new ArrayExample<>(new ClusterID(3),names.clone(),values));
130
131        values = new double[]{negate*1,0.25,5,10.0};
132        train.add(new ArrayExample<>(new ClusterID(0),names.clone(),values));
133        values = new double[]{negate*1.4,0.55,5.65,12.0};
134        train.add(new ArrayExample<>(new ClusterID(0),names.clone(),values));
135        values = new double[]{negate*1.9,0.25,5.9,15};
136        train.add(new ArrayExample<>(new ClusterID(0),names.clone(),values));
137
138        MutableDataset<ClusterID> test = new MutableDataset<>(new SimpleDataSourceProvenance("TestingData", OffsetDateTime.now(),clusteringFactory), clusteringFactory);
139
140        values = new double[]{2.0,0.45,3.5,negate*2.0};
141        test.add(new ArrayExample<>(new ClusterID(1),names.clone(),values));
142        values = new double[]{negate*2.0,0.55,negate*2.5,2.5};
143        test.add(new ArrayExample<>(new ClusterID(2),names.clone(),values));
144        values = new double[]{1.75,5.0,1.0,6.5};
145        test.add(new ArrayExample<>(new ClusterID(3),names.clone(),values));
146        values = new double[]{negate*1.5,0.25,5.0,20.0};
147        test.add(new ArrayExample<>(new ClusterID(0),names.clone(),values));
148
149        return new Pair<>(train,test);
150    }

224    /**

226     * ids used elsewhere in this class. This should make the example empty at prediction time.
