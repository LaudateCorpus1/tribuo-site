041import java.io.BufferedInputStream;

051/**
052 * Options for working with training and test data in a CLI.
053 */
054public final class DataOptions implements Options {
055    private static final Logger logger = Logger.getLogger(DataOptions.class.getName());
056
057    public enum InputFormat {
058        SERIALIZED, LIBSVM, TEXT, CSV, COLUMNAR
059    }
060
061    public enum Delimiter {
062        COMMA(','), TAB('\t'), SEMICOLON(';');
063
064        public final char value;
065        Delimiter(char value) {
066            this.value = value;
067        }
068    }
069
070    @Override
071    public String getOptionsDescription() {
072        return "Options for loading and processing train and test data.";
073    }
074
075    @Option(longName="hashing-dimension",usage="Hashing dimension used for standard text format.")
076    public int hashDim = 0;
077    @Option(longName="ngram",usage="Ngram size to generate when using standard text format.")
078    public int ngram = 2;
079    @Option(longName="term-counting",usage="Use term counts instead of boolean when using the standard text format.")
080    public boolean termCounting;
081    @Option(charName='f',longName="model-output-path",usage="Path to serialize model to.")
082    public Path outputPath;
083    @Option(charName='r',longName="seed",usage="RNG seed.")
084    public long seed = Trainer.DEFAULT_SEED;
085    @Option(charName='s',longName="input-format",usage="Loads the data using the specified format.")
086    public InputFormat inputFormat = InputFormat.LIBSVM;
087    @Option(longName="csv-response-name",usage="Response name in the csv file.")
088    public String csvResponseName;
089    @Option(longName="csv-delimiter",usage="Delimiter")
090    public Delimiter delimiter = Delimiter.COMMA;
091    @Option(longName="csv-quote-char",usage="Quote character in the CSV file.")
092    public char csvQuoteChar = '"';
093    @Option(longName="columnar-row-processor",usage="The name of the row processor from the config file.")
094    public RowProcessor<?> rowProcessor;
095    @Option(longName="min-count",usage="Minimum cardinality of the features.")
096    public int minCount = 0;
097    @Option(charName='u',longName="training-file",usage="Path to the training file.")
098    public Path trainingPath;
099    @Option(charName='v',longName="testing-file",usage="Path to the testing file.")
100    public Path testingPath;
101
102    public <T extends Output<T>> Pair<Dataset<T>,Dataset<T>> load(OutputFactory<T> outputFactory) throws IOException {
103        logger.info(String.format("Loading data from %s", trainingPath));
104        Dataset<T> train;
105        Dataset<T> test;
106        char separator;
107        switch (inputFormat) {
108            case SERIALIZED:
109                //
110                // Load Tribuo serialised datasets.
111                logger.info("Deserialising dataset from " + trainingPath);
112                try (ObjectInputStream ois = new ObjectInputStream(new BufferedInputStream(new FileInputStream(trainingPath.toFile())));
113                     ObjectInputStream oits = new ObjectInputStream(new BufferedInputStream(new FileInputStream(testingPath.toFile())))) {
114                    @SuppressWarnings("unchecked")
115                    Dataset<T> tmp = (Dataset<T>) ois.readObject();
116                    train = tmp;
117                    if (minCount > 0) {
118                        logger.info("Found " + train.getFeatureIDMap().size() + " features");
119                        logger.info("Removing features that occur fewer than " + minCount + " times.");
120                        train = new MinimumCardinalityDataset<>(train,minCount);
121                    }
122                    logger.info(String.format("Loaded %d training examples for %s", train.size(), train.getOutputs().toString()));
123                    logger.info("Found " + train.getFeatureIDMap().size() + " features, and " + train.getOutputInfo().size() + " response dimensions");
124                    @SuppressWarnings("unchecked")
125                    Dataset<T> deserTest = (Dataset<T>) oits.readObject();
126                    test = new ImmutableDataset<>(deserTest,deserTest.getSourceProvenance(),deserTest.getOutputFactory(),train.getFeatureIDMap(),train.getOutputIDInfo(),true);
127                } catch (ClassNotFoundException e) {
128                    throw new IllegalArgumentException("Unknown class in serialised files", e);
129                }
130                break;
131            case LIBSVM:
132                //
133                // Load the libsvm text-based data format.
134                LibSVMDataSource<T> trainSVMSource = new LibSVMDataSource<>(trainingPath,outputFactory);
135                train = new MutableDataset<>(trainSVMSource);
136                boolean zeroIndexed = trainSVMSource.isZeroIndexed();
137                int maxFeatureID = trainSVMSource.getMaxFeatureID();
138                if (minCount > 0) {
139                    logger.info("Removing features that occur fewer than " + minCount + " times.");
140                    train = new MinimumCardinalityDataset<>(train,minCount);
141                }
142                logger.info(String.format("Loaded %d training examples for %s", train.size(), train.getOutputs().toString()));
143                logger.info("Found " + train.getFeatureIDMap().size() + " features, and " + train.getOutputInfo().size() + " response dimensions");
144                test = new ImmutableDataset<>(new LibSVMDataSource<>(testingPath,outputFactory,zeroIndexed,maxFeatureID), train.getFeatureIDMap(), train.getOutputIDInfo(), false);
145                break;
146            case TEXT:
147                //
148                // Using a simple Java break iterator to generate ngram features.
149                TextFeatureExtractor<T> extractor;
150                if (hashDim > 0) {
151                    extractor = new TextFeatureExtractorImpl<>(new TokenPipeline(new BreakIteratorTokenizer(Locale.US), ngram, termCounting, hashDim));
152                } else {
153                    extractor = new TextFeatureExtractorImpl<>(new TokenPipeline(new BreakIteratorTokenizer(Locale.US), ngram, termCounting));
154                }
155
156                TextDataSource<T> trainSource = new SimpleTextDataSource<>(trainingPath, outputFactory, extractor);
157                train = new MutableDataset<>(trainSource);
158                if (minCount > 0) {
159                    logger.info("Removing features that occur fewer than " + minCount + " times.");
160                    train = new MinimumCardinalityDataset<>(train,minCount);
161                }
162
163                logger.info(String.format("Loaded %d training examples for %s", train.size(), train.getOutputs().toString()));
164                logger.info("Found " + train.getFeatureIDMap().size() + " features, and " + train.getOutputInfo().size() + " response dimensions");
165
166                TextDataSource<T> testSource = new SimpleTextDataSource<>(testingPath, outputFactory, extractor);
167                test = new ImmutableDataset<>(testSource, train.getFeatureIDMap(), train.getOutputIDInfo(), false);
168                break;
169            case CSV:
170                //
171                // Load the data using the simple CSV loader
172                if (csvResponseName == null) {
173                    throw new IllegalArgumentException("Please supply a response column name");
174                }
175                separator = delimiter.value;
176                CSVLoader<T> loader = new CSVLoader<>(separator,outputFactory);
177                train = new MutableDataset<>(loader.loadDataSource(trainingPath,csvResponseName));
178                logger.info(String.format("Loaded %d training examples for %s", train.size(), train.getOutputs().toString()));
179                logger.info("Found " + train.getFeatureIDMap().size() + " features, and " + train.getOutputInfo().size() + " response dimensions");
180                test = new MutableDataset<>(loader.loadDataSource(testingPath,csvResponseName));
181                break;
182            case COLUMNAR:
183                if (rowProcessor == null) {
184                    throw new IllegalArgumentException("Please supply a RowProcessor");
185                }
186                OutputFactory<?> rowOutputFactory = rowProcessor.getResponseProcessor().getOutputFactory();
187                if (!rowOutputFactory.equals(outputFactory)) {
188                    throw new IllegalArgumentException("The RowProcessor doesn't use the same kind of OutputFactory as the one supplied. RowProcessor has " + rowOutputFactory.getClass().getSimpleName() + ", supplied " + outputFactory.getClass().getName());
189                }
190                @SuppressWarnings("unchecked") // checked by the if statement above
191                RowProcessor<T> typedRowProcessor = (RowProcessor<T>) rowProcessor;
192                separator = delimiter.value;
193                train = new MutableDataset<>(new CSVDataSource<>(trainingPath,typedRowProcessor,true,separator,csvQuoteChar));
194                logger.info(String.format("Loaded %d training examples for %s", train.size(), train.getOutputs().toString()));
195                logger.info("Found " + train.getFeatureIDMap().size() + " features, and " + train.getOutputInfo().size() + " response dimensions");
196                test = new MutableDataset<>(new CSVDataSource<>(testingPath,typedRowProcessor,true,separator,csvQuoteChar));
197                break;
198            default:
199                throw new IllegalArgumentException("Unsupported input format " + inputFormat);
200        }
201        logger.info(String.format("Loaded %d testing examples", test.size()));
202        return new Pair<>(train,test);
203    }
204
205    public <T extends Output<T>> void saveModel(Model<T> model) throws IOException {
206        FileOutputStream fout = new FileOutputStream(outputPath.toFile());
207        ObjectOutputStream oout = new ObjectOutputStream(fout);
208        oout.writeObject(model);
209        oout.close();
210        fout.close();
211        logger.info("Serialized model to file: " + outputPath);
212    }
213}
