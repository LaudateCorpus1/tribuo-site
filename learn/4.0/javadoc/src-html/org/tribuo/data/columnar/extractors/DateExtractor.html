017package org.tribuo.data.columnar.extractors;
018
019import com.oracle.labs.mlrg.olcut.config.Config;
020import com.oracle.labs.mlrg.olcut.provenance.ConfiguredObjectProvenance;
021import com.oracle.labs.mlrg.olcut.provenance.impl.ConfiguredObjectProvenanceImpl;
022
023import java.time.LocalDate;
024import java.time.format.DateTimeFormatter;
025import java.time.format.DateTimeParseException;
026import java.util.Optional;
027import java.util.logging.Level;
028import java.util.logging.Logger;
029
030/**
031 * Extracts the field value and translates it to a {@link LocalDate} based on the specified {@link DateTimeFormatter}.
032 * <p>
033 * Returns an empty optional if the date failed to parse.
034 */
035public class DateExtractor extends SimpleFieldExtractor<LocalDate> {
036    private static final Logger logger = Logger.getLogger(DateExtractor.class.getName());
037
038    @Config(mandatory = true, description = "The expected date format.")
039    private String dateFormat;
040    private DateTimeFormatter formatter;
041
042    /**
043     * for olcut
044     */
045    private DateExtractor() {}
046
047    public DateExtractor(String fieldName, String metadataName, String dateFormat) {
048        super(fieldName, metadataName);
049        this.dateFormat = dateFormat;
050        postConfig();
051    }
052
053    public DateExtractor(String fieldName, String metadataName, DateTimeFormatter formatter) {
054        super(fieldName, metadataName);
055        this.formatter = formatter;
056    }
057
058    @Override
059    public void postConfig() {
060        if (dateFormat != null) {
061            formatter = DateTimeFormatter.ofPattern(dateFormat);
062        } else {
063            formatter = DateTimeFormatter.BASIC_ISO_DATE;
064        }
065        if (metadataName == null || metadataName.isEmpty()) {
066            metadataName = fieldName;
067        }
068    }
069
070    @Override
071    public Class<LocalDate> getValueType() {
072        return LocalDate.class;
073    }
074
075    @Override
076    protected Optional<LocalDate> extractField(String s) {
077        try {
078            return Optional.of(LocalDate.parse(s, formatter));
079        } catch (DateTimeParseException e) {
080            logger.log(Level.WARNING, e.getParsedString());
081            logger.log(Level.WARNING, String.format("Unable to parse date %s with formatter %s", s, formatter.toString()));
082            return Optional.empty();
083        }
084    }
085
086    @Override
087    public String toString() {
088        return "DateExtractor(fieldName=" + fieldName + ", metadataName=" + metadataName + ", dateFormat=" + formatter.toString() + ")";
089    }
090
091    @Override
092    public ConfiguredObjectProvenance getProvenance() {
093        return new ConfiguredObjectProvenanceImpl(this, "DateExtractor");
094    }
095}
