046/**
047 * A processor which takes a Map of String to String and returns an {@link Example}.
048 * <p>
049 * It accepts a {@link ResponseProcessor} which converts the response field into an {@link Output},
050 * a Map of {@link FieldProcessor}s which converts fields into {@link ColumnarFeature}s, and a Set
051 * of {@link FeatureProcessor}s which processes the list of {@link ColumnarFeature}s before {@link Example}
052 * construction. Optionally metadata and weights can be extracted using {@link FieldExtractor}s
053 * and written into each example as they are constructed.
054 * <p>
055 * If the metadata extractors are invalid (i.e. two extractors write to the same metadata key),
056 * the RowProcessor throws {@link PropertyException}.
057 */
058public class RowProcessor<T extends Output<T>> implements Configurable, Provenancable<ConfiguredObjectProvenance> {
059
060    private static final Logger logger = Logger.getLogger(RowProcessor.class.getName());
061
062    private static final String FEATURE_NAME_REGEX = "["+ColumnarFeature.JOINER+FieldProcessor.NAMESPACE+"]";
063
064    private static final Pattern FEATURE_NAME_PATTERN = Pattern.compile(FEATURE_NAME_REGEX);
065
066
067    @Config(description="Extractors for the example metadata.")
068    private List<FieldExtractor<?>> metadataExtractors = Collections.emptyList();
069
070    @Config(description="Extractor for the example weight.")
071    protected FieldExtractor<Float> weightExtractor = null;
072
073    @Config(mandatory = true,description="Processor which extracts the response.")
074    protected ResponseProcessor<T> responseProcessor;
075
076    @Config(mandatory = true,description="The list of field processors to use.")
077    private List<FieldProcessor> fieldProcessorList;
078
079    // fieldProcessorList is unpacked into this map to make the config files less complex.
080    // fieldProcessorMap is the store of record for field processors.
081    protected Map<String,FieldProcessor> fieldProcessorMap;
082
083    @Config(description="A set of feature processors to apply after extraction.")
084    private Set<FeatureProcessor> featureProcessors = new HashSet<>();
085
086    @Config(description="A map from a regex to field processors to apply to fields matching the regex.")
087    protected Map<String,FieldProcessor> regexMappingProcessors = new HashMap<>();
088
089    protected boolean configured;

099    public RowProcessor(List<FieldExtractor<?>> metadataExtractors, FieldExtractor<Float> weightExtractor,
100                        ResponseProcessor<T> responseProcessor, Map<String,FieldProcessor> fieldProcessorMap,

104
105    public RowProcessor(List<FieldExtractor<?>> metadataExtractors, FieldExtractor<Float> weightExtractor,
106                        ResponseProcessor<T> responseProcessor, Map<String,FieldProcessor> fieldProcessorMap,
107                        Map<String,FieldProcessor> regexMappingProcessors, Set<FeatureProcessor> featureProcessors) {
108        this.metadataExtractors = metadataExtractors.isEmpty() ? Collections.emptyList() : new ArrayList<>(metadataExtractors);
109        this.weightExtractor = weightExtractor;
110        this.responseProcessor = responseProcessor;
111        this.fieldProcessorMap = new HashMap<>(fieldProcessorMap);
112        this.regexMappingProcessors = regexMappingProcessors.isEmpty() ? Collections.emptyMap() : new HashMap<>(regexMappingProcessors);
113        this.featureProcessors.addAll(featureProcessors);
114        postConfig();
115    }

122    @Override
123    public void postConfig() {
124        configured = regexMappingProcessors.isEmpty();
125        if (fieldProcessorList != null) {
126            fieldProcessorMap = fieldProcessorList.stream().collect(Collectors.toMap(FieldProcessor::getFieldName, Function.identity()));
127        } else {
128            fieldProcessorList = new ArrayList<>();
129            fieldProcessorList.addAll(fieldProcessorMap.values());
130        }
131        Set<String> metadataNames = new HashSet<>();
132        for (FieldExtractor<?> extractor : metadataExtractors) {
133            String newMetadataName = extractor.getMetadataName();
134            if (metadataNames.contains(newMetadataName)) {
135                throw new PropertyException("","metadataExtractors",
136                        "Two metadata extractors found referencing the same metadata name '" + newMetadataName + "'");
137            } else {
138                metadataNames.add(newMetadataName);
139            }
140        }
141    }

167    /**
168     * Generate an {@link Example} from the supplied row. Returns an empty Optional if
169     * there are no features, or the response is required but it was not found. The latter case is
170     * used at training time.
171     * @param row The row to process.
172     * @param outputRequired If an Output must be found in the row to return an Example.
173     * @return An Optional containing an Example if the row was valid, an empty Optional otherwise.
174     */
175    public Optional<Example<T>> generateExample(ColumnarIterator.Row row, boolean outputRequired) {
176        String responseValue = row.getRowData().get(responseProcessor.getFieldName());
177        Optional<T> labelOpt = responseProcessor.process(responseValue);
178        if (!labelOpt.isPresent() && outputRequired) {
179            return Optional.empty();
180        }
181
182        List<ColumnarFeature> features = generateFeatures(row.getRowData());
183
184        if (features.isEmpty()) {
185            logger.warning(String.format("Row %d empty of features, omitting", row.getIndex()));
186            return Optional.empty();
187        } else {
188            T label = labelOpt.orElse(responseProcessor.getOutputFactory().getUnknownOutput());
189
190            Map<String,Object> metadata = generateMetadata(row);
191
192            Example<T> example;
193            if (weightExtractor == null) {
194                example = new ArrayExample<>(label,metadata);
195            } else {
196                example = new ArrayExample<>(label,
197                        weightExtractor.extract(row).orElse(Example.DEFAULT_WEIGHT),
198                        metadata);
199            }
200            example.addAll(features);
201            return Optional.of(example);
202        }
203    }

205    /**
206     * Generate an {@link Example} from the supplied row. Returns an empty Optional if
207     * there are no features, or the response is required but it was not found.
208     * <p>
209     * Supplies -1 as the example index, used in cases where the index isn't meaningful.
210     * @param row The row to process.
211     * @param outputRequired If an Output must be found in the row to return an Example.
212     * @return An Optional containing an Example if the row was valid, an empty Optional otherwise.
213     */
214    public Optional<Example<T>> generateExample(Map<String,String> row, boolean outputRequired) {
215        return generateExample(-1,row,outputRequired);
216    }

218    /**
219     * Generate an {@link Example} from the supplied row. Returns an empty Optional if
220     * there are no features, or the response is required but it was not found. The latter case is
221     * used at training time.
222     * @param idx The index for use in the example metadata if desired.
223     * @param row The row to process.
224     * @param outputRequired If an Output must be found in the row to return an Example.
225     * @return An Optional containing an Example if the row was valid, an empty Optional otherwise.
226     */
227    public Optional<Example<T>> generateExample(long idx, Map<String,String> row, boolean outputRequired) {
228        return generateExample(new ColumnarIterator.Row(idx, new ArrayList<>(row.keySet()), row), outputRequired);
229    }

231    /**
232     * Generates the example metadata from the supplied row and index.
233     * @param row The row to process.
234     * @return A (possibly empty) map containing the metadata.
235     */
236    public Map<String,Object> generateMetadata(ColumnarIterator.Row row) {
237        if (metadataExtractors.isEmpty()) {
238            return Collections.emptyMap();
239        } else {
240            Map<String,Object> metadataMap = new HashMap<>();
241            long idx = row.getIndex();
242
243            for (FieldExtractor<?> field : metadataExtractors) {
244                String metadataName = field.getMetadataName();
245                Optional<?> extractedValue = field.extract(row);
246                if(extractedValue.isPresent()) {
247                    metadataMap.put(metadataName, extractedValue.get());
248                } else {
249                    logger.warning("Failed to extract field with name " + metadataName + " from index " + idx);
250                }
251            }
252
253            return metadataMap;
254        }
255    }
256
257    /**
258     * Generates the features from the supplied row.
259     * @param row The row to process.
260     * @return A (possibly empty) list of {@link ColumnarFeature}s.
261     */
262    public List<ColumnarFeature> generateFeatures(Map<String,String> row) {
263        if (!configured) {
264            throw new IllegalStateException("expandRegexMapping not called, yet there are fieldProcessorMap which have not been bound to a field name.");
265        }
266        List<ColumnarFeature> features = new ArrayList<>();
267
268        for (Map.Entry<String,FieldProcessor> e : fieldProcessorMap.entrySet()) {
269            String value = row.get(e.getKey());
270            if (value != null) {
271                value = value.replace('\n', ' ').trim();
272                features.addAll(e.getValue().process(value));
273            }
274        }
275
276        for (FeatureProcessor f : featureProcessors) {
277            features = f.process(features);
278        }
279
280        return features;
281    }

295    public String getDescription() {
296        String weightExtractorStr = weightExtractor == null ? "null" : weightExtractor.toString();
297        if (configured || regexMappingProcessors.isEmpty()) {
298            return "RowProcessor(responseProcessor=" + responseProcessor.toString() +
299                    ",fieldProcessorMap=" + fieldProcessorMap.toString() +
300                    ",featureProcessors=" + featureProcessors.toString() +
301                    ",metadataExtractors=" + metadataExtractors.toString() +
302                    ",weightExtractor=" + weightExtractorStr + ")";
303        } else {
304            return "RowProcessor(responseProcessor=" + responseProcessor.toString() +
305                    ",fieldProcessorMap=" + fieldProcessorMap.toString() +
306                    ",regexMappingProcessors=" + regexMappingProcessors.toString() +
307                    ",featureProcessors=" + featureProcessors.toString() +
308                    ",metadataExtractors=" + metadataExtractors.toString() +
309                    ",weightExtractor=" + weightExtractorStr + ")";
310        }
311    }

319     * Returns the metadata keys and value types that are extracted
320     * by this RowProcessor.
321     * @return The metadata keys and value types.
322     */
323    public Map<String,Class<?>> getMetadataTypes() {
324        if (metadataExtractors.isEmpty()) {
325            return Collections.emptyMap();
326        } else {
327            Map<String, Class<?>> types = new HashMap<>();
328
329            for (FieldExtractor<?> extractor : metadataExtractors) {
330                types.put(extractor.getMetadataName(), extractor.getValueType());
331            }
332
333            return types;
334        }

345    /**
346     * Uses similar logic to {@link org.tribuo.transform.TransformationMap#validateTransformations} to check the regexes
347     * against the {@link ImmutableFeatureMap} contained in the supplied {@link Model}.
348     * Throws an IllegalArgumentException if any regexes overlap with
349     * themselves, or with the currently defined set of fieldProcessorMap.
350     * @param model The model to use to expand the regexes.
351     */
352    public void expandRegexMapping(Model<T> model) {
353        expandRegexMapping(model.getFeatureIDMap());
354    }
355
356    /**
357     * Uses similar logic to {@link org.tribuo.transform.TransformationMap#validateTransformations} to check the regexes
358     * against the supplied feature map. Throws an IllegalArgumentException if any regexes overlap with
359     * themselves, or with the currently defined set of fieldProcessorMap.
360     * @param featureMap The feature map to use to expand the regexes.
361     */
362    public void expandRegexMapping(ImmutableFeatureMap featureMap) {
363        ArrayList<String> fieldNames = new ArrayList<>(featureMap.size());
364
365        for (VariableInfo v : featureMap) {
366            String[] split = FEATURE_NAME_PATTERN.split(v.getName(),1);
367            String fieldName = split[0];
368            fieldNames.add(fieldName);
369        }
370
371        expandRegexMapping(fieldNames);
372    }
373
374    /**
375     * Uses similar logic to {@link org.tribuo.transform.TransformationMap#validateTransformations} to check the regexes
376     * against the supplied list of field names. Throws an IllegalArgumentException if any regexes overlap with
377     * themselves, or with the currently defined set of fieldProcessorMap or if there are unmatched regexes after
378     * processing.
379     * @param fieldNames The list of field names.
380     */
381    public void expandRegexMapping(Collection<String> fieldNames) {
382        if (configured) {
383            logger.warning("RowProcessor was already configured, yet expandRegexMapping was called with " + fieldNames.toString());
384        }
385        Set<String> foundFields = partialExpandRegexMapping(fieldNames);
386
387        if (foundFields.size() != regexMappingProcessors.size()) {
388            throw new IllegalArgumentException("Failed to match all the regexes, found " + foundFields.size() + ", required " + regexMappingProcessors.size());
389        } else {
390            regexMappingProcessors.clear();
391            configured = true;
392        }
393    }
394
395    /**
396     * Caveat Implementor! This method contains the logic of {@link org.tribuo.data.columnar.RowProcessor#expandRegexMapping}
397     * without any of the checks that ensure the RowProcessor is in a valid state. This can be used in a subclass to expand a regex mapping
398     * several times for a single instance of RowProcessor. The caller is responsible for ensuring that fieldNames are not duplicated
399     * within or between calls.
400     * @param fieldNames The list of field names - should contain only previously unseen field names.
401     * @return the set of regexes that were matched by fieldNames.
402     */
403    protected Set<String> partialExpandRegexMapping(Collection<String> fieldNames) {
404        HashSet<String> foundFields = new HashSet<>();
405        // Loop over all regexes
406        for (Map.Entry<String,FieldProcessor> e : regexMappingProcessors.entrySet()) {
407            Pattern p = Pattern.compile(e.getKey());
408            // Loop over all field names
409            for (String s : fieldNames) {
410                // Check if the pattern matches the field name
411                if (p.matcher(s).matches()) {
412                    // If it matches, add the field to the fieldProcessorMap map
413                    FieldProcessor f = fieldProcessorMap.put(s,e.getValue().copy(s));
414
415                    if (f != null) {
416                        throw new IllegalArgumentException("Regex " + p.toString() + " matched field " + s + " which already had a field processor " + f.toString());
417                    }
418
419                    foundFields.add(e.getKey());
420                }
421            }
422        }
423        return foundFields;
424    }
