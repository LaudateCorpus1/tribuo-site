017package org.tribuo.data.columnar;
018
019import com.oracle.labs.mlrg.olcut.config.Configurable;
020import com.oracle.labs.mlrg.olcut.provenance.ConfiguredObjectProvenance;
021import com.oracle.labs.mlrg.olcut.provenance.Provenancable;
022
023import java.util.List;
024
025/**
026 * An interface for things that process the columns in a data set.
027 */
028public interface FieldProcessor extends Configurable, Provenancable<ConfiguredObjectProvenance> {
029
030    public static final String NAMESPACE = "#";
031
032    /**
033     * The types of generated features.
034     */
035    public enum GeneratedFeatureType {
036        /**
037         * Categoricals binarised into separate features.
038         */
039        BINARISED_CATEGORICAL,
040        /**
041         * Categorical features with the values converted into doubles.
042         */
043        CATEGORICAL,
044        /**
045         * Real valued features.
046         */
047        REAL,
048        /**
049         * Text features.
050         */
051        TEXT
052    }
053
054    /**
055     * Gets the field name this FieldProcessor uses.
056     * @return The field name.
057     */
058    public String getFieldName();
059
060    /**
061     * Processes the field value and generates a (possibly empty) list of {@link ColumnarFeature}s.
062     * @param value The field value to process.
063     * @return A list of {@link ColumnarFeature}s.
064     */
065    public List<ColumnarFeature> process(String value);
066
067    /**
068     * Returns the feature type this FieldProcessor generates.
069     * @return The feature type.
070     */
071    public GeneratedFeatureType getFeatureType();
072
073    /**
074     * Binarised categoricals can be namespaced, where the field name is appended with "#&lt;non-negative-int&gt;" to denote the
075     * namespace. This allows one FieldProcessor to emit multiple binarised categoricals from the same field value,
076     * provided each emitted feature is in a different namespace. Without this guarantee it's impossible to
077     * recover the original categorical distribution before binarisation.
078     *
079     * If there is only a single namespace, it is omitted from the feature name.
080     * @return The number of namespaces.
081     */
082    default public int getNumNamespaces() {
083        return 1;
084    }
085
086    /**
087     * Returns a copy of this FieldProcessor bound to the supplied newFieldName.
088     * @param newFieldName The new field name for the copy.
089     * @return A copy of this FieldProcessor.
090     */
091    public FieldProcessor copy(String newFieldName);
092}
