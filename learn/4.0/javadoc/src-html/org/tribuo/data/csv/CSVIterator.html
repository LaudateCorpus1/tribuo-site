017package org.tribuo.data.csv;
018
019import com.opencsv.CSVReader;
020import com.opencsv.CSVReaderBuilder;
021import com.opencsv.RFC4180ParserBuilder;
022import com.opencsv.exceptions.CsvValidationException;
023import org.tribuo.data.columnar.ColumnarIterator;
024
025import java.io.IOException;
026import java.io.Reader;
027import java.net.URI;
028import java.nio.file.Files;
029import java.nio.file.Paths;
030import java.util.Arrays;
031import java.util.Collections;
032import java.util.HashMap;
033import java.util.List;
034import java.util.Map;
035import java.util.Optional;
036import java.util.logging.Level;
037import java.util.logging.Logger;
038
039/**
040 * An iterator over a CSV file.
041 */
042public class CSVIterator extends ColumnarIterator implements AutoCloseable {
043    private static final Logger logger = Logger.getLogger(CSVIterator.class.getName());
044
045    public final static char SEPARATOR = ',';
046    public final static char QUOTE = '"';
047
048
049    private final CSVReader reader;
050    // We read numRows for idx from the CSVReader, so we need to keep track of whether the CSVReader read a header row
051    private int rowOffset = 1;
052
053    /**
054     * Builds a CSVIterator for the supplied Reader. Defaults to {@link CSVIterator#SEPARATOR} for the separator
055     * and {@link CSVIterator#QUOTE} for the quote.
056     * @param rdr The source to read.
057     */
058    public CSVIterator(Reader rdr) {
059        this(rdr, SEPARATOR, QUOTE);
060    }
061
062    /**
063     * Builds a CSVIterator for the supplied Reader.
064     * @param rdr The source to read.
065     * @param separator The separator character to use.
066     * @param quote The quote character to use.
067     */
068    public CSVIterator(Reader rdr, char separator, char quote) {
069        this(rdr, separator, quote, Collections.emptyList());
070    }
071
072    /**
073     * Builds a CSVIterator for the supplied URI. Defaults to {@link CSVIterator#SEPARATOR} for the separator
074     * and {@link CSVIterator#QUOTE} for the quote.
075     * @param dataFile The source to read.
076     * @throws IOException thrown if the file is not readable in some way.
077     */
078    public CSVIterator(URI dataFile) throws IOException {
079        this(Files.newBufferedReader(Paths.get(dataFile)));
080    }
081
082    /**
083     * Builds a CSVIterator for the supplied URI.
084     * @param dataFile The source to read.
085     * @param separator The separator character to use.
086     * @param quote The quote character to use.
087     * @throws IOException thrown if the file is not readable in some way.
088     */
089    public CSVIterator(URI dataFile, char separator, char quote) throws IOException {
090        this(Files.newBufferedReader(Paths.get(dataFile)), separator, quote);
091    }
092
093    /**
094     * Builds a CSVIterator for the supplied URI.
095     * @param dataFile The source to read.
096     * @param separator The separator character to use.
097     * @param quote The quote character to use.
098     * @param fields The headers to use.
099     * @throws IOException thrown if the file is not readable in some way.
100     */
101    public CSVIterator(URI dataFile, char separator, char quote, String[] fields) throws IOException {
102        this(Files.newBufferedReader(Paths.get(dataFile)), separator, quote, Arrays.asList(fields));
103    }
104
105    /**
106     * Builds a CSVIterator for the supplied URI.
107     * @param dataFile The source to read.
108     * @param separator The separator character to use.
109     * @param quote The quote character to use.
110     * @param fields The headers to use.
111     * @throws IOException thrown if the file is not readable in some way.
112     */
113    public CSVIterator(URI dataFile, char separator, char quote, List<String> fields) throws IOException {
114        this(Files.newBufferedReader(Paths.get(dataFile)), separator, quote, fields);
115    }
116
117    /**
118     * Builds a CSVIterator for the supplied Reader. If headers is null, read the headers from the csv file.
119     * @param rdr The source to read.
120     * @param separator The separator character to use.
121     * @param quote The quote character to use.
122     * @param fields The headers to use.
123     */
124    public CSVIterator(Reader rdr, char separator, char quote, String[] fields) {
125        this(rdr, separator, quote, fields == null ? null : Arrays.asList(fields));
126    }
127
128    /**
129     * Builds a CSVIterator for the supplied Reader. If headers is null, read the headers from the csv file.
130     * @param rdr The source to read.
131     * @param separator The separator character to use.
132     * @param quote The quote character to use.
133     * @param fields The headers to use.
134     */
135    public CSVIterator(Reader rdr, char separator, char quote, List<String> fields) {
136        reader = new CSVReaderBuilder(rdr).withCSVParser(new RFC4180ParserBuilder().withSeparator(separator).withQuoteChar(quote).build()).build();
137        try {
138            if (fields == null || fields.isEmpty()) {
139                String[] inducedHeader = reader.readNext();
140                if(inducedHeader == null) {
141                    logger.warning("Given an empty CSV");
142                } else {
143                    this.fields = Collections.unmodifiableList(Arrays.asList(inducedHeader));
144                    rowOffset++;
145                }
146            } else {
147                this.fields = Collections.unmodifiableList(fields);
148            }
149        } catch (CsvValidationException | IOException e) {
150            try {
151                reader.close();
152            } catch (IOException e2) {
153                logger.log(Level.WARNING, "Error closing reader in another error", e2);
154            }
155            throw new IllegalArgumentException("Error reading file caused by: " + e.getMessage());
156        }
157    }
158
159    /**
160     * Zips together the headers and the line into a Map.
161     * @param headers The field headers.
162     * @param line The extracted line.
163     * @param rowNum The row number used for error messages.
164     * @return A map from header to value.
165     */
166    private static Map<String,String> zip(List<String> headers, String[] line, long rowNum) {
167        if (headers.size() != line.length) {
168            throw new IllegalArgumentException("On row " + rowNum + " headers has " + headers.size() + " elements, current line has " + line.length + " elements.");
169        }
170
171        Map<String,String> map = new HashMap<>();
172        for (int i = 0; i < headers.size(); i++) {
173            map.put(headers.get(i),line[i]);
174        }
175        return map;
176    }
177
178    @Override
179    protected Optional<Row> getRow() {
180        try {
181            String[] rawRow = reader.readNext();
182            if(rawRow != null) {
183                if(reader.getRecordsRead() % 50_000 == 0) {
184                    logger.info(String.format("Read %d records on %d lines", reader.getRecordsRead(), reader.getLinesRead()));
185                }
186                return Optional.of(new Row(reader.getRecordsRead() - rowOffset,
187                        fields,
188                        zip(fields, rawRow, reader.getRecordsRead())));
189            } else {
190                try {
191                    reader.close();
192                } catch (IOException e) {
193                    logger.log(Level.WARNING, "Error closing reader at end of file", e);
194                }
195                return Optional.empty();
196            }
197        } catch (CsvValidationException | IOException e) {
198            long linesRead = reader.getLinesRead();
199            long recordsRead = reader.getRecordsRead();
200            try {
201                reader.close();
202            } catch (IOException e2) {
203                logger.log(Level.WARNING, "Error closing reader in another error", e2);
204            }
205            throw new IllegalArgumentException(String.format("Error reading CSV on record %d, row %d", recordsRead, linesRead), e);
206        }
207    }
208
209    @Override
210    public void close() throws IOException{
211        if(reader != null) {
212            reader.close();
213        }
214    }
215}
