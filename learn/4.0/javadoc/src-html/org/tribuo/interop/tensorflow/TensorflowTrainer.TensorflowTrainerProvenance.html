052/**
053 * Trainer for Tensorflow. Expects the underlying Tensorflow graph to have specific placeholders and
054 * targets listed below.
055 *
056 * <ul>
057 * <li>{@link TensorflowModel#INPUT_NAME} - the input minibatch.</li>
058 * <li>{@link TensorflowModel#OUTPUT_NAME} - the predicted output.</li>
059 * <li>{@link TensorflowTrainer#TARGET} - the output to predict.</li>
060 * <li>{@link TensorflowTrainer#TRAIN} - the train function to run (usually a single step of SGD).</li>
061 * <li>{@link TensorflowTrainer#TRAINING_LOSS} - the loss tensor to extract for logging.</li>
062 * <li>{@link TensorflowTrainer#EPOCH} - the current epoch number, used for gradient scaling.</li>
063 * <li>{@link TensorflowTrainer#IS_TRAINING} - a boolean placeholder to turn on dropout or other training specific functionality.</li>
064 * <li>{@link TensorflowTrainer#INIT} - the function to initialise the graph.</li>
065 * </ul>
066 *
067 * This trainer only works with graphs setup for minibatches. To recover single example training just use a batch size of 1.
068 * <p>
069 * This trainer uses the serialisation functionality in {@link TensorflowUtil}, as opposed to a SavedModel or a checkpoint.
070 * <p>
071 * N.B. Tensorflow support is experimental and may change without a major version bump.
072 */
073public final class TensorflowTrainer<T extends Output<T>> implements Trainer<T> {
074
075    private static final Logger logger = Logger.getLogger(TensorflowTrainer.class.getName());
076
077    public static final String TARGET = "target";
078    public static final String TRAIN = "train";
079    public static final String TRAINING_LOSS = "training_loss";
080    public static final String EPOCH = "epoch";
081    public static final String IS_TRAINING = "is_training";
082    public static final String INIT = "init";
083
084    @Config(mandatory=true,description="Path to the protobuf containing the graph.")
085    private Path graphPath;
086
087    private byte[] graphDef;
088
089    @Config(description="Test time batch size.")
090    private int testBatchSize = 16;
091
092    @Config(mandatory=true,description="Feature extractor.")
093    private ExampleTransformer<T> exampleTransformer;
094
095    @Config(mandatory=true,description="Response extractor.")
096    private OutputTransformer<T> outputTransformer;
097
098    @Config(description="Minibatch size.")
099    private int minibatchSize = 1;
100
101    @Config(description="Number of SGD epochs to run.")
102    private int epochs = 5;
103
104    @Config(description="Logging interval to print out the loss.")
105    private int loggingInterval = 100;
106
107    private int trainInvocationCounter = 0;
108
109    /**
110     * for olcut.
111     */
112    private TensorflowTrainer() {}
113
114    /**
115     * Constructs a Trainer for a tensorflow graph.
116     * @param graphPath The path to the graph protobuf. Must have the targets and placeholders specified above.
117     * @param exampleTransformer The example transformer to convert a Tribuo {@link Example} into a {@link Tensor}.
118     * @param outputTransformer The output transformer to convert a Tribuo {@link Output} into a {@link Tensor} and back. This encodes the output type.
119     * @param minibatchSize The minibatch size to use in training.
120     * @param epochs The number of SGD epochs to run.
121     * @param testBatchSize The minibatch size to use at test time.
122     * @throws IOException If the graphPath is invalid or failed to load.
123     */
124    public TensorflowTrainer(Path graphPath, ExampleTransformer<T> exampleTransformer, OutputTransformer<T> outputTransformer, int minibatchSize, int epochs, int testBatchSize) throws IOException {
125        this.graphPath = graphPath;
126        this.exampleTransformer = exampleTransformer;
127        this.outputTransformer = outputTransformer;
128        this.minibatchSize = minibatchSize;
129        this.epochs = epochs;
130        this.testBatchSize = testBatchSize;
131        postConfig();
132    }
133
134    /**
135     * Constructs a Trainer for a tensorflow graph.
136     * @param graphDef The graph definition as a byte array. Must have the targets and placeholders specified above.
137     * @param exampleTransformer The example transformer to convert a Tribuo {@link Example} into a {@link Tensor}.
138     * @param outputTransformer The output transformer to convert a Tribuo {@link Output} into a {@link Tensor} and back. This encodes the output type.
139     * @param minibatchSize The minibatch size to use in training.
140     * @param epochs The number of SGD epochs to run.
141     * @param testBatchSize The minibatch size to use at test time.
142     */
143    public TensorflowTrainer(byte[] graphDef, ExampleTransformer<T> exampleTransformer, OutputTransformer<T> outputTransformer, int minibatchSize, int epochs, int testBatchSize) {
144        this.graphPath = null;
145        this.graphDef = graphDef;
146        this.exampleTransformer = exampleTransformer;
147        this.outputTransformer = outputTransformer;
148        this.minibatchSize = minibatchSize;
149        this.epochs = epochs;
150        this.testBatchSize = testBatchSize;
151    }
152
153    @Override
154    public void postConfig() throws IOException {
155        graphDef = Files.readAllBytes(graphPath);
156    }
157
158    @Override
159    public Model<T> train(Dataset<T> examples, Map<String,Provenance> runProvenance) {
160        ImmutableFeatureMap featureMap = examples.getFeatureIDMap();
161        ImmutableOutputInfo<T> outputInfo = examples.getOutputIDInfo();
162        ArrayList<Example<T>> batch = new ArrayList<>();
163        trainInvocationCounter++;
164
165        try (Graph graph = new Graph();
166             Session session = new Session(graph);
167             Tensor<?> isTraining = Tensor.create(true)) {
168            // Load in the graph definition
169            graph.importGraphDef(graphDef);
170
171            // Initialises the parameters.
172            session.runner().addTarget(INIT).run();
173            logger.info("Initialised the model parameters");
174
175            int interval = 0;
176            for (int i = 0; i < epochs; i++) {
177                logger.log(Level.INFO,"Starting epoch " + i);
178                Tensor<?> epoch = Tensor.create(i);
179                for (int j = 0; j < examples.size(); j += minibatchSize) {
180                    batch.clear();
181                    for (int k = j; k < (j+ minibatchSize) && k < examples.size(); k++) {
182                        batch.add(examples.getExample(k));
183                    }
184                    //logger.info("Batch = " + batch.size());
185                    Tensor<?> input = exampleTransformer.transform(batch,featureMap);
186                    Tensor<?> target = outputTransformer.transform(batch,outputInfo);
187                    Tensor<?> loss = session.runner()
188                            .feed(TensorflowModel.INPUT_NAME, input)
189                            .feed(TARGET, target)
190                            .feed(EPOCH, epoch)
191                            .feed(IS_TRAINING, isTraining)
192                            .addTarget(TRAIN)
193                            .fetch(TRAINING_LOSS)
194                            .run().get(0);
195                    if (interval % loggingInterval == 0) {
196                        logger.log(Level.INFO, "Training loss = " + loss.floatValue());
197                    }
198                    input.close();
199                    target.close();
200                    loss.close();
201                    interval++;
202                }
203                epoch.close();
204            }
205
206            //System.out.println("After training");
207            //TensorflowModel.print(session);
208
209            // This call **must** happen before the trainedGraphDef is generated.
210            TensorflowUtil.annotateGraph(graph,session);
211
212            byte[] trainedGraphDef = graph.toGraphDef();
213
214            Map<String,Object> tensorMap = TensorflowUtil.serialise(graph,session);
215
216            ModelProvenance modelProvenance = new ModelProvenance(TensorflowModel.class.getName(), OffsetDateTime.now(), examples.getProvenance(), getProvenance(), runProvenance);
217            TensorflowModel<T> tfModel = new TensorflowModel<>("tf-model", modelProvenance, featureMap,
218                    outputInfo, trainedGraphDef, tensorMap, testBatchSize, exampleTransformer, outputTransformer);
219
220            return tfModel;
221        } catch (TensorFlowException e) {
222            logger.log(Level.SEVERE, "TensorFlow threw an error", e);
223            throw new IllegalStateException(e);
224        }
225    }

227    @Override
228    public String toString() {
229        String path = graphPath==null?"":graphPath.toString();
230        return "TensorflowTrainer(graphPath="+path+",exampleTransformer="
231                +exampleTransformer.toString()+",outputTransformer="+outputTransformer.toString()
232                +",minibatchSize="+ minibatchSize +",epochs="+ epochs +")";
233    }

237        return trainInvocationCounter;

242        return new TensorflowTrainerProvenance(this);

245    public static final class TensorflowTrainerProvenance extends SkeletalTrainerProvenance {
246        private static final long serialVersionUID = 1L;
247
248        public static final String GRAPH_HASH = "graph-hash";
249        public static final String GRAPH_LAST_MOD = "graph-last-modified";
250
251        private final HashProvenance graphHash;
252        private final DateTimeProvenance graphLastModified;
253
254        <T extends Output<T>> TensorflowTrainerProvenance(TensorflowTrainer<T> host) {
255            super(host);
256            // instance parameters
257            if (host.graphPath != null) {
258                this.graphHash = new HashProvenance(DEFAULT_HASH_TYPE,GRAPH_HASH,ProvenanceUtil.hashResource(DEFAULT_HASH_TYPE,host.graphPath));
259                this.graphLastModified = new DateTimeProvenance(GRAPH_LAST_MOD, OffsetDateTime.ofInstant(Instant.ofEpochMilli(host.graphPath.toFile().lastModified()), ZoneId.systemDefault()));
260            } else {
261                this.graphHash = new HashProvenance(DEFAULT_HASH_TYPE,GRAPH_HASH,hashArray(DEFAULT_HASH_TYPE,host.graphDef));
262                this.graphLastModified = new DateTimeProvenance(GRAPH_LAST_MOD, OffsetDateTime.now());
263            }
264        }
265
266        public TensorflowTrainerProvenance(Map<String,Provenance> map) {
267            this(extractTFProvenanceInfo(map));
268        }
269
270        private TensorflowTrainerProvenance(ExtractedInfo info) {
271            super(info);
272            this.graphHash = (HashProvenance) info.instanceValues.get(GRAPH_HASH);
273            this.graphLastModified = (DateTimeProvenance) info.instanceValues.get(GRAPH_LAST_MOD);
274        }
275
276        /**
277         * Hashes a byte array using the specified {@link com.oracle.labs.mlrg.olcut.provenance.ProvenanceUtil.HashType}.
278         * @param hashType The type of hash to perform.
279         * @param input The input array.
280         * @return A hexadecimal string representation of the hash.
281         */
282        private static String hashArray(com.oracle.labs.mlrg.olcut.provenance.ProvenanceUtil.HashType hashType, byte[] input) {
283            java.security.MessageDigest md = hashType.getDigest();
284            md.update(input);
285            return ProvenanceUtil.bytesToHexString(md.digest());
286        }
287
288        @Override
289        public Map<String, PrimitiveProvenance<?>> getInstanceValues() {
290            Map<String, PrimitiveProvenance<?>> map = super.getInstanceValues();
291
292            map.put(graphHash.getKey(),graphHash);
293            map.put(graphLastModified.getKey(),graphLastModified);
294
295            return map;
296        }
297
298        protected static ExtractedInfo extractTFProvenanceInfo(Map<String,Provenance> map) {
299            ExtractedInfo info = SkeletalTrainerProvenance.extractProvenanceInfo(map);
300            info.instanceValues.put(GRAPH_HASH,ObjectProvenance.checkAndExtractProvenance(map,GRAPH_HASH,HashProvenance.class,TensorflowTrainerProvenance.class.getSimpleName()));
301            info.instanceValues.put(GRAPH_LAST_MOD,ObjectProvenance.checkAndExtractProvenance(map,GRAPH_LAST_MOD,DateTimeProvenance.class,TensorflowTrainerProvenance.class.getSimpleName()));
302            return info;
303        }
304    }
