017package org.tribuo.interop.tensorflow;
018
019import com.oracle.labs.mlrg.olcut.config.Config;
020import com.oracle.labs.mlrg.olcut.config.PropertyException;
021import com.oracle.labs.mlrg.olcut.provenance.ConfiguredObjectProvenance;
022import com.oracle.labs.mlrg.olcut.provenance.impl.ConfiguredObjectProvenanceImpl;
023import org.tribuo.Example;
024import org.tribuo.Feature;
025import org.tribuo.ImmutableFeatureMap;
026import org.tribuo.Output;
027import org.tribuo.math.la.SparseVector;
028import org.tribuo.math.la.VectorTuple;
029import org.tensorflow.Tensor;
030
031import java.util.List;
032
033/**
034 * Image transformer. Assumes the feature id numbers are linearised ids of the form
035 * [0,0,0] = 0, [1,0,0] = 1, ..., [i,0,0] = i, [0,1,0] = i+1, ..., [i,j,0] = i*j, ...
036 * [0,0,1] = (i*j)+1, ..., [i,j,k] = i*j*k.
037 */
038public class ImageTransformer<T extends Output<T>> implements ExampleTransformer<T> {
039    private static final long serialVersionUID = 1L;
040
041    @Config(mandatory=true,description="Image width.")
042    private int width;
043
044    @Config(mandatory=true,description="Image height.")
045    private int height;
046
047    @Config(mandatory=true,description="Number of channels.")
048    private int channels;
049
050    /**
051     * For olcut.
052     */
053    private ImageTransformer() {}
054
055    public ImageTransformer(int width, int height, int channels) {
056        if (width < 1 || height < 1 || channels < 1) {
057            throw new IllegalArgumentException("Inputs must be positive integers, found ["+width+","+height+","+channels+"]");
058        }
059        this.width = width;
060        this.height = height;
061        this.channels = channels;
062    }
063
064    @Override
065    public void postConfig() {
066        if (width < 1 || height < 1 || channels < 1) {
067            throw new PropertyException("","Inputs must be positive integers, found ["+width+","+height+","+channels+"]");
068        }
069    }
070
071    /**
072     * Transform implicitly pads unseen values with zero.
073     * @param example The example to transform.
074     * @param featureIDMap The feature id mapping to use.
075     * @return A 3d tensor, (width, height, channels) for this example.
076     */
077    @Override
078    public Tensor<?> transform(Example<T> example, ImmutableFeatureMap featureIDMap) {
079        float[][][][] image = new float[1][][][];
080        image[0] = innerTransform(example,featureIDMap);
081        return Tensor.create(image);
082    }
083
084    /**
085     * Actually performs the transformation. Implicitly pads unseen values
086     * with zero.
087     * @param example The example to transform.
088     * @param featureIDMap The feature id mapping to use.
089     * @return A 3d array, (width,height,channels) representing the example.
090     */
091    float[][][] innerTransform(Example<T> example, ImmutableFeatureMap featureIDMap) {
092        float[][][] image = new float[width][height][channels];
093
094        for (Feature f : example) {
095            int id = featureIDMap.getID(f.getName());
096            int curWidth = id % width;
097            int curHeight = (id / width) % height;
098            int curChannel = id / (width * height);
099            image[curWidth][curHeight][curChannel] = (float) f.getValue();
100        }
101
102        return image;
103    }
104
105    /**
106     * Actually performs the transformation. Implicitly pads unseen values
107     * with zero.
108     * @param vector The vector to transform.
109     * @return A 3d array, (width,height,channels) representing the vector.
110     */
111    float[][][] innerTransform(SparseVector vector) {
112        float[][][] image = new float[width][height][channels];
113
114        for (VectorTuple f : vector) {
115            int id = f.index;
116            int curWidth = id % width;
117            int curHeight = (id / width) % height;
118            int curChannel = id / (width * height);
119            image[curWidth][curHeight][curChannel] = (float) f.value;
120        }
121
122        return image;
123    }
124
125    /**
126     * Transform implicitly pads unseen values with zero.
127     * <p>
128     * Converts a batch of examples into a Tensor.
129     * @param examples The examples to transform.
130     * @param featureIDMap The feature id mapping to use.
131     * @return A 4d tensor, (batch-id, width, height, channels) for this example.
132     */
133    @Override
134    public Tensor<?> transform(List<Example<T>> examples, ImmutableFeatureMap featureIDMap) {
135        float[][][][] image = new float[examples.size()][][][];
136
137        int i = 0;
138        for (Example<T> example : examples) {
139            image[i] = innerTransform(example,featureIDMap);
140            i++;
141        }
142
143        return Tensor.create(image);
144    }
145
146    @Override
147    public Tensor<?> transform(SparseVector vector) {
148        float[][][][] image = new float[1][][][];
149        image[0] = innerTransform(vector);
150        return Tensor.create(image);
151    }
152
153    @Override
154    public Tensor<?> transform(List<SparseVector> vectors) {
155        float[][][][] image = new float[vectors.size()][][][];
156
157        int i = 0;
158        for (SparseVector vector : vectors) {
159            image[i] = innerTransform(vector);
160            i++;
161        }
162
163        return Tensor.create(image);
164    }
165
166    @Override
167    public String toString() {
168        return "ImageTransformer(width="+width+",height="+height+",channels="+channels+")";
169    }
170
171    @Override
172    public ConfiguredObjectProvenance getProvenance() {
173        return new ConfiguredObjectProvenanceImpl(this,"ExampleTransformer");
174    }
175}
