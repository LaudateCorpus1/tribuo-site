017package org.tribuo.classification.baseline;
018
019import com.oracle.labs.mlrg.olcut.config.Config;
020import com.oracle.labs.mlrg.olcut.config.PropertyException;
021import com.oracle.labs.mlrg.olcut.provenance.Provenance;
022import org.tribuo.Dataset;
023import org.tribuo.ImmutableFeatureMap;
024import org.tribuo.ImmutableOutputInfo;
025import org.tribuo.Model;
026import org.tribuo.MutableOutputInfo;
027import org.tribuo.Trainer;
028import org.tribuo.classification.Label;
029import org.tribuo.provenance.ModelProvenance;
030import org.tribuo.provenance.TrainerProvenance;
031import org.tribuo.provenance.impl.TrainerProvenanceImpl;
032
033import java.time.OffsetDateTime;
034import java.util.Map;
035
036/**
037 * A trainer for simple baseline classifiers. Use this only for comparison purposes, if you can't beat these
038 * baselines, your ML system doesn't work.
039 */
040public final class DummyClassifierTrainer implements Trainer<Label> {
041
042    public enum DummyType { STRATIFIED, MOST_FREQUENT, UNIFORM, CONSTANT }
043
044    @Config(mandatory = true,description="Type of dummy classifier.")
045    private DummyType dummyType;
046
047    @Config(description="Label to use for the constant classifier.")
048    private String constantLabel;
049
050    @Config(description="Seed for the RNG.")
051    private long seed = 1L;
052
053    private int invocationCount = 0;
054
055    private DummyClassifierTrainer() {}
056
057    @Override
058    public void postConfig() {
059        if ((dummyType == DummyType.CONSTANT) && (constantLabel == null)) {
060            throw new PropertyException("","constantLabel","Please supply a label string when using the type CONSTANT.");
061        }
062    }
063
064    @Override
065    public Model<Label> train(Dataset<Label> examples, Map<String, Provenance> instanceProvenance) {
066        ModelProvenance provenance = new ModelProvenance(DummyClassifierModel.class.getName(), OffsetDateTime.now(), examples.getProvenance(), getProvenance(), instanceProvenance);
067        ImmutableFeatureMap featureMap = examples.getFeatureIDMap();
068        invocationCount++;
069        switch (dummyType) {
070            case CONSTANT:
071                MutableOutputInfo<Label> labelInfo = examples.getOutputInfo().generateMutableOutputInfo();
072                Label constLabel = new Label(constantLabel);
073                labelInfo.observe(constLabel);
074                return new DummyClassifierModel(provenance,featureMap,labelInfo.generateImmutableOutputInfo(),constLabel);
075            case MOST_FREQUENT: {
076                ImmutableOutputInfo<Label> immutableLabelInfo = examples.getOutputIDInfo();
077                return new DummyClassifierModel(provenance, featureMap, immutableLabelInfo);
078            }
079            case UNIFORM:
080            case STRATIFIED: {
081                ImmutableOutputInfo<Label> immutableLabelInfo = examples.getOutputIDInfo();
082                return new DummyClassifierModel(provenance, featureMap, immutableLabelInfo, dummyType, seed);
083            }
084            default:
085                throw new IllegalStateException("Unknown dummyType " + dummyType);
086        }
087    }
088
089    @Override
090    public int getInvocationCount() {
091        return invocationCount;
092    }
093
094    @Override
095    public String toString() {
096        switch (dummyType) {
097            case CONSTANT:
098                return "DummyClassifierTrainer(dummyType="+dummyType+",constantLabel="+constantLabel+")";
099            case MOST_FREQUENT: {
100                return "DummyClassifierTrainer(dummyType="+dummyType+")";
101            }
102            case UNIFORM:
103            case STRATIFIED: {
104                return "DummyClassifierTrainer(dummyType="+dummyType+",seed="+seed+")";
105            }
106            default:
107                return "DummyClassifierTrainer(dummyType="+dummyType+")";
108        }
109    }
110
111    @Override
112    public TrainerProvenance getProvenance() {
113        return new TrainerProvenanceImpl(this);
114    }
115
116    /**
117     * Creates a trainer which creates models which return random labels sampled from the training label distribution.
118     * @param seed The RNG seed to use.
119     * @return A classification trainer.
120     */
121    public static DummyClassifierTrainer createStratifiedTrainer(long seed) {
122        DummyClassifierTrainer trainer = new DummyClassifierTrainer();
123        trainer.dummyType = DummyType.STRATIFIED;
124        trainer.seed = seed;
125        return trainer;
126    }
127
128    /**
129     * Creates a trainer which creates models which return a fixed label.
130     * @param constantLabel The label to return.
131     * @return A classification trainer.
132     */
133    public static DummyClassifierTrainer createConstantTrainer(String constantLabel) {
134        DummyClassifierTrainer trainer = new DummyClassifierTrainer();
135        trainer.dummyType = DummyType.CONSTANT;
136        trainer.constantLabel = constantLabel;
137        return trainer;
138    }
139
140    /**
141     * Creates a trainer which creates models which return random labels sampled uniformly from the labels seen at training time.
142     * @param seed The RNG seed to use.
143     * @return A classification trainer.
144     */
145    public static DummyClassifierTrainer createUniformTrainer(long seed) {
146        DummyClassifierTrainer trainer = new DummyClassifierTrainer();
147        trainer.dummyType = DummyType.UNIFORM;
148        trainer.seed = seed;
149        return trainer;
150    }
151
152    /**
153     * Creates a trainer which creates models which return a fixed label, the one which was most frequent in the training data.
154     * @return A classification trainer.
155     */
156    public static DummyClassifierTrainer createMostFrequentTrainer() {
157        DummyClassifierTrainer trainer = new DummyClassifierTrainer();
158        trainer.dummyType = DummyType.MOST_FREQUENT;
159        return trainer;
160    }
161}
