039/**
040 * An implementation of a viterbi model.
041 */
042public class ViterbiModel extends SequenceModel<Label> {
043
044    private static final long serialVersionUID = 1L;
045
046    public enum ScoreAggregation {
047        ADD, MULTIPLY
048    }
049
050    private final Model<Label> model;
051
052    private final LabelFeatureExtractor labelFeatureExtractor;
053
054    /**
055     * Specifies the maximum number of candidate paths to keep track of. In general, this number
056     * should be higher than the number of possible classifications at any given point in the
057     * sequence. This guarantees that highest-possible scoring sequence will be returned. If,
058     * however, the number of possible classifications is quite high and/or you are concerned about
059     * throughput performance, then you may want to reduce the number of candidate paths to
060     * maintain.
061     */
062    private final int stackSize;
063
064    /**
065     * Specifies the score aggregation algorithm.
066     */
067    private final ScoreAggregation scoreAggregation;
068
069    ViterbiModel(String name, ModelProvenance description,
070                        Model<Label> model, LabelFeatureExtractor labelFeatureExtractor, int stackSize, ScoreAggregation scoreAggregation) {
071        super(name, description, model.getFeatureIDMap(), model.getOutputIDInfo());
072        this.model = model;
073        this.labelFeatureExtractor = labelFeatureExtractor;
074        this.stackSize = stackSize;
075        this.scoreAggregation = scoreAggregation;
076    }
077
078    @Override
079    public List<List<Prediction<Label>>> predict(SequenceDataset<Label> examples) {
080        List<List<Prediction<Label>>> predictions = new ArrayList<>();
081        for (SequenceExample<Label> e : examples) {
082            predictions.add(predict(e));
083        }
084        return predictions;
085    }
086
087    @Override
088    public List<Prediction<Label>> predict(SequenceExample<Label> examples) {
089        if (stackSize == 1) {
090            List<Label> labels = new ArrayList<>();
091            List<Prediction<Label>> returnValues = new ArrayList<>();
092            for (Example<Label> example : examples) {
093                List<Feature> labelFeatures = extractFeatures(labels);
094                example.addAll(labelFeatures);
095                Prediction<Label> prediction = model.predict(example);
096                labels.add(prediction.getOutput());
097                returnValues.add(prediction);
098            }
099            return returnValues;
100        } else {
101            return viterbi(examples);
102        }
103
104    }
105
106    private List<Feature> extractFeatures(List<Label> labels) {
107        List<Feature> labelFeatures = new ArrayList<>();
108        for (Feature labelFeature : labelFeatureExtractor.extractFeatures(labels, 1.0)) {
109            int id = featureIDMap.getID(labelFeature.getName());
110            if (id > -1) {
111                labelFeatures.add(labelFeature);
112            }
113        }
114        return labelFeatures;
115    }
116
117    /**
118     * This implementation of Viterbi requires at most stackSize * sequenceLength calls to the
119     * classifier. If this proves to be too expensive, then consider using a smaller stack size.
120     *
121     * @param examples a sequence-worth of features. Each {@code List<Feature>} in features should correspond to
122     *                 all of the features for a given element in a sequence to be classified.
123     * @return a list of Predictions - one for each member of the sequence.
124     * @see LabelFeatureExtractor
125     */
126    private List<Prediction<Label>> viterbi(SequenceExample<Label> examples) {
127        // find the best paths through the label lattice
128        Collection<Path> paths = null;
129        int[] numUsed = new int[examples.size()];
130        int i = 0;
131        for (Example<Label> example : examples) {
132            // if this is the first instance, start new paths for each label
133            if (paths == null) {
134                paths = new ArrayList<>();
135                Prediction<Label> prediction = this.model.predict(example);
136                numUsed[i] = prediction.getNumActiveFeatures();
137                Map<String, Label> distribution = prediction.getOutputScores();
138                for (Label label : this.getTopLabels(distribution)) {
139                    paths.add(new Path(label, label.getScore(), null));
140                }
141            } else {
142                // for later instances, find the best previous path for each label
143                Map<Label, Path> maxPaths = new HashMap<>();
144                for (Path path : paths) {
145                    Example<Label> clonedExample = example.copy();
146                    List<Label> previousLabels = new ArrayList<>(path.labels);
147                    List<Feature> labelFeatures = extractFeatures(previousLabels);
148                    clonedExample.addAll(labelFeatures);
149                    Prediction<Label> prediction = this.model.predict(clonedExample);
150                    // TODO this isn't quite correct as it includes label features.
151                    numUsed[i] = prediction.getNumActiveFeatures();
152                    Map<String, Label> distribution = prediction.getOutputScores();
153
154                    for (Label label : this.getTopLabels(distribution)) {
155                        double labelScore = label.getScore();
156                        double score = this.scoreAggregation == ScoreAggregation.ADD ? path.score + labelScore : path.score * labelScore;
157                        Path maxPath = maxPaths.get(label);
158                        if (maxPath == null || score > maxPath.score) {
159                            maxPaths.put(label, new Path(label, score, path));
160                        }
161                    }
162                }
163                paths = maxPaths.values();
164            }
165            i++;
166        }
167
168        Path maxPath = Collections.max(paths);
169
170        ArrayList<Prediction<Label>> output = new ArrayList<>();
171
172        for (int j = 0; j < examples.size(); j++) {
173            Example<Label> e = examples.get(j);
174            output.add(new Prediction<>(maxPath.labels.get(j), numUsed[j], e));
175        }
176
177        return output;
178    }
179
180    protected List<Label> getTopLabels(Map<String, Label> distribution) {
181        return getTopLabels(distribution, this.stackSize);
182    }
183
184    protected static List<Label> getTopLabels(Map<String, Label> distribution, int stackSize) {
185        return distribution.values().stream().sorted(Comparator.comparingDouble(Label::getScore).reversed()).limit(stackSize)
186                .collect(Collectors.toList());
187        // get just the labels that fit within the stack
188    }
189
190    private static class Path implements Comparable<Path> {
191
192        public final double score;
193
194        public final Path parent;
195
196        public final List<Label> labels;
197
198        public Path(Label label, double score, Path parent) {
199            this.score = score;
200            this.parent = parent;
201            this.labels = new ArrayList<>();
202            if (this.parent != null) {
203                this.labels.addAll(this.parent.labels);
204            }
205            this.labels.add(label);
206        }
207
208        @Override
209        public int compareTo(Path that) {
210            return Double.compare(this.score, that.score);
211        }
212
213    }
214
215    public int getStackSize() {
216        return stackSize;
217    }
218
219    public ScoreAggregation getScoreAggregation() {
220        return scoreAggregation;
221    }
222
223    @Override
224    public Map<String, List<Pair<String, Double>>> getTopFeatures(int n) {
225        return model.getTopFeatures(n);
226    }
