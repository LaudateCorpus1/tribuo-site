017package org.tribuo.classification.sequence.example;
018
019import org.tribuo.Example;
020import org.tribuo.Feature;
021import org.tribuo.classification.Label;
022import org.tribuo.classification.LabelFactory;
023import org.tribuo.impl.ListExample;
024import org.tribuo.provenance.SimpleDataSourceProvenance;
025import org.tribuo.sequence.MutableSequenceDataset;
026import org.tribuo.sequence.SequenceExample;
027
028import java.time.OffsetDateTime;
029import java.util.ArrayList;
030import java.util.List;
031
032/**
033 * A data generator for smoke testing sequence label models.
034 */
035public final class SequenceDataGenerator {
036
037    private static final LabelFactory labelFactory = new LabelFactory();
038
039    private SequenceDataGenerator() { }
040
041    public static MutableSequenceDataset<Label> generateGorillaDataset(int numCopies) {
042        List<SequenceExample<Label>> examples = new ArrayList<>();
043
044        for (int i = 0; i < numCopies; i++) {
045            examples.add(generateGorillaA());
046            examples.add(generateGorillaB());
047        }
048
049        return new MutableSequenceDataset<>(examples, new SimpleDataSourceProvenance("ExampleSequenceDataset", OffsetDateTime.now(),labelFactory),labelFactory);
050    }
051
052    public static SequenceExample<Label> generateGorillaA() {
053        //"The silverback gorilla is angry"
054        List<Example<Label>> examples = new ArrayList<>();
055
056        Example<Label> the = new ListExample<>(new Label("O"));
057        the.add(new Feature("A",1.0));
058        the.add(new Feature("B",1.0));
059        the.add(new Feature("W=the",1.0));
060        examples.add(the);
061
062        Example<Label> silverback = new ListExample<>(new Label("Monkey"));
063        silverback.add(new Feature("C",1.0));
064        silverback.add(new Feature("D",1.0));
065        silverback.add(new Feature("W=silverback",1.0));
066        examples.add(silverback);
067
068        Example<Label> gorilla = new ListExample<>(new Label("Monkey"));
069        gorilla.add(new Feature("D",1.0));
070        gorilla.add(new Feature("E",1.0));
071        gorilla.add(new Feature("W=gorilla",1.0));
072        examples.add(gorilla);
073
074        Example<Label> is = new ListExample<>(new Label("O"));
075        is.add(new Feature("B",1.0));
076        is.add(new Feature("W=is",1.0));
077        examples.add(is);
078
079        Example<Label> angry = new ListExample<>(new Label("Status"));
080        angry.add(new Feature("F",1.0));
081        angry.add(new Feature("G",1.0));
082        angry.add(new Feature("W=angry",1.0));
083        examples.add(angry);
084
085        return new SequenceExample<>(examples);
086    }
087
088    public static SequenceExample<Label> generateGorillaB() {
089        //"That is one angry looking gorilla"
090        List<Example<Label>> examples = new ArrayList<>();
091
092        Example<Label> that = new ListExample<>(new Label("O"));
093        that.add(new Feature("A",1.0));
094        that.add(new Feature("B",1.0));
095        that.add(new Feature("W=that",1.0));
096        examples.add(that);
097
098        Example<Label> is = new ListExample<>(new Label("O"));
099        is.add(new Feature("B",1.0));
100        is.add(new Feature("W=is",1.0));
101        examples.add(is);
102
103        Example<Label> one = new ListExample<>(new Label("O"));
104        one.add(new Feature("B",1.0));
105        one.add(new Feature("H",1.0));
106        one.add(new Feature("W=one",1.0));
107        examples.add(one);
108
109        Example<Label> angry = new ListExample<>(new Label("Status"));
110        angry.add(new Feature("F",1.0));
111        angry.add(new Feature("G",1.0));
112        angry.add(new Feature("W=angry",1.0));
113        examples.add(angry);
114
115        Example<Label> looking = new ListExample<>(new Label("O"));
116        looking.add(new Feature("I",1.0));
117        looking.add(new Feature("J",1.0));
118        looking.add(new Feature("W=looking",1.0));
119        examples.add(looking);
120
121        Example<Label> gorilla = new ListExample<>(new Label("Monkey"));
122        gorilla.add(new Feature("D",1.0));
123        gorilla.add(new Feature("E",1.0));
124        gorilla.add(new Feature("W=gorilla",1.0));
125        examples.add(gorilla);
126
127        return new SequenceExample<>(examples);
128    }
129
130    /**
131     * This generates a sequence example with features that are unused by the training data.
132     * @return A {@link SequenceExample} which is invalid in the context of the Gorilla example data.
133     */
134    public static SequenceExample<Label> generateInvalidExample() {
135        //"invalid example"
136        List<Example<Label>> examples = new ArrayList<>();
137
138        Example<Label> invalid = new ListExample<>(new Label("O"));
139        invalid.add(new Feature("1",1.0));
140        invalid.add(new Feature("2",1.0));
141        invalid.add(new Feature("W=invalid",1.0));
142        examples.add(invalid);
143
144        Example<Label> example = new ListExample<>(new Label("O"));
145        example.add(new Feature("3",1.0));
146        example.add(new Feature("2",1.0));
147        example.add(new Feature("W=example",1.0));
148        examples.add(example);
149
150        return new SequenceExample<>(examples);
151    }
152
153    /**
154     * This generates a sequence example where the first example has no features.
155     * @return A {@link SequenceExample} which is invalid as one example contains no features.
156     */
157    public static SequenceExample<Label> generateOtherInvalidExample() {
158        //"invalid example"
159        List<Example<Label>> examples = new ArrayList<>();
160
161        Example<Label> invalid = new ListExample<>(new Label("O"));
162        examples.add(invalid);
163
164        Example<Label> example = new ListExample<>(new Label("O"));
165        example.add(new Feature("3",1.0));
166        example.add(new Feature("2",1.0));
167        example.add(new Feature("W=example",1.0));
168        examples.add(example);
169
170        return new SequenceExample<>(examples);
171    }
172
173    /**
174     * This generates a sequence example with no examples.
175     * @return A {@link SequenceExample} which is invalid as it contains no examples.
176     */
177    public static SequenceExample<Label> generateEmptyExample() {
178        List<Example<Label>> examples = new ArrayList<>();
179        return new SequenceExample<>(examples);
180    }
181}
